20  import com.agorapulse.micronaut.aws.kinesis.annotation.KinesisClient;
21  import com.agorapulse.micronaut.aws.kinesis.annotation.PartitionKey;
22  import com.agorapulse.micronaut.aws.kinesis.annotation.SequenceNumber;
23  import com.agorapulse.micronaut.aws.kinesis.annotation.Stream;
24  import com.agorapulse.micronaut.aws.util.ConfigurationUtil;
25  import com.amazonaws.services.kinesis.AmazonKinesis;
26  import com.amazonaws.services.kinesis.model.PutRecordResult;
27  import com.amazonaws.services.kinesis.model.PutRecordsRequestEntry;
28  import com.amazonaws.services.kinesis.model.ResourceNotFoundException;
29  import com.fasterxml.jackson.core.JsonProcessingException;
30  import com.fasterxml.jackson.databind.ObjectMapper;
31  import groovy.transform.Undefined;
32  import io.micronaut.aop.MethodInterceptor;
33  import io.micronaut.aop.MethodInvocationContext;

40  import javax.inject.Singleton;

44  @Singleton
45  @Requires(classes = AmazonKinesis.class)
46  public class KinesisClientIntroduction implements MethodInterceptor<Object, Object> {
47  
48      private static final String KEY = "key";
49      private static final String SEQUENCE = "sequence";
50  
51      private static class RecordArguments {
52          Argument<?> partitionKey;
53          Argument<?> data;
54          Argument<?> sequenceNumber;
55  
56          boolean isValid() {
57              return partitionKey != null && data != null;
58          }
59  
60      }
61  
62      private final BeanContext beanContext;
63      private final ObjectMapper objectMapper;
64  
65      public KinesisClientIntroduction(BeanContext beanContext, ObjectMapper objectMapper) {
66          this.beanContext = beanContext;
67          this.objectMapper = objectMapper;
68      }
69  
70      @Override
71      public Object intercept(MethodInvocationContext<Object, Object> context) {
72          AnnotationValue<KinesisClient> clientAnnotationValue = context.getAnnotation(KinesisClient.class);
73  
74          if (clientAnnotationValue == null) {
75              throw new IllegalStateException("Invocation beanContext is missing required annotation KinesisClient");
76          }
77  
78          String configurationName = clientAnnotationValue.getValue(String.class).orElse(ConfigurationUtil.DEFAULT_CONFIGURATION_NAME);
79          KinesisService service = beanContext.getBean(
80              KinesisService.class,
81              ConfigurationUtil.isDefaultConfigurationName(configurationName) ? null : Qualifiers.byName(configurationName)
82          );
83  
84          String streamName = clientAnnotationValue.get(KinesisClient.Constants.STREAM, String.class).orElse(null);
85  
86          AnnotationValue<Stream> streamAnnotationValue = context.getAnnotation(Stream.class);
87          if (streamAnnotationValue != null) {
88              streamName = streamAnnotationValue.getRequiredValue(String.class);
89          }
90  
91          if (streamName == null || Undefined.STRING.equals(streamName)) {
92              streamName = service.getDefaultStreamName();
93          }
94  
95          try {
96              return doIntercept(context, service, streamName);
97          } catch (ResourceNotFoundException ignored) {
98              service.createStream(streamName);
99              return doIntercept(context, service, streamName);
100         }
101     }
102 
103     private Object doIntercept(MethodInvocationContext<Object, Object> context, KinesisService service, String streamName) {
104         if (context.getArguments().length == 1) {
105             Argument arg = context.getArguments()[0];
106             Class argType = arg.getType();
107             Object param = context.getParameters().get(arg.getName()).getValue();
108 
109             if (Event.class.isAssignableFrom(argType)) {
110                 return service.putEvent(streamName, (Event) param);
111             }
112 
113             if (PutRecordsRequestEntry.class.isAssignableFrom(argType)) {
114                 return service.putRecords(streamName, Collections.singletonList((PutRecordsRequestEntry) param));
115             }
116 
117             if (Iterable.class.isAssignableFrom(argType) && arg.hasTypeVariables() && arg.getFirstTypeVariable().isPresent()) {
118                 Class iterableType = arg.getFirstTypeVariable().get().getType();
119 
120                 if (Event.class.isAssignableFrom(iterableType)) {
121                     return service.putEvents(streamName, toList((Iterable<Event>) param));
122                 }
123 
124                 if (PutRecordsRequestEntry.class.isAssignableFrom(iterableType)) {
125                     return service.putRecords(streamName, toList((Iterable<PutRecordsRequestEntry>) param));
126                 }
127 
128                 return service.putRecords(streamName, toJsonPutRequests(param));
129             }
130 
131             if (argType.isArray()) {
132                 if (Event.class.isAssignableFrom(argType.getComponentType())) {
133                     return service.putEvents(streamName, toList(Arrays.asList((Event[]) param)));
134                 }
135                 if (PutRecordsRequestEntry.class.isAssignableFrom(argType.getComponentType())) {
136                     return service.putRecords(streamName, toList(Arrays.asList((PutRecordsRequestEntry[]) param)));
137                 }
138                 if (byte.class.equals(argType.getComponentType())) {
139                     return service.putRecord(streamName, createDefaultParititonKey(), (byte[]) param);
140                 }
141                 return service.putRecords(streamName, toJsonPutRequests(param));
142             }
143 
144             if (CharSequence.class.isAssignableFrom(argType)) {
145                 return service.putRecord(streamName, createDefaultParititonKey(), ((CharSequence) param).toString());
146             }
147 
148             return sendJson(service, streamName, createDefaultParititonKey(), param, null);
149         }
150 
151         if (context.getArguments().length == 2 || context.getArguments().length == 3) {
152             RecordArguments recordArguments = findArguments(context.getArguments());
153 
154             String partitionKey = String.valueOf(context.getParameters().get(recordArguments.partitionKey.getName()).getValue());
155             String sequenceNumber = recordArguments.sequenceNumber == null ? null : String.valueOf(context.getParameters().get(recordArguments.sequenceNumber.getName()).getValue());
156             Object data = context.getParameters().get(recordArguments.data.getName()).getValue();
157             Class<?> dataType = recordArguments.data.getType();
158 
159             if (String.class.isAssignableFrom(dataType)) {
160                 return service.putRecord(streamName, partitionKey, (String) data, sequenceNumber);
161             }
162 
163             if (dataType.isArray() && byte.class.equals(dataType.getComponentType())) {
164                 return service.putRecord(streamName, partitionKey, (byte[]) data, sequenceNumber);
165             }
166 
167             return sendJson(service, streamName, partitionKey, data, sequenceNumber);
168         }
169 
170         throw new UnsupportedOperationException("Cannot implement method " + context.getExecutableMethod());
171     }
172 
173     private List<PutRecordsRequestEntry> toJsonPutRequests(Object param) {
174         Iterable objects = param instanceof Iterable ? (Iterable) param : Arrays.asList((Object[]) param);
175         List<PutRecordsRequestEntry> ret = new ArrayList<>();
176         for (Object o : objects) {
177             ret.add(new PutRecordsRequestEntry().withData(ByteBuffer.wrap(json(o))).withPartitionKey(createDefaultParititonKey()));
178         }
179         return ret;
180     }
181 
182     private byte[] json(Object data) {
183         try {
184             return objectMapper.writeValueAsBytes(data);
185         } catch (JsonProcessingException e) {
186             throw new IllegalArgumentException("Failed to marshal " + data + " to JSON", e);
187         }
188     }
189 
190     private PutRecordResult sendJson(KinesisService service, String streamName, String partitionKey, Object data, String sequenceNumber) {
191         return service.putRecord(streamName, partitionKey, json(data), sequenceNumber);
192     }
193 
194     private String createDefaultParititonKey() {
195         return UUID.randomUUID().toString();
196     }
197 
198     private RecordArguments findArguments(Argument[] arguments) {
199         RecordArguments names = new RecordArguments();
200         for(Argument<?> argument : arguments) {
201             if (argument.isAnnotationPresent(PartitionKey.class) || argument.getName().toLowerCase().contains(KEY)) {
202                 names.partitionKey = argument;
203                 continue;
204             }
205             if (argument.isAnnotationPresent(SequenceNumber.class) || argument.getName().toLowerCase().contains(SEQUENCE)) {
206                 names.sequenceNumber = argument;
207                 continue;
208             }
209             names.data = argument;
210         }
211 
212         if (!names.isValid()) {
213             throw new UnsupportedOperationException("Method needs to have at least one argument annotated with @PartitionKey and one without any annotation or it needs to have single argument implementing Event");
214         }
215 
216         return names;
217     }
218 
219     private static <T> List<T> toList(Iterable<T> iterable) {
220         if (iterable instanceof List) {
221             return (List<T>) iterable;
222         }
223 
224         List<T> ret = new ArrayList<>();
225         iterable.forEach(ret::add);
226         return ret;
227     }
