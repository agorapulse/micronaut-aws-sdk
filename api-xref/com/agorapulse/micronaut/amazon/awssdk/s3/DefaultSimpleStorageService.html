18  package com.agorapulse.micronaut.amazon.awssdk.s3;
19  
20  import io.micronaut.http.multipart.PartData;
21  import org.reactivestreams.Publisher;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import reactor.core.publisher.Flux;
25  import reactor.core.publisher.Mono;
26  import software.amazon.awssdk.awscore.exception.AwsServiceException;
27  import software.amazon.awssdk.core.ResponseInputStream;
28  import software.amazon.awssdk.core.exception.SdkClientException;
29  import software.amazon.awssdk.core.sync.RequestBody;
30  import software.amazon.awssdk.services.s3.S3Client;
31  import software.amazon.awssdk.services.s3.model.*;
32  import software.amazon.awssdk.services.s3.presigner.S3Presigner;
33  import software.amazon.awssdk.services.s3.presigner.model.PresignedGetObjectRequest;
34  import software.amazon.awssdk.services.s3.presigner.model.PresignedPutObjectRequest;
35  import software.amazon.awssdk.utils.IoUtils;
36  import software.amazon.awssdk.utils.Md5Utils;
37  
38  import java.io.File;
39  import java.io.IOException;
40  import java.io.InputStream;
41  import java.time.Duration;
42  import java.time.Instant;
43  import java.util.Date;
44  import java.util.List;
45  import java.util.function.Consumer;
46  import java.util.stream.Collectors;
47  
48  public class DefaultSimpleStorageService implements SimpleStorageService {
49  
50      private static final Logger LOGGER = LoggerFactory.getLogger(DefaultSimpleStorageService.class);
51      private final S3Client s3;
52      private final S3Presigner presigner;
53      private final String defaultBucketName;
54  
55      public DefaultSimpleStorageService(String bucket, S3Client s3, S3Presigner presigner) {
56          this.s3 = s3;
57          defaultBucketName = bucket;
58          this.presigner = presigner;
59      }
60  
61      @Override
62      public String getDefaultBucketName() {
63          return defaultBucketName;
64      }
65  
66      @Override
67      public void createBucket(String bucketName) {
68          s3.createBucket(b -> b.bucket(bucketName));
69      }
70  
71      @Override
72      public void deleteBucket(String bucketName) {
73          s3.deleteBucket(b -> b.bucket(bucketName));
74      }
75  
76      @Override
77      public boolean deleteFile(String bucketName, String key) {
78          try {
79              s3.deleteObject(b -> b.bucket(bucketName).key(key));
80              return true;
81          } catch (AwsServiceException | SdkClientException e) {
82              LOGGER.warn(String.format("Exception deleting object %s/%s", bucketName, key), e);
83              return false;
84          }
85      }
86  
87      @Override
88      public boolean deleteFiles(String bucketName, String prefix) {
89          if (prefix.split("/").length >= 2) {
90              throw new IllegalArgumentException("Multiple delete are only allowed in sub/sub directories: " + prefix);
91          }
92  
93          return Flux.from(listObjectSummaries(bucketName, prefix))
94              .map(o -> deleteFile(bucketName, o.key()))
95              .onErrorResume(throwable -> {
96                  LOGGER.warn(String.format("Exception deleting objects in %s/%s", bucketName, prefix), throwable);
97                  return Mono.just(false);
98              })
99              .filter(r -> !r)
100             .count()
101             .blockOptional()
102             .map(Long::intValue)
103             .orElse(0) == 0;
104     }
105 
106     @Override
107     public boolean exists(String bucketName, String key) {
108         if (key == null || key.isEmpty()) {
109             return false;
110         }
111         try {
112             return getObject(bucketName, key) != null;
113         } catch (AwsServiceException | SdkClientException e) {
114             LOGGER.warn(String.format("Exception obtaining object existence %s/%s", bucketName, key), e);
115             return false;
116         }
117     }
118 
119     @Override
120     public GetObjectResponse getObject(String bucketName, String key) {
121         try (ResponseInputStream<GetObjectResponse> stream = s3.getObject(b -> b.bucket(bucketName).key(key))) {
122             return stream.response();
123         } catch (IOException e) {
124             throw new IllegalArgumentException("Exception closing stream", e);
125         }
126     }
127 
128     @Override
129     public File getFile(String bucketName, String key, File localFile) {
130         s3.getObject(b -> b.bucket(bucketName).key(key), localFile.toPath());
131         return localFile;
132     }
133 
134     @Override
135     public List<String> listBucketNames() {
136         return s3.listBuckets().buckets().stream().map(Bucket::name).collect(Collectors.toList());
137     }
138 
139     @Override
140     public Publisher<ListObjectsV2Response> listObjects(String bucketName, String prefix) {
141         return Flux.fromIterable(s3.listObjectsV2Paginator(b -> b.bucket(bucketName).prefix(prefix)));
142     }
143 
144     @Override
145     public String generatePresignedUrl(String bucketName, String key, Date expirationDate) {
146         PresignedGetObjectRequest request = presigner.presignGetObject(b ->
147             b.getObjectRequest(r -> r.bucket(bucketName).key(key)).signatureDuration(Duration.between(Instant.now(), expirationDate.toInstant()))
148         );
149         return request.url().toExternalForm();
150     }
151 
152     @Override
153     public String generateUploadUrl(String bucketName, String key, Date expirationDate) {
154         PresignedPutObjectRequest request = presigner.presignPutObject(b ->
155             b.putObjectRequest(r -> r.bucket(bucketName).key(key)).signatureDuration(Duration.between(Instant.now(), expirationDate.toInstant()))
156         );
157         return request.url().toExternalForm();
158     }
159 
160     @Override
161     public String storeInputStream(String bucketName, String path, InputStream input, Consumer<PutObjectRequest.Builder> additionalConfig) {
162         try {
163             s3.putObject(
164                 b -> {
165                     additionalConfig.accept(b);
166                     b.bucket(bucketName).key(path);
167                 },
168                 RequestBody.fromBytes(IoUtils.toByteArray(input))
169             );
170             return s3.utilities().getUrl(b -> b.bucket(bucketName).key(path)).toExternalForm();
171         } catch (AwsServiceException | IOException exception) {
172             return "";
173         }
174     }
175 
176     @Override
177     public String storeFile(String bucketName, String path, File file, Consumer<PutObjectRequest.Builder> additionalConfig) {
178         try {
179             s3.putObject(
180                 b -> {
181                     b.bucket(bucketName).key(path);
182                     additionalConfig.accept(b);
183                 },
184                 RequestBody.fromFile(file)
185             );
186             return s3.utilities().getUrl(b -> b.bucket(bucketName).key(path)).toExternalForm();
187         } catch (AwsServiceException exception) {
188             return "";
189         }
190     }
191 
192     @Override
193     public String storeMultipartFile(String bucketName, String path, PartData partData, Consumer<PutObjectRequest.Builder> additionalConfig) throws IOException {
194         byte[] bytes = partData.getBytes();
195         return storeInputStream(bucketName, path, partData.getInputStream(), b -> {
196             b.contentLength(Integer.valueOf(bytes.length).longValue());
197             b.contentMD5(Md5Utils.md5AsBase64(bytes));
198             partData.getContentType().ifPresent(t -> b.contentType(t.getName()));
199             additionalConfig.accept(b);
200         });
201     }
202 
203     @Override
204     public String moveObject(String sourceBucketName, String sourceKey, String destinationBucketName, String destinationKey) {
205         try {
206             CopyObjectRequest.Builder request = CopyObjectRequest.builder()
207                 .copySource(sourceBucketName + "/" + sourceKey)
208                 .destinationBucket(destinationBucketName)
209                 .destinationKey(destinationKey);
210 
211             s3.copyObject(request.build());
212 
213             GetObjectAclResponse acl = s3.getObjectAcl(b -> b.bucket(sourceBucketName).key(sourceKey));
214             s3.putObjectAcl(b -> {
215                 b.bucket(destinationBucketName).key(destinationKey).accessControlPolicy(p -> p.owner(acl.owner()).grants(acl.grants()));
216             });
217 
218             s3.deleteObject(b -> b.bucket(sourceBucketName).key(sourceKey));
219 
220             return s3.utilities().getUrl(b -> b.bucket(destinationBucketName).key(destinationKey)).toExternalForm();
221         } catch (AwsServiceException e) {
222             LOGGER.error(String.format("Exception moving object %s/%s to %s/%s", sourceBucketName, sourceKey, destinationBucketName, destinationKey), e);
223             return null;
224         }
225     }
