46  @Singleton
47  @Requires(
48      property = "aws.kinesis"
49  )
50  public class KinesisListenerMethodProcessor implements ExecutableMethodProcessor<KinesisListener>, ApplicationEventListener<ShutdownEvent>, Closeable {
51  
52      private static final Logger LOGGER = LoggerFactory.getLogger(KinesisListenerMethodProcessor.class);
53  
54      private static class StringListener implements BiConsumer<String, KinesisClientRecord> {
55  
56          private final ExecutableMethod method;
57          private final Object bean;
58  
59          public StringListener(ExecutableMethod method, Object bean) {
60              this.method = method;
61              this.bean = bean;
62          }
63  
64          @Override
65          public void accept(String s, KinesisClientRecord record) {
66              method.invoke(bean, s);
67          }
68      }
69  
70      private static class RecordListener implements BiConsumer<String, KinesisClientRecord> {
71  
72          private final ExecutableMethod method;
73          private final Object bean;
74  
75          RecordListener(ExecutableMethod method, Object bean) {
76              this.method = method;
77              this.bean = bean;
78          }
79  
80          @Override
81          public void accept(String s, KinesisClientRecord record) {
82              method.invoke(bean, record);
83          }
84      }
85  
86      private static class StringAndRecordListener implements BiConsumer<String, KinesisClientRecord> {
87  
88          private final ExecutableMethod method;
89          private final Object bean;
90  
91          StringAndRecordListener(ExecutableMethod method, Object bean) {
92              this.method = method;
93              this.bean = bean;
94          }
95  
96          @Override
97          public void accept(String s, KinesisClientRecord record) {
98              method.invoke(bean, s, record);
99          }
100     }
101 
102     private static class EventListener implements BiConsumer<String, KinesisClientRecord> {
103 
104         private final ExecutableMethod method;
105         private final Object bean;
106         private final ObjectMapper mapper;
107 
108         private final String consumerFilterKey;
109 
110         EventListener(ExecutableMethod method, Object bean, ObjectMapper mapper, String consumerFilterKey) {
111             this.method = method;
112             this.bean = bean;
113             this.mapper = mapper;
114             this.consumerFilterKey = consumerFilterKey;
115         }
116 
117         @Override
118         public void accept(String s, KinesisClientRecord record) {
119             Class type = method.getArguments()[0].getType();
120             try {
121                 Object event = mapper.readValue(s, type);
122 
123                 if (event instanceof Event && StringUtils.isNotEmpty(consumerFilterKey)) {
124                     String eventKey = ((Event) event).getConsumerFilterKey();
125                     if (!consumerFilterKey.equals(eventKey)) {
126                         LOGGER.info("Ignoring event because expected consumer filter key {} is not equal to the event's filter key {}: {}", consumerFilterKey, eventKey, s);
127                         return;
128                     }
129                 }
130 
131                 method.invoke(bean, event);
132             } catch (IOException e) {
133                 throw new IllegalArgumentException("Failed to unmarshall string " + s + " as type " + type);
134             }
135         }
136     }
137 
138     private static class EventAndRecordListener implements BiConsumer<String, KinesisClientRecord> {
139 
140         private final ExecutableMethod method;
141         private final Object bean;
142         private final ObjectMapper mapper;
143 
144         EventAndRecordListener(ExecutableMethod method, Object bean, ObjectMapper mapper) {
145             this.method = method;
146             this.bean = bean;
147             this.mapper = mapper;
148         }
149 
150         @Override
151         public void accept(String s, KinesisClientRecord record) {
152             Class type = method.getArguments()[0].getType();
153             try {
154                 method.invoke(bean, mapper.readValue(s, type), record);
155             } catch (IOException e) {
156                 throw new IllegalArgumentException("Failed to unmarshall string " + s + " as type " + type);
157             }
158         }
159     }
160 
161     private final BeanContext beanContext;
162     private final ObjectMapper objectMapper;
163     private final KinesisWorkerFactory kinesisWorkerFactory;
164 
165     private final String consumerFilterKey;
166 
167     private final ConcurrentHashMap<String, KinesisWorker> workers = new ConcurrentHashMap<>();
168 
169     public KinesisListenerMethodProcessor(
170         BeanContext beanContext,
171         ObjectMapper objectMapper,
172         KinesisWorkerFactory kinesisWorkerFactory,
173         @Value("${aws.kinesis.consumer-filter-key:}") String consumerFilterKey
174     ) {
175         this.beanContext = beanContext;
176         this.objectMapper = objectMapper;
177         this.kinesisWorkerFactory = kinesisWorkerFactory;
178         this.consumerFilterKey = consumerFilterKey;
179     }
180 
181     @Override
182     public void process(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
183         Argument[] arguments = method.getArguments();
184 
185         if (arguments.length > 2) {
186             throw new IllegalArgumentException("Method must implement at most two arguments");
187         }
188 
189         if (arguments.length < 1) {
190             throw new IllegalArgumentException("Method must implement at least one arguments");
191         }
192 
193         if (arguments.length == 2 && !KinesisClientRecord.class.isAssignableFrom(arguments[1].getType())) {
194             throw new IllegalArgumentException("Second argument must be Record");
195         }
196 
197         io.micronaut.context.Qualifier<Object> qualifer = beanDefinition
198             .getAnnotationTypeByStereotype(Qualifier.class)
199             .map(type -> Qualifiers.byAnnotation(beanDefinition, type))
200             .orElse(null);
201 
202         Class beanType = beanDefinition.getBeanType();
203         Object bean = beanContext.getBean(beanType, qualifer);
204 
205         BiConsumer<String, KinesisClientRecord> consumer = createConsumer(method, bean);
206 
207         String configurationName = method.getValue(KinesisListener.class, String.class).get();
208 
209         KinesisWorker worker = workers.computeIfAbsent(
210             configurationName, key -> {
211                 KinesisWorker w = kinesisWorkerFactory.create(getKinesisConfiguration(key));
212 
213                 LOGGER.info("Kinesis worker for configuration {} created", key);
214 
215                 w.start();
216 
217                 return w;
218             }
219         );
220 
221         worker.addConsumer(consumer);
222 
223         LOGGER.info("Kinesis listener for method {} declared in {} registered", method, beanDefinition.getBeanType());
224     }
225 
226     @Override
227     public void onApplicationEvent(ShutdownEvent event) {
228         close();
229     }
230 
231     @Override
232     public void close() {
233         workers.values().forEach(KinesisWorker::shutdown);
234     }
235 
236     private KinesisClientConfiguration getKinesisConfiguration(String key) {
237         try {
238             return beanContext.getBean(KinesisClientConfiguration.class, Qualifiers.byName(key));
239         } catch (NoSuchBeanException ignored) {
240             LOGGER.error("Cannot setup listener Kinesis listener, application name is missing. Configuration for Kinesis client with name '{}' is missing", key);
241             return null;
242         }
243     }
244 
245     private BiConsumer<String, KinesisClientRecord> createConsumer(ExecutableMethod method, Object bean) {
246         Argument[] arguments = method.getArguments();
247 
248         if (arguments.length == 2) {
249             if (CharSequence.class.isAssignableFrom(arguments[0].getType())) {
250                 return new StringAndRecordListener(method, bean);
251             }
252             return new EventAndRecordListener(method, bean, objectMapper);
253         }
254 
255         if (CharSequence.class.isAssignableFrom(arguments[0].getType())) {
256             return new StringListener(method, bean);
257         }
258 
259         if (KinesisClientRecord.class.isAssignableFrom(arguments[0].getType())) {
260             return new RecordListener(method, bean);
261         }
262 
263         return new EventListener(method, bean, objectMapper, consumerFilterKey);
264     }
