18  package com.agorapulse.micronaut.amazon.awssdk.kinesis.worker;
19  
20  import io.micronaut.core.util.StringUtils;
21  import software.amazon.awssdk.services.cloudwatch.CloudWatchAsyncClient;
22  import software.amazon.awssdk.services.dynamodb.DynamoDbAsyncClient;
23  import software.amazon.awssdk.services.kinesis.KinesisAsyncClient;
24  import software.amazon.kinesis.common.ConfigsBuilder;
25  import software.amazon.kinesis.processor.ShardRecordProcessorFactory;
26  
27  import io.micronaut.core.annotation.NonNull;
28  import javax.validation.constraints.Min;
29  import javax.validation.constraints.NotEmpty;
30  import java.net.URI;
31  import java.util.Optional;
32  
33  /**
34   * Default configuration for Kinesis listener.
35   */
36  public abstract class KinesisClientConfiguration {
37  
38      private static final long DEFAULT_FAILOVER_TIME_MILLIS = 10000L;
39      private static final long DEFAULT_SHARD_SYNC_INTERVAL_MILLIS = 1000L;
40      private static final long DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS = 10000L;
41      private static final boolean DEFAULT_CLEANUP_LEASES_UPON_SHARDS_COMPLETION = true;
42      private static final long DEFAULT_TASK_BACKOFF_TIME_MILLIS = 500L;
43      private static final int DEFAULT_MAX_LEASES_FOR_WORKER = Integer.MAX_VALUE;
44      private static final int DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME = 1;
45      private static final int DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY = 10;
46      private static final int DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY = 10;
47      private static final int DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS = 50;
48  
49      @NotEmpty private final String applicationName;
50      @NotEmpty private final String workerId;
51      @NotEmpty private String stream;
52  
53      private String consumerFilterKey = "";
54      private String tableName;
55      private String kinesisEndpoint;
56      private String dynamoDBEndpoint;
57  
58      private boolean callProcessRecordsEvenForEmptyRecordList;
59      private boolean ignoreUnexpectedChildShards;
60      private boolean cleanupLeasesUponShardCompletion = DEFAULT_CLEANUP_LEASES_UPON_SHARDS_COMPLETION;
61      private boolean skipShardSyncAtWorkerInitializationIfLeasesExist;
62  
63      @Min(1L) private long failoverTimeMillis = DEFAULT_FAILOVER_TIME_MILLIS;
64      @Min(1L) private long shardSyncIntervalMillis = DEFAULT_SHARD_SYNC_INTERVAL_MILLIS;
65      @Min(1L) private long parentShardPollIntervalMillis = DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS;
66      @Min(1L) private long taskBackoffTimeMillis = DEFAULT_TASK_BACKOFF_TIME_MILLIS;
67  
68      private int maxLeasesForWorker = DEFAULT_MAX_LEASES_FOR_WORKER;
69      private int maxLeasesToStealAtOneTime = DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME;
70      private int initialLeaseTableReadCapacity = DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY;
71      private int initialLeaseTableWriteCapacity = DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY;
72      private int maxListShardsRetryAttempts = DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS;
73  
74      private Optional<Long> logWarningForTaskAfterMillis = Optional.empty();
75  
76      protected KinesisClientConfiguration(@NonNull String applicationName, @NonNull  String workerId, String consumerFilterKey) {
77          this.applicationName = applicationName;
78          this.workerId = workerId;
79          this.consumerFilterKey = consumerFilterKey;
80      }
81  
82      protected ConfigsBuilder getConfigsBuilder(
83          Optional<DynamoDbAsyncClient> amazonDynamoDB,
84          Optional<KinesisAsyncClient> amazonKinesis,
85          Optional<CloudWatchAsyncClient> amazonCloudWatch,
86          ShardRecordProcessorFactory recordProcessorFactory
87      ) {
88          ConfigsBuilder builder = new ConfigsBuilder(
89              stream,
90              applicationName,
91              StringUtils.isEmpty(kinesisEndpoint) ? amazonKinesis.orElseGet(() -> KinesisAsyncClient.builder().build()) : KinesisAsyncClient.builder().endpointOverride(URI.create(kinesisEndpoint)).build(),
92              StringUtils.isEmpty(dynamoDBEndpoint) ? amazonDynamoDB.orElseGet(() -> DynamoDbAsyncClient.builder().build()) : DynamoDbAsyncClient.builder().endpointOverride(URI.create(dynamoDBEndpoint)).build(),
93              amazonCloudWatch.orElseGet(() -> CloudWatchAsyncClient.builder().build()),
94              workerId,
95              recordProcessorFactory
96          );
97  
98          if (StringUtils.isNotEmpty(tableName)) {
99              builder.tableName(tableName);
100         } else if (StringUtils.isNotEmpty(consumerFilterKey)) {
101             builder.tableName(consumerFilterKey + "_" + stream);
102         } else {
103             builder.tableName(stream);
104         }
105 
106         if (failoverTimeMillis != DEFAULT_FAILOVER_TIME_MILLIS) {
107             builder.leaseManagementConfig().failoverTimeMillis(failoverTimeMillis);
108         }
109 
110         if (shardSyncIntervalMillis != DEFAULT_SHARD_SYNC_INTERVAL_MILLIS) {
111             builder.coordinatorConfig().shardConsumerDispatchPollIntervalMillis(shardSyncIntervalMillis);
112         }
113 
114         if (callProcessRecordsEvenForEmptyRecordList) {
115             builder.processorConfig().callProcessRecordsEvenForEmptyRecordList(true);
116         }
117 
118         if (parentShardPollIntervalMillis != DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS) {
119             builder.coordinatorConfig().parentShardPollIntervalMillis(parentShardPollIntervalMillis);
120         }
121 
122         if (!cleanupLeasesUponShardCompletion) {
123             builder.leaseManagementConfig().cleanupLeasesUponShardCompletion(false);
124         }
125 
126         if (ignoreUnexpectedChildShards) {
127             builder.leaseManagementConfig().ignoreUnexpectedChildShards(true);
128         }
129 
130         if (taskBackoffTimeMillis != DEFAULT_TASK_BACKOFF_TIME_MILLIS) {
131             builder.lifecycleConfig().taskBackoffTimeMillis(taskBackoffTimeMillis);
132         }
133 
134         if (maxLeasesForWorker != DEFAULT_MAX_LEASES_FOR_WORKER) {
135             builder.leaseManagementConfig().maxLeasesForWorker(maxLeasesForWorker);
136         }
137 
138         if (maxLeasesToStealAtOneTime != DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME) {
139             builder.leaseManagementConfig().maxLeasesToStealAtOneTime(maxLeasesToStealAtOneTime);
140         }
141 
142         if (initialLeaseTableReadCapacity != DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY) {
143             builder.leaseManagementConfig().initialLeaseTableReadCapacity(initialLeaseTableReadCapacity);
144         }
145 
146         if (initialLeaseTableWriteCapacity != DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY) {
147             builder.leaseManagementConfig().initialLeaseTableWriteCapacity(initialLeaseTableWriteCapacity);
148         }
149 
150         if (skipShardSyncAtWorkerInitializationIfLeasesExist) {
151             builder.coordinatorConfig().skipShardSyncAtWorkerInitializationIfLeasesExist(true);
152         }
153 
154         if (logWarningForTaskAfterMillis.isPresent()) {
155             builder.lifecycleConfig().logWarningForTaskAfterMillis(logWarningForTaskAfterMillis);
156         }
157 
158         if (maxListShardsRetryAttempts != DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS) {
159             builder.retrievalConfig().maxListShardsRetryAttempts(maxListShardsRetryAttempts);
160         }
161 
162         return builder;
163     }
164

166         this.stream = stream;
167     }

170         this.tableName = tableName;
171     }
172

174         this.consumerFilterKey = consumerFilterKey;
175     }
176

178         this.kinesisEndpoint = kinesisEndpoint;
179     }
180 
181     public void setDynamoDBEndpoint(String dynamoDBEndpoint) {
182         this.dynamoDBEndpoint = dynamoDBEndpoint;
183     }
184 
185     public void setCallProcessRecordsEvenForEmptyRecordList(boolean callProcessRecordsEvenForEmptyRecordList) {
186         this.callProcessRecordsEvenForEmptyRecordList = callProcessRecordsEvenForEmptyRecordList;
187     }
188

190         this.ignoreUnexpectedChildShards = ignoreUnexpectedChildShards;
191     }

199     }

206         this.shardSyncIntervalMillis = shardSyncIntervalMillis;
207     }

215     }

218         this.maxLeasesForWorker = maxLeasesForWorker;
219     }
220

222         this.maxLeasesToStealAtOneTime = maxLeasesToStealAtOneTime;
223     }
224 
225     public void setInitialLeaseTableReadCapacity(int initialLeaseTableReadCapacity) {
226         this.initialLeaseTableReadCapacity = initialLeaseTableReadCapacity;
227     }
228 
229     public void setInitialLeaseTableWriteCapacity(int initialLeaseTableWriteCapacity) {
230         this.initialLeaseTableWriteCapacity = initialLeaseTableWriteCapacity;
231     }
232 
233     public void setMaxListShardsRetryAttempts(int maxListShardsRetryAttempts) {
234         this.maxListShardsRetryAttempts = maxListShardsRetryAttempts;
235     }
236 
237     public void setLogWarningForTaskAfterMillis(Optional<Long> logWarningForTaskAfterMillis) {
238         this.logWarningForTaskAfterMillis = logWarningForTaskAfterMillis;
239     }

242         return applicationName;
243     }

246         return workerId;
247     }

250         return stream;
251     }
