18  package com.agorapulse.micronaut.amazon.awssdk.kinesis.worker;
19  
20  import io.micronaut.core.util.StringUtils;
21  import software.amazon.awssdk.services.cloudwatch.CloudWatchAsyncClient;
22  import software.amazon.awssdk.services.dynamodb.DynamoDbAsyncClient;
23  import software.amazon.awssdk.services.kinesis.KinesisAsyncClient;
24  import software.amazon.kinesis.common.ConfigsBuilder;
25  import software.amazon.kinesis.processor.ShardRecordProcessorFactory;
26  
27  import javax.annotation.Nonnull;
28  import javax.validation.constraints.Min;
29  import javax.validation.constraints.NotEmpty;
30  import java.net.URI;
31  import java.util.Optional;
32  
33  /**
34   * Default configuration for Kinesis listener.
35   */
36  public abstract class KinesisClientConfiguration {
37  
38      private static final long DEFAULT_FAILOVER_TIME_MILLIS = 10000L;
39      private static final long DEFAULT_SHARD_SYNC_INTERVAL_MILLIS = 1000L;
40      private static final long DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS = 10000L;
41      private static final boolean DEFAULT_CLEANUP_LEASES_UPON_SHARDS_COMPLETION = true;
42      private static final long DEFAULT_TASK_BACKOFF_TIME_MILLIS = 500L;
43      private static final int DEFAULT_MAX_LEASES_FOR_WORKER = Integer.MAX_VALUE;
44      private static final int DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME = 1;
45      private static final int DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY = 10;
46      private static final int DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY = 10;
47      private static final int DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS = 50;
48  
49      @NotEmpty private final String applicationName;
50      @NotEmpty private final String workerId;
51      @NotEmpty private String stream;
52  
53      private String tableName;
54      private String kinesisEndpoint;
55      private String dynamoDBEndpoint;
56  
57      private boolean callProcessRecordsEvenForEmptyRecordList;
58      private boolean ignoreUnexpectedChildShards;
59      private boolean cleanupLeasesUponShardCompletion = DEFAULT_CLEANUP_LEASES_UPON_SHARDS_COMPLETION;
60      private boolean skipShardSyncAtWorkerInitializationIfLeasesExist;
61  
62      @Min(1L) private long failoverTimeMillis = DEFAULT_FAILOVER_TIME_MILLIS;
63      @Min(1L) private long shardSyncIntervalMillis = DEFAULT_SHARD_SYNC_INTERVAL_MILLIS;
64      @Min(1L) private long parentShardPollIntervalMillis = DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS;
65      @Min(1L) private long taskBackoffTimeMillis = DEFAULT_TASK_BACKOFF_TIME_MILLIS;
66  
67      private int maxLeasesForWorker = DEFAULT_MAX_LEASES_FOR_WORKER;
68      private int maxLeasesToStealAtOneTime = DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME;
69      private int initialLeaseTableReadCapacity = DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY;
70      private int initialLeaseTableWriteCapacity = DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY;
71      private int maxListShardsRetryAttempts = DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS;
72  
73      private Optional<Long> logWarningForTaskAfterMillis = Optional.empty();
74  
75      protected KinesisClientConfiguration(@Nonnull String applicationName, @Nonnull  String workerId) {
76          this.applicationName = applicationName;
77          this.workerId = workerId;
78      }
79  
80      protected ConfigsBuilder getConfigsBuilder(
81          Optional<DynamoDbAsyncClient> amazonDynamoDB,
82          Optional<KinesisAsyncClient> amazonKinesis,
83          Optional<CloudWatchAsyncClient> amazonCloudWatch,
84          ShardRecordProcessorFactory recordProcessorFactory
85      ) {
86          ConfigsBuilder builder = new ConfigsBuilder(
87              stream,
88              applicationName,
89              StringUtils.isEmpty(kinesisEndpoint) ? amazonKinesis.orElseGet(() -> KinesisAsyncClient.builder().build()) : KinesisAsyncClient.builder().endpointOverride(URI.create(kinesisEndpoint)).build(),
90              StringUtils.isEmpty(dynamoDBEndpoint) ? amazonDynamoDB.orElseGet(() -> DynamoDbAsyncClient.builder().build()) : DynamoDbAsyncClient.builder().endpointOverride(URI.create(dynamoDBEndpoint)).build(),
91              amazonCloudWatch.orElseGet(() -> CloudWatchAsyncClient.builder().build()),
92              workerId,
93              recordProcessorFactory
94          );
95  
96          if (StringUtils.isNotEmpty(tableName)) {
97              builder.tableName(tableName);
98          }
99  
100         if (failoverTimeMillis != DEFAULT_FAILOVER_TIME_MILLIS) {
101             builder.leaseManagementConfig().failoverTimeMillis(failoverTimeMillis);
102         }
103 
104         if (shardSyncIntervalMillis != DEFAULT_SHARD_SYNC_INTERVAL_MILLIS) {
105             builder.coordinatorConfig().shardConsumerDispatchPollIntervalMillis(shardSyncIntervalMillis);
106         }
107 
108         if (callProcessRecordsEvenForEmptyRecordList) {
109             builder.processorConfig().callProcessRecordsEvenForEmptyRecordList(true);
110         }
111 
112         if (parentShardPollIntervalMillis != DEFAULT_PARENT_SHARD_POLL_INTERVAL_MILLIS) {
113             builder.coordinatorConfig().parentShardPollIntervalMillis(parentShardPollIntervalMillis);
114         }
115 
116         if (!cleanupLeasesUponShardCompletion) {
117             builder.leaseManagementConfig().cleanupLeasesUponShardCompletion(false);
118         }
119 
120         if (ignoreUnexpectedChildShards) {
121             builder.leaseManagementConfig().ignoreUnexpectedChildShards(true);
122         }
123 
124         if (taskBackoffTimeMillis != DEFAULT_TASK_BACKOFF_TIME_MILLIS) {
125             builder.lifecycleConfig().taskBackoffTimeMillis(taskBackoffTimeMillis);
126         }
127 
128         if (maxLeasesForWorker != DEFAULT_MAX_LEASES_FOR_WORKER) {
129             builder.leaseManagementConfig().maxLeasesForWorker(maxLeasesForWorker);
130         }
131 
132         if (maxLeasesToStealAtOneTime != DEFAULT_MAX_LEASES_TO_STEAL_AT_ONE_TIME) {
133             builder.leaseManagementConfig().maxLeasesToStealAtOneTime(maxLeasesToStealAtOneTime);
134         }
135 
136         if (initialLeaseTableReadCapacity != DEFAULT_INITIAL_LEASE_TABLE_READ_CAPACITY) {
137             builder.leaseManagementConfig().initialLeaseTableReadCapacity(initialLeaseTableReadCapacity);
138         }
139 
140         if (initialLeaseTableWriteCapacity != DEFAULT_INITIAL_LEASE_TABLE_WRITE_CAPACITY) {
141             builder.leaseManagementConfig().initialLeaseTableWriteCapacity(initialLeaseTableWriteCapacity);
142         }
143 
144         if (skipShardSyncAtWorkerInitializationIfLeasesExist) {
145             builder.coordinatorConfig().skipShardSyncAtWorkerInitializationIfLeasesExist(true);
146         }
147 
148         if (logWarningForTaskAfterMillis.isPresent()) {
149             builder.lifecycleConfig().logWarningForTaskAfterMillis(logWarningForTaskAfterMillis);
150         }
151 
152         if (maxListShardsRetryAttempts != DEFAULT_MAX_LIST_SHARDS_RETRY_ATTEMPTS) {
153             builder.retrievalConfig().maxListShardsRetryAttempts(maxListShardsRetryAttempts);
154         }
155 
156         return builder;
157     }
158 
159     public void setStream(String stream) {
160         this.stream = stream;
161     }
162 
163     public void setTableName(String tableName) {
164         this.tableName = tableName;
165     }
166 
167     public void setKinesisEndpoint(String kinesisEndpoint) {
168         this.kinesisEndpoint = kinesisEndpoint;
169     }
170 
171     public void setDynamoDBEndpoint(String dynamoDBEndpoint) {
172         this.dynamoDBEndpoint = dynamoDBEndpoint;
173     }
174 
175     public void setCallProcessRecordsEvenForEmptyRecordList(boolean callProcessRecordsEvenForEmptyRecordList) {
176         this.callProcessRecordsEvenForEmptyRecordList = callProcessRecordsEvenForEmptyRecordList;
177     }
178 
179     public void setIgnoreUnexpectedChildShards(boolean ignoreUnexpectedChildShards) {
180         this.ignoreUnexpectedChildShards = ignoreUnexpectedChildShards;
181     }
182 
183     public void setCleanupLeasesUponShardCompletion(boolean cleanupLeasesUponShardCompletion) {
184         this.cleanupLeasesUponShardCompletion = cleanupLeasesUponShardCompletion;
185     }
186 
187     public void setSkipShardSyncAtWorkerInitializationIfLeasesExist(boolean skipShardSyncAtWorkerInitializationIfLeasesExist) {
188         this.skipShardSyncAtWorkerInitializationIfLeasesExist = skipShardSyncAtWorkerInitializationIfLeasesExist;
189     }
190 
191     public void setFailoverTimeMillis(long failoverTimeMillis) {
192         this.failoverTimeMillis = failoverTimeMillis;
193     }
194 
195     public void setShardSyncIntervalMillis(long shardSyncIntervalMillis) {
196         this.shardSyncIntervalMillis = shardSyncIntervalMillis;
197     }
198 
199     public void setParentShardPollIntervalMillis(long parentShardPollIntervalMillis) {
200         this.parentShardPollIntervalMillis = parentShardPollIntervalMillis;
201     }
202 
203     public void setTaskBackoffTimeMillis(long taskBackoffTimeMillis) {
204         this.taskBackoffTimeMillis = taskBackoffTimeMillis;
205     }
206 
207     public void setMaxLeasesForWorker(int maxLeasesForWorker) {
208         this.maxLeasesForWorker = maxLeasesForWorker;
209     }
210 
211     public void setMaxLeasesToStealAtOneTime(int maxLeasesToStealAtOneTime) {
212         this.maxLeasesToStealAtOneTime = maxLeasesToStealAtOneTime;
213     }
214 
215     public void setInitialLeaseTableReadCapacity(int initialLeaseTableReadCapacity) {
216         this.initialLeaseTableReadCapacity = initialLeaseTableReadCapacity;
217     }
218 
219     public void setInitialLeaseTableWriteCapacity(int initialLeaseTableWriteCapacity) {
220         this.initialLeaseTableWriteCapacity = initialLeaseTableWriteCapacity;
221     }
222 
223     public void setMaxListShardsRetryAttempts(int maxListShardsRetryAttempts) {
224         this.maxListShardsRetryAttempts = maxListShardsRetryAttempts;
225     }
226 
227     public void setLogWarningForTaskAfterMillis(Optional<Long> logWarningForTaskAfterMillis) {
228         this.logWarningForTaskAfterMillis = logWarningForTaskAfterMillis;
229     }
230 
231     public String getApplicationName() {
232         return applicationName;
233     }
234 
235     public String getWorkerId() {
236         return workerId;
237     }
238 
239     public String getStream() {
240         return stream;
241     }
