788     static boolean isMimeTypeSupported(String mimeType) {
789         return !UNSUPPORTED_MIME_TYPES.contains(mimeType);

792     static boolean isFileExtensionSupported(String filename) {
793         return !UNSUPPORTED_EXTENSIONS.contains(fileExtension(filename));
794     }

796     @NonNull
797     static String fileExtension(final String filename) {
798         int i = filename.lastIndexOf('.');
799         return i < 0 ? "" : filename.substring(i);
800     }
801 
802     @NonNull
803     static String mimeTypeFromFilename(String filename) {
804         return ALL_EXTENSIONS_TO_MIME_TYPES.getOrDefault(fileExtension(filename), Collections.emptyList()).stream().findFirst().orElse("");
805     }
