67      @Override
68      public String getPlatformApplicationArn(PlatformType platformType) {
69          switch (platformType) {
70              case ADM:
71                  return checkNotEmpty(configuration.getAdm().getArn(), "Amazon Device Manager application arn must be defined in config");
72              case APNS:
73                  return checkNotEmpty(configuration.getApns().getArn(), "Apple Push Notification service application arn must be defined in config");
74              case APNS_SANDBOX:
75                  return checkNotEmpty(configuration.getApnsSandbox().getArn(), "Apple Push Notification service Sandbox application arn must be defined in config");
76              case GCM:
77                  return checkNotEmpty(configuration.getGcm().getArn(), "Google Cloud Messaging (Firebase) application arn must be defined in config");
78              default:
79                  throw new IllegalArgumentException("Unknown platform type " + platformType);
80          }
81      }

112     @Override

114         LOGGER.debug("Creating topic sns with name " + topicName);
115         return client.createTopic((CreateTopicRequest.Builder b) ->
116              b.name(topicName).attributes(Collections.singletonMap("FifoTopic", Boolean.toString(SimpleNotificationService.isFifoTopic(topicName))))

120     @Override
121     public Publisher<Topic> listTopics() {
122         return Flux.generate(client::listTopics, (ListTopicsResponse listTopicsResult, SynchronousSink<List<Topic>> sink) -> {
123             sink.next(listTopicsResult.topics());
124 
125             if (listTopicsResult.nextToken() != null) {
126                 try {
127                     return client.listTopics(r -> r.nextToken(listTopicsResult.nextToken()));
128 
129                 } catch (SdkClientException e) {
130                     sink.error(e);
131                 }
132             }
133 
134             sink.complete();
135 
136             return ListTopicsResponse.builder().build();
137         }).flatMap(Flux::fromIterable);
138     }

203     @Override
204     public String createPlatformEndpoint(String platformApplicationArn, String deviceToken, String customUserData) {
205         try {
206             LOGGER.debug("Creating platform endpoint with token " + deviceToken);
207             CreatePlatformEndpointRequest.Builder request = CreatePlatformEndpointRequest.builder()
208                 .platformApplicationArn(platformApplicationArn)
209                 .token(deviceToken);
210             if (customUserData != null && !customUserData.isEmpty()) {
211                 request.customUserData(customUserData);
212             }
213             return client.createPlatformEndpoint(request.build()).endpointArn();
214         } catch (InvalidParameterException ipe) {
215             String message = ipe.getMessage();
216             LOGGER.debug("Exception message: " + message);
217             Pattern p = Pattern.compile(".*Endpoint (arn:aws:sns[^ ]+) already exists with the same Token.*");
218             Matcher m = p.matcher(message);
219             if (m.matches()) {
220                 // The platform endpoint already exists for this token, but with additional custom data that
221                 // createEndpoint doesn't want to overwrite. Just use the existing platform endpoint.
222                 return m.group(1);
223             }
224             // Rethrow the exception, the input is actually bad.
225             throw ipe;
226         }
227     }

245     @Override
246     public String validateDeviceToken(String platformApplicationArn, String endpointArn, String deviceToken, String customUserData) {
247         LOGGER.debug("Retrieving platform endpoint data...");
248         // Look up the platform endpoint and make sure the data in it is current, even if it was just created.
249         try {
250             GetEndpointAttributesResponse result = client.getEndpointAttributes(r -> r.endpointArn(endpointArn));
251             if (Objects.equals(result.attributes().get("Token"), deviceToken) && result.attributes().get("Enabled").equalsIgnoreCase(Boolean.TRUE.toString())) {
252                 setEndpointAttributes(endpointArn, Collections.singletonMap("CustomUserData", customUserData));
253                 return endpointArn;
254             }
255         } catch (NotFoundException ignored) {
256             // We had a stored ARN, but the platform endpoint associated with it disappeared. Recreate it.
257             return createPlatformEndpoint(platformApplicationArn, deviceToken, customUserData);
258         }
259 
260         LOGGER.debug("Platform endpoint update required...");
261 
262         // The platform endpoint is out of sync with the current data, update the token and enable it.
263         LOGGER.debug("Updating platform endpoint " + endpointArn);
264         try {
265             Map<String, String> attrs = new LinkedHashMap<>();
266             attrs.put("CustomUserData",  customUserData);
267             attrs.put("Enabled",  Boolean.TRUE.toString());
268             setEndpointAttributes(endpointArn, attrs);
269             return endpointArn;
270         } catch (InvalidParameterException ignored) {
271             deleteEndpoint(endpointArn);
272             return createPlatformEndpoint(platformApplicationArn, deviceToken, customUserData);
273         }
274     }

286     private static String checkNotEmpty(String arn, String errorMessage) {
287         if (arn == null || arn.isEmpty()) {
288             throw new IllegalStateException(errorMessage);
289         }
290         return arn;
291     }

293     private static Map<String, MessageAttributeValue> toAttributes(Map<String, String> attributes) {
294         if (attributes.isEmpty()) {
295             return Collections.emptyMap();
296         }
297 
298         return attributes.entrySet().stream().collect(Collectors.toMap(
299             Map.Entry::getKey,
300             e -> MessageAttributeValue.builder().stringValue(e.getValue()).dataType("String").build()
301         ));
302     }

304     private String ensureTopicArn(String nameOrArn) {
305         if (nameOrArn == null || nameOrArn.isEmpty()) {
306             return "";
307         }
308         if (nameOrArn.startsWith("arn:aws:sns")) {
309             return nameOrArn;
310         }
311 
312         if (namesToArn.containsKey(nameOrArn)) {
313             return namesToArn.get(nameOrArn);
314         }
315 
316         Flux.from(listTopics())
317             .takeUntil((Topic topic) -> topic.topicArn().endsWith(":" + nameOrArn))
318             .subscribe(topic -> {
319                 String topicName = topic.topicArn().substring(topic.topicArn().lastIndexOf(':') + 1);
320                 namesToArn.put(topicName, topic.topicArn());
321             });
322 
323         String topicArn = namesToArn.get(nameOrArn);
324 
325         if (topicArn != null && !topicArn.isEmpty()) {
326             return topicArn;
327         }
328 
329         return createTopic(nameOrArn);
330     }

332     private String buildAndroidMessage(Map<String, Object> data, String collapseKey, boolean delayWhileIdle, int timeToLive, boolean dryRun) {
333         Map<String, Object> value = new LinkedHashMap<>();
334         value.put("collapse_key", collapseKey);
335         value.put("data", data);
336         value.put("delay_while_idle", delayWhileIdle);
337         value.put("time_to_live", timeToLive);
338         value.put("dry_run", dryRun);
339         return toJson(value);
340     }
341 
342     private String buildIosMessage(Map<String, Object> data) {
343         return toJson(Collections.singletonMap("aps", data));
344     }
345 
346     private String publishToTarget(String endpointArn, String platformType, String jsonMessage) {
347         return client.publish(r -> {
348             r.targetArn(endpointArn).messageStructure("json").message(toJson(Collections.singletonMap(platformType, jsonMessage)));
349         }).messageId();
350     }

353         return client.setEndpointAttributes(r -> r.endpointArn(endpointArn).attributes(attributes));

361     private String toJson(Map<String, Object> message) {
362         try {
363             return objectMapper.writeValueAsString(message);
364         } catch (JsonProcessingException e) {
365             throw new IllegalArgumentException("Cannot write json for message " + message, e);
366         }
