35  /**
36   * Default simple queue service implementation.
37   */
38  public class DefaultSimpleQueueService implements SimpleQueueService {
39  
40      private static final Logger LOGGER = LoggerFactory.getLogger(SimpleQueueService.class);
41  
42      private final SqsClient client;
43      private final SimpleQueueServiceConfiguration configuration;
44  
45      private final ConcurrentMap<String, String> queueUrlByNames = new ConcurrentHashMap<>();
46  
47      public DefaultSimpleQueueService(
48          SqsClient client,
49          SimpleQueueServiceConfiguration configuration
50      ) {
51          this.client = client;
52          this.configuration = configuration;
53      }
54  
55      @Override
56      public String getDefaultQueueName() {
57          assertDefaultQueueName();
58          return configuration.getQueue();
59      }
60  
61      @Override
62      public boolean isCaching() {
63          return configuration.isCache();
64      }
65  
66      /**
67       * @param queueName the name of the queue
68       * @return queue URL
69       */
70      public String createQueue(String queueName) {
71          QueueConfiguration configuration = this.configuration.copy();
72          configuration.setQueue(queueName);
73  
74          if (queueName.endsWith(".fifo")) {
75              configuration.setFifo(true);
76          }
77  
78          return createQueue(configuration);
79      }
80  
81      /**
82       * @param configuration the configuration of the queue
83       * @return queue URL
84       */
85      public String createQueue(QueueConfiguration configuration) {
86          CreateQueueRequest.Builder createQueueRequest = CreateQueueRequest.builder().queueName(configuration.getQueue());
87  
88          Map<QueueAttributeName, String> attributes = new HashMap<>();
89  
90          if (configuration.getDelaySeconds() != null && configuration.getDelaySeconds() != 0) {
91              attributes.put(QueueAttributeName.DELAY_SECONDS, String.valueOf(configuration.getDelaySeconds()));
92          }
93  
94          if (configuration.getMessageRetentionPeriod() != null && configuration.getMessageRetentionPeriod() != 0) {
95              attributes.put(QueueAttributeName.MESSAGE_RETENTION_PERIOD, String.valueOf(configuration.getMessageRetentionPeriod()));
96          }
97  
98          if (configuration.getMaximumMessageSize() != null && configuration.getMaximumMessageSize() != 0) {
99              attributes.put(QueueAttributeName.MAXIMUM_MESSAGE_SIZE, String.valueOf(configuration.getMaximumMessageSize()));
100         }
101 
102         if (configuration.getVisibilityTimeout() != null && configuration.getVisibilityTimeout() != 0) {
103             attributes.put(QueueAttributeName.VISIBILITY_TIMEOUT, String.valueOf(configuration.getVisibilityTimeout()));
104         }
105 
106         if (configuration.getFifo()) {
107             attributes.put(QueueAttributeName.FIFO_QUEUE, Boolean.TRUE.toString());
108         }
109 
110         if (!attributes.isEmpty()) {
111             createQueueRequest.attributes(attributes);
112         }
113 
114         String queueUrl = client.createQueue(createQueueRequest.build()).queueUrl();
115 
116         LOGGER.debug("Queue created (queueUrl={})", queueUrl);
117 
118         addQueue(queueUrl);
119 
120         return queueUrl;
121     }

123     /**
124      * @param queueName
125      * @param receiptHandle
126      */
127     public void deleteMessage(String queueName, String receiptHandle) {
128         String queueUrl = getQueueUrl(queueName);
129         client.deleteMessage(DeleteMessageRequest.builder().queueUrl(queueUrl).receiptHandle(receiptHandle).build());
130         LOGGER.debug("Message deleted (queueUrl={}})", queueUrl);
131     }
132 
133     /**
134      * @param queueName
135      */
136     public void deleteQueue(String queueName) {
137         String queueUrl = getQueueUrl(queueName);
138         client.deleteQueue(b -> b.queueUrl(queueUrl));
139         removeQueue(queueUrl);
140     }
141 
142     /**
143      * @param queueName
144      * @return
145      */
146     public Map<QueueAttributeName, String> getQueueAttributes(String queueName) {
147         String queueUrl = getQueueUrl(queueName);
148 
149         GetQueueAttributesRequest.Builder getQueueAttributesRequest = GetQueueAttributesRequest
150             .builder()
151             .queueUrl(queueUrl)
152             .attributeNames(QueueAttributeName.ALL);
153 
154         try {
155             return client.getQueueAttributes(getQueueAttributesRequest.build()).attributes();
156         } catch (AwsServiceException exception) {
157             if ("AWS.SimpleQueueService.NonExistentQueue".equals(exception.awsErrorDetails().errorCode())) {
158                 removeQueue(queueUrl);
159             }
160             LOGGER.warn("An amazon service exception was caught while getting queue attributes", exception);
161             return Collections.emptyMap();
162         } catch (SdkClientException exception) {
163             LOGGER.warn("An amazon client exception was catched while getting queue attributes", exception);
164             return Collections.emptyMap();
165         }
166     }
167 
168     /**
169      * @return the queue URL
170      */
171     public String getQueueUrl(String queueName) {
172         if (queueName.startsWith("http://") || queueName.startsWith("https://")) {
173             return queueName;
174         }
175 
176         if (StringUtils.isNotEmpty(configuration.getQueueNamePrefix())) {
177             queueName = configuration.getQueueNamePrefix() + queueName;
178         }
179 
180         if (queueUrlByNames.isEmpty() && isCaching()) {
181             loadQueues();
182         }
183 
184         String queueUrl = isCaching() ? queueUrlByNames.get(queueName) : getQueueUrlDirect(queueName);
185 
186         if (StringUtils.isEmpty(queueUrl) && (configuration.isAutoCreateQueue())) {
187             queueUrl = createQueue(queueName);
188         }
189 
190         if (StringUtils.isEmpty(queueUrl)) {
191             throw QueueDoesNotExistException.builder().message("Queue " + queueName + " not found").build();
192         }
193 
194         return queueUrl;
195     }
196 
197     @Override
198     public String getQueueArn(String queueName) {
199         GetQueueAttributesResponse attributes = client.getQueueAttributes(b -> b.queueUrl(getQueueUrl(queueName)).attributeNames(QueueAttributeName.QUEUE_ARN));
200         return attributes.attributes().get(QueueAttributeName.QUEUE_ARN);
201     }

203     /**
204      * @param reload
205      * @return
206      */
207     public List<String> listQueueNames(boolean reload) {
208         if (queueUrlByNames.isEmpty() || reload) {
209             loadQueues();
210         }
211         return queueUrlByNames.keySet().stream().sorted().collect(Collectors.toList());
212     }
213 
214     /**
215      * @param reload
216      * @return
217      */
218     public List<String> listQueueUrls(boolean reload) {
219         if (queueUrlByNames.isEmpty() || reload) {
220             loadQueues();
221         }
222         return queueUrlByNames.values().stream().sorted().collect(Collectors.toList());
223     }

225     /**
226      * @param queueName
227      * @param maxNumberOfMessages
228      * @param visibilityTimeout
229      * @param waitTimeSeconds
230      * @return
231      */
232     public List<Message> receiveMessages(String queueName, int maxNumberOfMessages, int visibilityTimeout, int waitTimeSeconds) {
233         String queueUrl = getQueueUrl(queueName);
234 
235         ReceiveMessageRequest.Builder receiveMessageRequest = ReceiveMessageRequest.builder().queueUrl(queueUrl);
236         if (maxNumberOfMessages > 0) {
237             receiveMessageRequest.maxNumberOfMessages(maxNumberOfMessages);
238         }
239         if (visibilityTimeout > 0) {
240             receiveMessageRequest.visibilityTimeout(visibilityTimeout);
241         }
242         if (waitTimeSeconds > 0) {
243             receiveMessageRequest.waitTimeSeconds(waitTimeSeconds);
244         }
245         List<Message> messages = client.receiveMessage(receiveMessageRequest.build()).messages();
246         LOGGER.debug("Messages received (count={})", messages.size());
247         return messages;
248     }
249 
250     /**
251      * @param queueName
252      * @param messageBody
253      * @param delaySeconds
254      * @return
255      */
256     public String sendMessage(String queueName, String messageBody, int delaySeconds, String groupId) {
257         String queueUrl = getQueueUrl(queueName);
258 
259         SendMessageRequest.Builder request = SendMessageRequest.builder().queueUrl(queueUrl).messageBody(messageBody);
260         if (delaySeconds > 0) {
261             request.delaySeconds(delaySeconds);
262         }
263 
264         if (StringUtils.isNotEmpty(groupId)) {
265             request.messageGroupId(groupId);
266         }
267 
268         String messageId = client.sendMessage(request.build()).messageId();
269         LOGGER.debug("Message sent (messageId={})", messageId);
270         return messageId;
271     }
272 
273     /**
274      * @param queueName
275      * @param messageBody
276      * @return
277      */
278     public String sendMessage(String queueName, String messageBody, Consumer<SendMessageRequest.Builder> messageConfiguration) {
279         String queueUrl = getQueueUrl(queueName);
280 
281         SendMessageRequest.Builder request = SendMessageRequest.builder().queueUrl(queueUrl).messageBody(messageBody);
282 
283         messageConfiguration.accept(request);
284 
285         String messageId = client.sendMessage(request.build()).messageId();
286         LOGGER.debug("Message sent (messageId={})", messageId);
287         return messageId;
288     }
289 
290     void assertDefaultQueueName() {
291         if (StringUtils.isEmpty(configuration.getQueue())) {
292             throw new IllegalStateException("Queue not configured");
293         }
294     }
295 
296     // PRIVATE
297     private static String getQueueNameFromUrl(String queueUrl) {
298         return Arrays.stream(queueUrl.split("/")).reduce((first, second) -> second).orElse(null);
299     }
300 
301     private String getQueueUrlDirect(String queueName) {
302         try {
303             return client.getQueueUrl(b -> b.queueName(queueName)).queueUrl();
304         } catch (QueueDoesNotExistException ignored) {
305             return null;
306         }
307     }
308 
309     private void addQueue(String queueUrl) {
310         if (StringUtils.isEmpty(queueUrl)) {
311             throw new IllegalStateException("Queue URL cannot be null or empty");
312         }
313 
314         synchronized (queueUrlByNames) {
315             queueUrlByNames.put(getQueueNameFromUrl(queueUrl),  queueUrl);
316         }
317     }
318 
319     private void loadQueues() {
320         ListQueuesRequest.Builder listQueuesRequest = ListQueuesRequest.builder();
321         if (StringUtils.isNotEmpty(configuration.getQueueNamePrefix())) {
322             listQueuesRequest.queueNamePrefix(configuration.getQueueNamePrefix());
323         }
324 
325         Map<String, String> queueUrls = client.listQueues(listQueuesRequest.build())
326             .queueUrls()
327             .stream()
328             .collect(
329                 Collectors.toMap(
330                     DefaultSimpleQueueService::getQueueNameFromUrl,
331                     Function.identity()
332                 )
333             );
334 
335         synchronized (queueUrlByNames) {
336             queueUrlByNames.clear();
337             queueUrlByNames.putAll(queueUrls);
338         }
339     }
340 
341     private void removeQueue(String queueUrl) {
342         if (StringUtils.isEmpty(queueUrl)) {
343             throw new IllegalStateException("Queue URL cannot be null or empty");
344         }
345 
346         synchronized (queueUrlByNames) {
347             queueUrlByNames.remove(getQueueNameFromUrl(queueUrl));
348         }
349     }
