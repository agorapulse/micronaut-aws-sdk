72  /**
73   * Implementation of {@link TableSchema} that builds a table schema based on properties and annotations of a bean
74   * class. Example:
75   * <pre>
76   * <code>
77   * {@literal @}DynamoDbBean
78   * public class Customer {
79   *     private String accountId;
80   *     private int subId;            // primitive types are supported
81   *     private String name;
82   *     private Instant createdDate;
83   *
84   *     {@literal @}PartitionKey
85   *     public String getAccountId() { return this.accountId; }
86   *     public void setAccountId(String accountId) { this.accountId = accountId; }
87   *
88   *     {@literal @}SortKey
89   *     public int getSubId() { return this.subId; }
90   *     public void setSubId(int subId) { this.subId = subId; }
91   *
92   *     // Defines a GSI (customers_by_name) with a partition key of 'name'
93   *     {@literal @}SecondaryPartitionKey(indexNames = "customers_by_name")
94   *     public String getName() { return this.name; }
95   *     public void setName(String name) { this.name = name; }
96   *
97   *     // Defines an LSI (customers_by_date) with a sort key of 'createdDate' and also declares the
98   *     // same attribute as a sort key for the GSI named 'customers_by_name'
99   *     {@literal @}SecondarySortKey(indexNames = {"customers_by_date", "customers_by_name"})
100  *     public Instant getCreatedDate() { return this.createdDate; }
101  *     public void setCreatedDate(Instant createdDate) { this.createdDate = createdDate; }
102  * }
103  *
104  * </pre>
105  * <p>
106  * Creating an {@link BeanIntrospectionTableSchema} is a moderately expensive operation, and should be performed sparingly. This is
107  * usually done once at application startup.
108  * <p>
109  *
110  * @param <T> The type of object that this {@link TableSchema} maps to.
111  */
112 @SdkPublicApi
113 @ThreadSafe
114 public final class BeanIntrospectionTableSchema<T> extends WrappedTableSchema<T, StaticTableSchema<T>> {

150     public static <T> BeanIntrospectionTableSchema<T> create(Class<T> beanClass, BeanContext context, MetaTableSchemaCache metaTableSchemaCache) {
151         debugLog(beanClass, () -> "Creating bean schema");
152         // Fetch or create a new reference to this yet-to-be-created TableSchema in the cache
153         MetaTableSchema<T> metaTableSchema = metaTableSchemaCache.getOrCreate(beanClass);
154 
155         BeanIntrospectionTableSchema<T> newTableSchema =
156             new BeanIntrospectionTableSchema<>(createStaticTableSchema(beanClass, context, metaTableSchemaCache));
157         if (!metaTableSchema.isInitialized()) {
158             metaTableSchema.initialize(newTableSchema);
159         }
160         return newTableSchema;
161     }
162 
163     // Called when creating an immutable TableSchema recursively. Utilizes the MetaTableSchema cache to stop infinite
164     // recursion
165     static <T> TableSchema<T> recursiveCreate(Class<T> beanClass, BeanContext context, MetaTableSchemaCache metaTableSchemaCache) {
166         Optional<MetaTableSchema<T>> metaTableSchema = metaTableSchemaCache.get(beanClass);
167 
168         // If we get a cache hit...
169         if (metaTableSchema.isPresent()) {
170             // Either: use the cached concrete TableSchema if we have one
171             if (metaTableSchema.get().isInitialized()) {
172                 return metaTableSchema.get().concreteTableSchema();
173             }
174 
175             // Or: return the uninitialized MetaTableSchema as this must be a recursive reference and it will be
176             // initialized later as the chain completes
177             return metaTableSchema.get();
178         }
179 
180         // Otherwise: cache doesn't know about this class; create a new one from scratch
181         return create(beanClass, context, metaTableSchemaCache);
182 
183     }

185     private static <T> StaticTableSchema<T> createStaticTableSchema(Class<T> beanClass,
186                                                                     BeanContext beanContext,
187                                                                     MetaTableSchemaCache metaTableSchemaCache) {
188         Optional<BeanIntrospection<T>> introspectionOptional = BeanIntrospector.SHARED.findIntrospection(beanClass);
189 
190 
191         if (!introspectionOptional.isPresent()) {
192             throw new IllegalArgumentException("A DynamoDb bean class must be annotated with @Introspected, but " + beanClass.getTypeName() + " was not.");
193         }
194 
195         BeanIntrospection<T> introspection = introspectionOptional.get();
196 
197 
198         StaticTableSchema.Builder<T> builder = StaticTableSchema.builder(beanClass).newItemSupplier(introspection::instantiate);
199 
200         Optional<AnnotationValue<DynamoDbBean>> optionalDynamoDbBean = introspection.findAnnotation(DynamoDbBean.class);
201         if (optionalDynamoDbBean.isPresent()) {
202             builder.attributeConverterProviders(createConverterProvidersFromAnnotation(beanClass, optionalDynamoDbBean.get(), beanContext));
203         } else {
204             builder.attributeConverterProviders(new LegacyAttributeConverterProvider());
205         }
206 
207         List<StaticAttribute<T, ?>> attributes = introspection.getBeanProperties().stream()
208             .filter(p -> isMappableProperty(beanClass, p))
209             .map(propertyDescriptor -> extractAttributeFromProperty(beanClass, metaTableSchemaCache, builder, propertyDescriptor, beanContext))
210             .filter(Objects::nonNull)
211             .collect(Collectors.toList());
212 
213         builder.attributes(attributes);
214 
215         return builder.build();
216     }

218     private static <T, P> StaticAttribute<T, P> extractAttributeFromProperty(
219         Class<T> beanClass,
220         MetaTableSchemaCache metaTableSchemaCache,
221         StaticTableSchema.Builder<T> builder,
222         BeanProperty<T, P> propertyDescriptor,
223         BeanContext beanContext
224     ) {
225         Optional<AnnotationValue<Annotation>> dynamoDbFlatten = findAnnotation(propertyDescriptor, DynamoDbFlatten.class, Flatten.class);
226 
227         if (dynamoDbFlatten.isPresent()) {
228             builder.flatten(
229                 BeanIntrospectionTableSchema.create(propertyDescriptor.getType(), beanContext, metaTableSchemaCache),
230                 propertyDescriptor::get,
231                 propertyDescriptor::set
232             );
233             return null;
234         } else {
235             AttributeConfiguration attributeConfiguration = resolveAttributeConfiguration(propertyDescriptor);
236 
237             StaticAttribute.Builder<T, P> attributeBuilder = staticAttributeBuilder(propertyDescriptor, beanClass, metaTableSchemaCache, attributeConfiguration, beanContext);
238 
239             createAttributeConverterFromAnnotation(propertyDescriptor, beanContext).ifPresent(attributeBuilder::attributeConverter);
240 
241             addTagsToAttribute(attributeBuilder, propertyDescriptor);
242             return attributeBuilder.build();
243         }
244     }
245 
246     private static <T> AttributeConfiguration resolveAttributeConfiguration(BeanProperty<T, ?> propertyDescriptor) {
247         boolean shouldPreserveEmptyObject = findAnnotation(propertyDescriptor, DynamoDbPreserveEmptyObject.class, PreserveEmptyObjects.class).isPresent();
248         boolean shouldIgnoreNulls = findAnnotation(propertyDescriptor, DynamoDbIgnoreNulls.class, IgnoreNulls.class).isPresent();
249 
250         return AttributeConfiguration.builder()
251             .preserveEmptyObject(shouldPreserveEmptyObject)
252             .ignoreNulls(shouldIgnoreNulls)
253             .build();
254     }

256     @SuppressWarnings("unchecked")
257     private static List<AttributeConverterProvider> createConverterProvidersFromAnnotation(Class<?> beanClass, AnnotationValue<DynamoDbBean> dynamoDbBean, BeanContext beanContext) {
258         Class<? extends AttributeConverterProvider>[] providerClasses = (Class<? extends AttributeConverterProvider>[]) dynamoDbBean.classValues("converterProviders");
259         if (providerClasses.length == 0) {
260             providerClasses = new Class[]{LegacyAttributeConverterProvider.class};
261         }
262 
263         return Arrays.stream(providerClasses)
264             .peek(c -> debugLog(beanClass, () -> "Adding Converter: " + c.getTypeName()))
265             .map(c -> (AttributeConverterProvider) fromContextOrNew(c, beanContext).get())
266             .collect(Collectors.toList());
267     }

269     private static <T, P> StaticAttribute.Builder<T, P> staticAttributeBuilder(
270         BeanProperty<T, P> propertyDescriptor,
271         Class<T> beanClass,
272         MetaTableSchemaCache metaTableSchemaCache,
273         AttributeConfiguration attributeConfiguration,
274         BeanContext beanContext
275     ) {
276         Argument<P> propertyType = propertyDescriptor.asArgument();
277         EnhancedType<P> propertyTypeToken = convertTypeToEnhancedType(propertyType, metaTableSchemaCache, attributeConfiguration, beanContext);
278         return StaticAttribute.builder(beanClass, propertyTypeToken)
279             .name(attributeNameForProperty(propertyDescriptor))
280             .getter(propertyDescriptor::get)
281             // secondary indices can be read only
282             .setter(propertyDescriptor.isReadOnly() ? (bean, value) -> {
283             } : propertyDescriptor::set);
284     }

286     /**
287      * Converts a {@link Type} to an {@link EnhancedType}. Usually {@link EnhancedType#of} is capable of doing this all
288      * by itself, but for the BeanTableSchema we want to detect if a parameterized class is being passed without a
289      * converter that is actually another annotated class in which case we want to capture its schema and add it to the
290      * EnhancedType. Unfortunately this means we have to duplicate some of the recursive Type parsing that
291      * EnhancedClient otherwise does all by itself.
292      */
293     @SuppressWarnings("unchecked")
294     private static <T> EnhancedType<T> convertTypeToEnhancedType(
295         Argument<T> type,
296         MetaTableSchemaCache metaTableSchemaCache,
297         AttributeConfiguration attributeConfiguration,
298         BeanContext beanContext
299     ) {
300         if (List.class.equals(type.getType())) {
301             EnhancedType<?> enhancedType = convertTypeToEnhancedType(type.getTypeParameters()[0], metaTableSchemaCache, attributeConfiguration, beanContext);
302             return (EnhancedType<T>) EnhancedType.listOf(enhancedType);
303         } else if (Map.class.equals(type.getType())) {
304             EnhancedType<?> keyType = convertTypeToEnhancedType(type.getTypeVariable("K").orElseThrow(() -> new IllegalArgumentException("Missing key type")), metaTableSchemaCache, attributeConfiguration, beanContext);
305             EnhancedType<?> valueType = convertTypeToEnhancedType(type.getTypeVariable("V").orElseThrow(() -> new IllegalArgumentException("Missing value type")), metaTableSchemaCache, attributeConfiguration, beanContext);
306             return (EnhancedType<T>) EnhancedType.mapOf(keyType, valueType);
307         }
308 
309         Class<T> clazz = type.getType();
310 
311         Consumer<EnhancedTypeDocumentConfiguration.Builder> attrConfiguration = b -> b
312             .preserveEmptyObject(attributeConfiguration.preserveEmptyObject())
313             .ignoreNulls(attributeConfiguration.ignoreNulls());
314 
315         if (type.isAnnotationPresent(DynamoDbBean.class)) {
316             return EnhancedType.documentOf(
317                 clazz,
318                 BeanIntrospectionTableSchema.recursiveCreate(clazz, beanContext, metaTableSchemaCache),
319                 attrConfiguration
320             );
321         }
322 
323         return EnhancedType.of(clazz);
324     }

326     @SuppressWarnings("unchecked")
327     private static <T, P> Optional<AttributeConverter<P>> createAttributeConverterFromAnnotation(
328         BeanProperty<T, P> propertyDescriptor,
329         BeanContext beanContext
330     ) {
331         return findAnnotation(propertyDescriptor, DynamoDbConvertedBy.class, ConvertedBy.class)
332             .flatMap(AnnotationValue::classValue)
333             .map(clazz -> (AttributeConverter<P>) fromContextOrNew(clazz, beanContext).get());
334     }

336     /**
337      * This method scans all the annotations on a property and looks for a meta-annotation of
338      * {@link BeanTableSchemaAttributeTag}. If the meta-annotation is found, it attempts to create
339      * an annotation tag based on a standard named static method
340      * of the class that tag has been annotated with passing in the original property annotation as an argument.
341      */
342     private static <T> void addTagsToAttribute(StaticAttribute.Builder<?, ?> attributeBuilder,
343                                                BeanProperty<T, ?> propertyDescriptor) {
344 
345         findAnnotation(propertyDescriptor, UPDATE_BEHAVIOUR_ANNOTATIONS)
346             .flatMap(anno -> anno.enumValue(Enum.class))
347             .ifPresent(behavior -> attributeBuilder.addTag(StaticAttributeTags.updateBehavior(software.amazon.awssdk.enhanced.dynamodb.mapper.UpdateBehavior.valueOf(behavior.name()))));
348 
349         findAnnotation(propertyDescriptor, PARTITION_KEYS_ANNOTATIONS)
350             .ifPresent(anno -> attributeBuilder.addTag(StaticAttributeTags.primaryPartitionKey()));
351 
352         findAnnotation(propertyDescriptor, SORT_KEYS_ANNOTATIONS)
353             .ifPresent(anno -> attributeBuilder.addTag(StaticAttributeTags.primarySortKey()));
354 
355         findAnnotation(propertyDescriptor, SECONDARY_PARTITION_KEYS_ANNOTATIONS)
356             .map(anno -> anno.stringValues("indexNames"))
357             .ifPresent(indexNames -> attributeBuilder.addTag(StaticAttributeTags.secondaryPartitionKey(Arrays.asList(indexNames))));
358 
359         findAnnotation(propertyDescriptor, SECONDARY_SORT_KEYS_ANNOTATIONS)
360             .map(anno -> anno.stringValues("indexNames"))
361             .ifPresent(indexNames -> attributeBuilder.addTag(StaticAttributeTags.secondarySortKey(Arrays.asList(indexNames))));
362     }

364     private static <R> Supplier<R> fromContextOrNew(Class<R> clazz, BeanContext beanContext) {
365         Optional<R> optionalBean = beanContext.findBean(clazz);
366         if (optionalBean.isPresent()) {
367             return optionalBean::get;
368         }
369 
370         Optional<BeanIntrospection<R>> optionalBeanIntrospection = BeanIntrospector.SHARED.findIntrospection(clazz);
371         if (optionalBeanIntrospection.isPresent()) {
372             return optionalBeanIntrospection.get()::instantiate;
373         }
374 
375         try {
376             Constructor<R> constructor = clazz.getConstructor();
377             debugLog(clazz, () -> "Constructor: " + constructor);
378             return ObjectConstructor.create(clazz, constructor);
379         } catch (NoSuchMethodException e) {
380             throw new IllegalArgumentException(
381                 String.format("Class '%s' appears to have no default constructor thus cannot be used with the BeanTableSchema", clazz), e);
382         }
383     }

385     private static <T> String attributeNameForProperty(BeanProperty<T, ?> propertyDescriptor) {
386         return findAnnotation(propertyDescriptor, DynamoDbAttribute.class, Attribute.class)
387             .flatMap(AnnotationValue::stringValue)
388             .orElseGet(propertyDescriptor::getName);
389     }

391     private static <T> boolean isMappableProperty(Class<T> beanClass, BeanProperty<T, ?> propertyDescriptor) {
392 
393         if (propertyDescriptor.isWriteOnly()) {
394             debugLog(beanClass, () -> "Ignoring bean property " + propertyDescriptor.getName() + " because it is write only.");
395             return false;
396         }
397 
398         if (propertyDescriptor.isReadOnly()) {
399             debugLog(beanClass, () -> "Ignoring bean property " + propertyDescriptor.getName() + " because it is read only.");
400             return isSecondaryIndex(propertyDescriptor);
401         }
402 
403         if (propertyDescriptor.isAnnotationPresent(DynamoDbIgnore.class)) {
404             debugLog(beanClass, () -> "Ignoring bean property " + propertyDescriptor.getName() + " because it is ignored.");
405             return false;
406         }
407 
408         if (propertyDescriptor.isAnnotationPresent(Ignore.class)) {
409             debugLog(beanClass, () -> "Ignoring bean property " + propertyDescriptor.getName() + " because it is ignored.");
410             return false;
411         }
412 
413         return true;
414     }
415 
416     private static <T> boolean isSecondaryIndex(BeanProperty<T, ?> propertyDescriptor) {
417         return propertyDescriptor.getAnnotationNames().stream().anyMatch(name ->
418             SECONDARY_PARTITION_KEYS_ANNOTATIONS.contains(name) || SECONDARY_SORT_KEYS_ANNOTATIONS.contains(name)
419         );
420     }

422     private static void debugLog(Class<?> beanClass, Supplier<String> logMessage) {
423         BEAN_LOGGER.debug(() -> beanClass.getTypeName() + " - " + logMessage.get());
424     }

426     private static Optional<AnnotationValue<Annotation>> findAnnotation(AnnotationMetadataProvider source, Class<? extends Annotation>... annotationClasses) {
427         for (Class<? extends Annotation> annotationName : annotationClasses) {
428             @SuppressWarnings("unchecked")
429             Optional<AnnotationValue<Annotation>> maybeAnno = source.findAnnotation((Class<Annotation>) annotationName);
430             if (maybeAnno.isPresent()) {
431                 return maybeAnno;
432             }
433         }
434         return Optional.empty();
435     }
436 
437     private static Optional<AnnotationValue<Annotation>> findAnnotation(AnnotationMetadataProvider source, Iterable<String> annotationNames) {
438         for (String annotationName : annotationNames) {
439             Optional<AnnotationValue<Annotation>> maybeAnno = source.findAnnotation(annotationName);
440             if (maybeAnno.isPresent()) {
441                 return maybeAnno;
442             }
443         }
444         return Optional.empty();
445     }
