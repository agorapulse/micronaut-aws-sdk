115     @Override
116     public int updateAll(Publisher<T> items, UpdateBuilder<T, ?> update) {
117         // there is no batch update API, we can do batch updates in transaction but in that case it would cause
118         // doubling the writes
119 
120         BeanIntrospection<T> introspection = EntityIntrospection.getBeanIntrospection(table);
121         TableMetadata tableMetadata = table.tableSchema().tableMetadata();
122 
123         AtomicInteger counter = new AtomicInteger();
124 
125         Flux.from(items).map(this::postLoad).subscribe(entity -> {
126             introspection.getProperty(tableMetadata.primaryPartitionKey()).ifPresent(p -> update.partitionKey(p.get(entity)));
127             tableMetadata.primarySortKey().flatMap(introspection::getProperty).ifPresent(p -> update.sortKey(p.get(entity)));
128 
129             update.update(table, client, attributeConversionHelper, publisher);
130 
131             counter.incrementAndGet();
132         });
133 
134         return counter.get();
135     }

137     @Override
138     public T save(T entity) {
139         publisher.publishEvent(DynamoDbEvent.prePersist(entity));
140         T updated = table.updateItem(entity);
141         publisher.publishEvent(DynamoDbEvent.postPersist(updated));
142         return updated;

145     @Override
146     public Publisher<T> saveAll(Publisher<T> itemsToSave) {
147         List<T> saved = new ArrayList<>();
148         List<T> unprocessed = Flux.from(itemsToSave).buffer(BATCH_SIZE).map(batchItems -> enhancedClient.batchWriteItem(b -> {
149             b.writeBatches(batchItems.stream().map(i -> {
150                 publisher.publishEvent(DynamoDbEvent.prePersist(i));
151                 saved.add(i);
152                 return WriteBatch.builder(table.tableSchema().itemType().rawClass()).mappedTableResource(table).addPutItem(i).build();
153             }).collect(Collectors.toList()));
154         })).flatMap(r -> Flux.fromIterable(r.unprocessedPutItemsForTable(table))).collectList().blockOptional().orElse(Collections.emptyList());
155 
156         if (unprocessed.isEmpty()) {
157             saved.forEach(i -> publisher.publishEvent(DynamoDbEvent.postPersist(i)));
158             return Flux.fromIterable(saved);
159         }
160 
161         throw new IllegalArgumentException("Following items couldn't be saved:" + unprocessed.stream().map(Object::toString).collect(Collectors.joining(", ")));
162     }

178     public T delete(Key key) {
179         T item = table.tableSchema().mapToItem(key.primaryKeyMap(table.tableSchema()));
180         publisher.publishEvent(DynamoDbEvent.preRemove(item));
181         table.deleteItem(key);
182         publisher.publishEvent(DynamoDbEvent.postRemove(item));

186     @Override
187     public int deleteAll(Publisher<T> items) {
188         TableSchema<T> tableSchema = table.tableSchema();
189         List<T> deleted = new ArrayList<>();
190         List<Key> unprocessed = Flux.from(items).buffer(BATCH_SIZE).map(batchItems -> enhancedClient.batchWriteItem(b -> {
191             b.writeBatches(batchItems.stream().map(i -> {
192                     publisher.publishEvent(DynamoDbEvent.preRemove(i));
193                     deleted.add(i);
194                     return WriteBatch.builder(tableSchema.itemType().rawClass()).mappedTableResource(table).addDeleteItem(i).build();
195                 }
196             ).collect(Collectors.toList()));
197         })).flatMap(r -> Flux.fromIterable(r.unprocessedDeleteItemsForTable(table))).collectList().blockOptional().orElse(Collections.emptyList());
198 
199         if (unprocessed.isEmpty()) {
200             deleted.forEach(i -> publisher.publishEvent(DynamoDbEvent.postRemove(i)));
201             return deleted.size();
202         }
203 
204         throw new IllegalArgumentException("Following items couldn't be deleted:" + unprocessed.stream()
205             .map(k -> tableSchema.mapToItem(k.keyMap(tableSchema, TableMetadata.primaryIndexName()))).map(Object::toString)
206             .collect(Collectors.joining(", ")));
207     }

241     @Override
242     public void createTable() {
243         Map<String, ProjectionType> types = getProjectionTypes();
244         TableMetadata tableMetadata = table.tableSchema().tableMetadata();
245         table.createTable(b -> {
246             List<EnhancedLocalSecondaryIndex> localSecondaryIndices = new ArrayList<>();
247             List<EnhancedGlobalSecondaryIndex> globalSecondaryIndices = new ArrayList<>();
248 
249             tableMetadata.indices().forEach(i -> {
250                 if (TableMetadata.primaryIndexName().equals(i.name())) {
251                     return;
252                 }
253                 ProjectionType type = types.getOrDefault(i.name(), ProjectionType.KEYS_ONLY);
254                 if (tableMetadata.primaryPartitionKey().equals(tableMetadata.indexPartitionKey(i.name()))) {
255                     localSecondaryIndices.add(EnhancedLocalSecondaryIndex.create(i.name(), Projection.builder().projectionType(type).build()));
256                 } else {
257                     globalSecondaryIndices.add(EnhancedGlobalSecondaryIndex.builder().indexName(i.name()).projection(Projection.builder().projectionType(type).build()).build());
258                 }
259             });
260 
261             if (!localSecondaryIndices.isEmpty()) {
262                 b.localSecondaryIndices(localSecondaryIndices);
263             }
264 
265             if (!globalSecondaryIndices.isEmpty()) {
266                 b.globalSecondaryIndices(globalSecondaryIndices);
267             }
268         });
269 
270     }

272     private DetachedQuery<T> simplePartitionAndSort(Object partitionKey, Object sortKey) {
273         return doWithKey(partitionKey, sortKey, key -> {
274             if (key.sortKeyValue().isPresent()) {
275                 return Builders.query(q -> q.partitionKey(key.partitionKeyValue()).sortKey(s -> s.eq(key.sortKeyValue().get())));
276             }
277             return Builders.query(q -> q.partitionKey(key.partitionKeyValue()));
278         });
279     }

281     private Publisher<T> getAll(AttributeValue hashKey, Publisher<AttributeValue> rangeKeys) {
282         TableSchema<T> tableSchema = table.tableSchema();
283         return Flux.from(rangeKeys).buffer(BATCH_SIZE).map(batchRangeKeys -> enhancedClient.batchGetItem(b -> {
284             b.readBatches(batchRangeKeys.stream().map(k ->
285                 ReadBatch.builder(tableSchema.itemType().rawClass()).mappedTableResource(table).addGetItem(Key.builder().partitionValue(hashKey).sortValue(k).build()).build()
286             ).collect(Collectors.toList()));
287         })).flatMap(r -> Flux.fromIterable(r.resultsForTable(table)).map(this::postLoad));
288     }

290     private Map<String, ProjectionType> getProjectionTypes() {
291         Map<String, ProjectionType> types = new HashMap<>();
292 
293         BeanIntrospection<T> introspection = EntityIntrospection.getBeanIntrospection(table);
294         introspection.getBeanProperties().forEach(p -> {
295             AnnotationValue<com.agorapulse.micronaut.amazon.awssdk.dynamodb.annotation.Projection> projectionAnnotation = p.getAnnotation(com.agorapulse.micronaut.amazon.awssdk.dynamodb.annotation.Projection.class);
296 
297             if (projectionAnnotation == null) {
298                 return;
299             }
300 
301             Collection<String> indexNames = new ArrayList<>();
302 
303             indexNames.addAll(collectIndicesFromAnnotation(p, DynamoDbSecondarySortKey.class));
304             indexNames.addAll(collectIndicesFromAnnotation(p, SecondarySortKey.class));
305             indexNames.addAll(collectIndicesFromAnnotation(p, DynamoDbSecondaryPartitionKey.class));
306             indexNames.addAll(collectIndicesFromAnnotation(p, SecondaryPartitionKey.class));
307 
308             if (indexNames.isEmpty()) {
309                 return;
310             }
311 
312             ProjectionType type = projectionAnnotation.enumValue(ProjectionType.class).orElse(ProjectionType.KEYS_ONLY);
313             for (String name : indexNames) {
314                 types.put(name, type);
315             }
316         });
317 
318         return types;
319     }
320 
321     private List<String> collectIndicesFromAnnotation(BeanProperty<T, Object> p, Class<? extends Annotation> indexAnnotationClass) {
322         return p.findAnnotation(indexAnnotationClass).map(anno -> Arrays.asList(anno.stringValues("indexNames"))).orElse(Collections.emptyList());
323     }

325     private T postLoad(T i) {
326         publisher.publishEvent(DynamoDbEvent.postLoad(i));
327         return i;

330     private <R> R doWithKey(Object partitionKey, Object sortKey, Function<Key, R> function) {
331         String hashKeyName = table.tableSchema().tableMetadata().primaryPartitionKey();
332         AttributeValue partitionKeyValue = attributeConversionHelper.convert(table, hashKeyName, partitionKey);
333 
334         if (sortKey == null) {
335             return function.apply(Key.builder().partitionValue(partitionKeyValue).build());
336         }
337 
338         String rangeKeyName = table.tableSchema().tableMetadata().primarySortKey().get();
339         AttributeValue sortKeyValue = attributeConversionHelper.convert(table, rangeKeyName, sortKey);
340         return function.apply(
341             Key.builder()
342                 .partitionValue(partitionKeyValue)
343                 .sortValue(sortKeyValue)
344                 .build()
345         );
346     }
347 
348     private <R> Publisher<R> doWithKeys(Object partitionKey, Publisher<?> sortKeys, BiFunction<AttributeValue, Publisher<AttributeValue>, Publisher<R>> function) {
349         String hashKeyName = table.tableSchema().tableMetadata().primaryPartitionKey();
350         AttributeValue partitionKeyValue = attributeConversionHelper.convert(table, hashKeyName, partitionKey);
351 
352         Optional<String> sortKeyName = table.tableSchema().tableMetadata().primarySortKey();
353         return function.apply(partitionKeyValue, Flux.from(sortKeys).map(key -> attributeConversionHelper.convert(table, sortKeyName.get(), key)));
354     }
