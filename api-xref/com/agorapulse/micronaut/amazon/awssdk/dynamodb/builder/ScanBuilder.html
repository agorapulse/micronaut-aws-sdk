18  package com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder;
19  
20  import software.amazon.awssdk.enhanced.dynamodb.model.ScanEnhancedRequest;
21  import java.util.Arrays;
22  import java.util.Collection;
23  import java.util.function.Consumer;
24  
25  /**
26   * Builder for DynamoDB scans.
27   * @param <T> type of the DynamoDB entity
28   */
29  public interface ScanBuilder<T> extends DetachedScan<T> {
30  
31      /**
32       * Demand consistent reads.
33       * @param read the read keyword
34       * @return self
35       */
36      ScanBuilder<T> consistent(Builders.Read read);
37  
38      /**
39       * Demand inconsistent reads.
40       * @param read the read keyword
41       * @return self
42       */
43      ScanBuilder<T> inconsistent(Builders.Read read);
44  
45      /**
46       * Select the index on which this scan will be executed.
47       * @param name the name of the index to be used
48       * @return self
49       */
50      ScanBuilder<T> index(String name);
51  
52      /**
53       * One or more filter conditions.
54       *
55       * @param conditions consumer to build the conditions
56       * @return self
57       */
58      ScanBuilder<T> filter(Consumer<FilterConditionCollector<T>> conditions);
59  
60      /**
61       * Sets the desired pagination of the scans.
62       *
63       * This only sets the optimal pagination of the scans and does not limit the number of items returned.
64       *
65       * Use <code>{@link io.reactivex.Flowable#take(long)}</code> to limit the number results returned from the scan.
66       *
67       * @param page number of entities loaded by one scan request (not a number of total entities returned)
68       * @return self
69       */
70      ScanBuilder<T> page(int page);
71  
72      /**
73       * Sets the maximum number of items to be returned from the queries.
74       *
75       * This is a shortcut for calling <code>{@link io.reactivex.Flowable#take(long)}</code> on the result Flowable.
76       *
77       * @param max the maximum number of items returned
78       * @return self
79       */
80      ScanBuilder<T> limit(int max);
81  
82      /**
83       * Sets the scan offset by defining the exclusive start value.
84       * @param lastEvaluatedKey exclusive start value
85       * @return self
86       */
87      ScanBuilder<T> lastEvaluatedKey(Object lastEvaluatedKey);
88  
89      /**
90       * Configures the native scan expression.
91       *
92       * This method is an extension point which allows to configure properties which are not provides by this builder.
93       *
94       * @param configurer consumer to configure the native scan expression
95       * @return self
96       */
97      ScanBuilder<T> configure(Consumer<ScanEnhancedRequest.Builder> configurer);
98  
99      /**
100      * Limits which properties of the returned entities will be populated.
101      * @param propertyPaths property paths to be populated in the returned entities
102      * @return self
103      */
104     ScanBuilder<T> only(Collection<String> propertyPaths);
105 
106     /**
107      * Limits which properties of the returned entities will be populated.
108      * @param propertyPaths property paths to be populated in the returned entities
109      * @return self
110      */
111     default ScanBuilder<T> only(String... propertyPaths) {
112         return only(Arrays.asList(propertyPaths));
113     }
114 
115 }
