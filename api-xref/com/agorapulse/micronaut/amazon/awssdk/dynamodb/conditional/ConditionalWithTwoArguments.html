18  package com.agorapulse.micronaut.amazon.awssdk.dynamodb.conditional;
19  
20  import software.amazon.awssdk.enhanced.dynamodb.Expression;
21  import software.amazon.awssdk.enhanced.dynamodb.TableSchema;
22  import software.amazon.awssdk.enhanced.dynamodb.model.QueryConditional;
23  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
24  
25  import java.util.Collections;
26  
27  class ConditionalWithTwoArguments implements QueryConditional {
28  
29      static final String EQUAL_TO_TEMPLATE = "%s = %s";
30      static final String NOT_EQUAL_TO_TEMPLATE = "%s <> %s";
31      static final String LESS_THAN_TEMPLATE = "%s < %s";
32      static final String LESS_THAN_OR_EQUAL_TO_TEMPLATE = "%s <= %s";
33      static final String GREATER_THAN_TEMPLATE = "%s > %s";
34      static final String GREATER_THAN_OR_EQUAL_TO_TEMPLATE = "%s >= %s";
35      static final String BEGINS_WITH_TEMPLATE = "begins_with(%s, %s)";
36      static final String ATTRIBUTE_TYPE_TEMPLATE = "attribute_type(%s, %s)";
37      static final String CONTAINS_TEMPLATE = "contains(%s, %s)";
38  
39      private final String template;
40      private final String property;
41      private final AttributeValue value;
42  
43      ConditionalWithTwoArguments(String template, String property, AttributeValue value) {
44          this.template = template;
45          this.property = property;
46          this.value = value;
47      }
48  
49      @Override
50      public Expression expression(TableSchema<?> tableSchema, String indexName) {
51          String propertyKeyToken = QueryConditionalFactory.expressionKey(property);
52          String propertyValueToken = QueryConditionalFactory.expressionValue(property);
53          String queryExpression = String.format(template, propertyKeyToken, propertyValueToken);
54  
55          return Expression.builder()
56              .expression(queryExpression)
57              .expressionNames(Collections.singletonMap(propertyKeyToken, property))
58              .expressionValues(Collections.singletonMap(propertyValueToken, value))
59              .build();
60      }
61  
62  }
