18 package com.agorapulse.micronaut.http.examples.planets;
19 
20 import com.amazonaws.serverless.exceptions.ContainerInitializationException;
21 import com.amazonaws.services.lambda.runtime.Context;
22 import com.amazonaws.services.lambda.runtime.RequestStreamHandler;
23 import io.micronaut.context.ApplicationContext;
24 import io.micronaut.context.ApplicationContextBuilder;
25 import io.micronaut.function.aws.proxy.MicronautLambdaContainerHandler;
26 import org.slf4j.Logger;
27 import org.slf4j.LoggerFactory;
28 
29 import java.io.IOException;
30 import java.io.InputStream;
31 import java.io.OutputStream;
32 import java.util.function.Consumer;
33 
34 public class MicronautHandler implements RequestStreamHandler {
35 
36     private static final Logger LOGGER = LoggerFactory.getLogger(MicronautHandler.class);
37 
38     private static MicronautLambdaContainerHandler handler;
39     private static ApplicationContextBuilder builder;
40 
41     static {
42         reset();
43     }
44 
45     /**
46      * Resets the current handler. For testing purposes only.
47      */
48     public static void reset() {
49         reset(b -> {});
50     }
51 
52     /**
53      * Resets the current handler. For testing purposes only.
54      *
55      * @param configuration builder customizer
56      */
57     public static void reset(Consumer<ApplicationContextBuilder> configuration) {
58         try {
59             builder = ApplicationContext.builder();
60             configuration.accept(builder);
61             handler = new MicronautLambdaContainerHandler(builder);
62         } catch (ContainerInitializationException e) {
63             // if we fail here. We re-throw the exception to force another cold start
64             if (LOGGER.isErrorEnabled()) {
65                 LOGGER.error("Exception in container initialization", e);
66             }
67             throw new IllegalStateException("Could not initialize Micronaut", e);
68         }
69     }
70 
71     public static ApplicationContext getApplicationContext() {
72         if (handler == null) {
73             reset();
74         }
75         return handler.getApplicationContext();
76     }
77 
78     @Override
79     public void handleRequest(InputStream inputStream, OutputStream outputStream, Context context)
80             throws IOException {
81         handler.proxyStream(inputStream, outputStream, context);
82     }
83 }
