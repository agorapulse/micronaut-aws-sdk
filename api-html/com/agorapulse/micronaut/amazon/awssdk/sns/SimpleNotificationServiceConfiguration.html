018 package com.agorapulse.micronaut.amazon.awssdk.sns;
019 
020 import com.agorapulse.micronaut.amazon.awssdk.core.DefaultRegionAndEndpointConfiguration;
021 import io.micronaut.context.env.Environment;
022 
023 /**
024  * Default simple queue service configuration.
025  */
026 @SuppressWarnings("AbstractClassWithoutAbstractMethod")
027 public abstract class SimpleNotificationServiceConfiguration extends DefaultRegionAndEndpointConfiguration {
028 
029     protected SimpleNotificationServiceConfiguration(String prefix, Environment environment) {
030         apns = forPlatform(prefix, "apns", "ios", environment);
031         apnsSandbox = forPlatform(prefix, "apns-sandbox", "ios-sandbox", environment);
032         gcm = forPlatform(prefix, "gcm", "android", environment);
033         adm = forPlatform(prefix, "adm", "amazon", environment);
034     }
035 
036     @SuppressWarnings("DuplicateStringLiteral")
037     private static Application forPlatform(final String prefix, final String platform, final String fallbackPlatform, final Environment environment) {
038         return new Application(
039             environment.get(prefix + "." + platform + ".arn", String.class).orElseGet(() ->
040                 environment.get(prefix + "." + platform + ".applicationArn", String.class).orElseGet(() ->
041                     environment.get(prefix + "." + fallbackPlatform + ".arn", String.class).orElseGet(() ->
042                         environment.get(prefix + "." + fallbackPlatform + ".applicationArn", String.class).orElse(null)
043                     )
044                 )
045             )
046         );
047     }
048 
049     @Deprecated
050     public final Application getIos() {
051         return apns;
052     }
053 
054     @Deprecated
055     public final Application getIosSandbox() {
056         return apnsSandbox;
057     }
058 
059     @Deprecated
060     public final Application getAndroid() {
061         return gcm;
062     }
063 
064     @Deprecated
065     public final Application getAmazon() {
066         return adm;
067     }
068 
069     public Application getApns() {
070         return apns;
071     }
072 
073     public Application getApnsSandbox() {
074         return apnsSandbox;
075     }
076 
077     public Application getGcm() {
078         return gcm;
079     }
080 
081     public Application getAdm() {
082         return adm;
083     }
084 
085     public String getTopic() {
086         return topic;
087     }
088 
089     public void setTopic(String topic) {
090         this.topic = topic;
091     }
092 
093     private final Application apns;
094     private final Application apnsSandbox;
095     private final Application gcm;
096     private final Application adm;
097     private String topic = "";
098 
099     public static class Application {
100 
101         public Application(String arn) {
102             this.arn = arn;
103         }
104 
105         public final String getArn() {
106             return arn;
107         }
108 
109         private final String arn;
110 
111     }
