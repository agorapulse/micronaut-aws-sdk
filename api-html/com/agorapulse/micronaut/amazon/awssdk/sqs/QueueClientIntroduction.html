018 package com.agorapulse.micronaut.amazon.awssdk.sqs;
019 
020 import com.agorapulse.micronaut.amazon.awssdk.core.util.ConfigurationUtil;
021 import com.agorapulse.micronaut.amazon.awssdk.sqs.annotation.Queue;
022 import com.agorapulse.micronaut.amazon.awssdk.sqs.annotation.QueueClient;
023 import com.fasterxml.jackson.core.JsonProcessingException;
024 import com.fasterxml.jackson.databind.ObjectMapper;
025 import groovy.transform.Undefined;
026 import io.micronaut.aop.MethodInterceptor;
027 import io.micronaut.aop.MethodInvocationContext;
028 import io.micronaut.context.BeanContext;
029 import io.micronaut.context.annotation.Requires;
030 import io.micronaut.core.annotation.AnnotationValue;
031 import io.micronaut.core.type.Argument;
032 import io.micronaut.inject.qualifiers.Qualifiers;
033 import software.amazon.awssdk.services.sqs.SqsClient;
034 import software.amazon.awssdk.services.sqs.model.QueueDoesNotExistException;
035 
036 import jakarta.inject.Singleton;
037 import java.util.Map;
038 import java.util.Optional;
039 import java.util.function.Function;
040 
041 @Singleton
042 @Requires(classes = SqsClient.class)
043 public class QueueClientIntroduction implements MethodInterceptor<Object, Object> {
044 
045     private static final String GROUP = "group";
046     private static final String DELAY = "delay";
047 
048     private static final Function<String, Optional<String>> EMPTY_IF_UNDEFINED = (String s) -> Undefined.STRING.equals(s) ? Optional.empty() : Optional.of(s);
049     private static final Function<Integer, Optional<Integer>> EMPTY_IF_ZERO = (Integer i) -> i == 0 ? Optional.empty() : Optional.of(i);
050 
051     private static class QueueArguments {
052         Argument<?> message;
053         Argument<?> delay;
054         Argument<?> group;
055 
056         boolean isValid() {
057             return message != null;
058         }
059 
060     }
061 
062     private final BeanContext beanContext;
063     private final ObjectMapper objectMapper;
064 
065     public QueueClientIntroduction(BeanContext beanContext, ObjectMapper objectMapper) {
066         this.beanContext = beanContext;
067         this.objectMapper = objectMapper;
068     }
069 
070     @Override
071     public Object intercept(MethodInvocationContext<Object, Object> context) {
072         AnnotationValue<QueueClient> clientAnnotationValue = context.getAnnotation(QueueClient.class);
073 
074         if (clientAnnotationValue == null) {
075             throw new IllegalStateException("Invocation beanContext is missing required annotation QueueClient");
076         }
077 
078         String configurationName = clientAnnotationValue.getValue(String.class).orElse(ConfigurationUtil.DEFAULT_CONFIGURATION_NAME);
079         SimpleQueueService service = beanContext.getBean(
080             SimpleQueueService.class,
081             ConfigurationUtil.isDefaultConfigurationName(configurationName) ? null : Qualifiers.byName(configurationName)
082         );
083 
084         String queueName = clientAnnotationValue.get(QueueClient.Constants.QUEUE, String.class).flatMap(EMPTY_IF_UNDEFINED).orElse(null);
085         String group = clientAnnotationValue.get(QueueClient.Constants.GROUP, String.class).flatMap(EMPTY_IF_UNDEFINED).orElse(null);
086         Integer delay = clientAnnotationValue.get(QueueClient.Constants.DELAY, Integer.class).flatMap(EMPTY_IF_ZERO).orElse(0);
087 
088         AnnotationValue<Queue> queueAnnotationValue = context.getAnnotation(Queue.class);
089 
090         if (queueAnnotationValue != null) {
091             queueName = queueAnnotationValue.getRequiredValue(String.class);
092             group = queueAnnotationValue.get(QueueClient.Constants.GROUP, String.class).flatMap(EMPTY_IF_UNDEFINED).orElse(group);
093             delay = queueAnnotationValue.get(QueueClient.Constants.DELAY, Integer.class).flatMap(EMPTY_IF_ZERO).orElse(delay);
094         }
095 
096         if (queueName == null) {
097             queueName = service.getDefaultQueueName();
098         }
099 
100         try {
101             return doIntercept(context, service, queueName, group, delay);
102         } catch (QueueDoesNotExistException ignored) {
103             service.createQueue(queueName);
104             return doIntercept(context, service, queueName, group, delay);
105         }
106     }
107 
108     private Object doIntercept(MethodInvocationContext<Object, Object> context, SimpleQueueService service, String queueName, String group, Integer delay) {
109         Argument[] arguments = context.getArguments();
110         Map<String, Object> params = context.getParameterValueMap();
111 
112         if (arguments.length == 1 && context.getMethodName().startsWith("delete")) {
113             service.deleteMessage(queueName, String.valueOf(params.get(arguments[0].getName())));
114             return null;
115         }
116 
117         if (arguments.length >= 1 && arguments.length <= 3) {
118             QueueArguments queueArguments = findArguments(arguments);
119 
120 
121             if (queueArguments.delay != null) {
122                 Object delayParameter = params.get(queueArguments.delay.getName());
123                 delay = ((Number)delayParameter).intValue();
124             }
125 
126             if (queueArguments.group != null) {
127                 Object groupParameter = params.get(queueArguments.group.getName());
128                 group = String.valueOf(groupParameter);
129             }
130 
131             Object message = params.get(queueArguments.message.getName());
132             Class<?> messageType = queueArguments.message.getType();
133 
134             if (CharSequence.class.isAssignableFrom(messageType)) {
135                 return service.sendMessage(queueName, message.toString(), delay, group);
136             }
137 
138             if (messageType.isArray() && byte.class.equals(messageType.getComponentType())) {
139                 return service.sendMessage(queueName, new String((byte[]) message), delay, group);
140             }
141 
142             return sendJson(service, queueName, message, delay, group);
143         }
144 
145         throw new UnsupportedOperationException("Cannot implement method " + context.getExecutableMethod());
146     }
147 
148     private String sendJson(SimpleQueueService service, String queueName, Object message, int delay, String group) {
149         try {
150             return service.sendMessage(queueName, objectMapper.writeValueAsString(message), delay, group);
151         } catch (JsonProcessingException e) {
152             throw new IllegalArgumentException("Failed to marshal " + message + " to JSON", e);
153         }
154     }
155 
156     private QueueArguments findArguments(Argument[] arguments) {
157         QueueArguments names = new QueueArguments();
158 
159         for (Argument<?> argument : arguments) {
160             if (argument.getName().toLowerCase().contains(GROUP)) {
161                 names.group = argument;
162                 continue;
163             }
164             if (argument.getName().toLowerCase().contains(DELAY) || Number.class.isAssignableFrom(argument.getType())) {
165                 names.delay = argument;
166                 continue;
167             }
168             names.message = argument;
169         }
170 
171         if (!names.isValid()) {
172             throw new UnsupportedOperationException("Method needs to have at least one argument which name does not contain group or delay");
173         }
174 
175         return names;
176     }
