066     /**
067      * @param queueName the name of the queue
068      * @return queue URL
069      */
070     public String createQueue(String queueName) {
071         QueueConfiguration configuration = this.configuration.copy();
072         configuration.setQueue(queueName);
073 
074         if (queueName.endsWith(".fifo")) {
075             configuration.setFifo(true);
076         }
077 
078         return createQueue(configuration);
079     }

081     /**
082      * @param configuration the configuration of the queue
083      * @return queue URL
084      */
085     public String createQueue(QueueConfiguration configuration) {
086         CreateQueueRequest.Builder createQueueRequest = CreateQueueRequest.builder().queueName(configuration.getQueue());
087 
088         Map<QueueAttributeName, String> attributes = new HashMap<>();
089 
090         if (configuration.getDelaySeconds() != null && configuration.getDelaySeconds() != 0) {
091             attributes.put(QueueAttributeName.DELAY_SECONDS, String.valueOf(configuration.getDelaySeconds()));
092         }
093 
094         if (configuration.getMessageRetentionPeriod() != null && configuration.getMessageRetentionPeriod() != 0) {
095             attributes.put(QueueAttributeName.MESSAGE_RETENTION_PERIOD, String.valueOf(configuration.getMessageRetentionPeriod()));
096         }
097 
098         if (configuration.getMaximumMessageSize() != null && configuration.getMaximumMessageSize() != 0) {
099             attributes.put(QueueAttributeName.MAXIMUM_MESSAGE_SIZE, String.valueOf(configuration.getMaximumMessageSize()));
100         }
101 
102         if (configuration.getVisibilityTimeout() != null && configuration.getVisibilityTimeout() != 0) {
103             attributes.put(QueueAttributeName.VISIBILITY_TIMEOUT, String.valueOf(configuration.getVisibilityTimeout()));
104         }
105 
106         if (configuration.getFifo()) {
107             attributes.put(QueueAttributeName.FIFO_QUEUE, Boolean.TRUE.toString());
108         }
109 
110         if (!attributes.isEmpty()) {
111             createQueueRequest.attributes(attributes);
112         }
113 
114         String queueUrl = client.createQueue(createQueueRequest.build()).queueUrl();
115 
116         LOGGER.debug("Queue created (queueUrl={})", queueUrl);
117 
118         addQueue(queueUrl);
119 
120         return queueUrl;
121     }

142     /**
143      * @param queueName
144      * @return
145      */
146     public Map<QueueAttributeName, String> getQueueAttributes(String queueName) {
147         String queueUrl = getQueueUrl(queueName);
148 
149         GetQueueAttributesRequest.Builder getQueueAttributesRequest = GetQueueAttributesRequest
150             .builder()
151             .queueUrl(queueUrl)
152             .attributeNames(QueueAttributeName.ALL);
153 
154         try {
155             return client.getQueueAttributes(getQueueAttributesRequest.build()).attributes();
156         } catch (AwsServiceException exception) {
157             if ("AWS.SimpleQueueService.NonExistentQueue".equals(exception.awsErrorDetails().errorCode())) {
158                 removeQueue(queueUrl);
159             }
160             LOGGER.warn("An amazon service exception was caught while getting queue attributes", exception);
161             return Collections.emptyMap();
162         } catch (SdkClientException exception) {
163             LOGGER.warn("An amazon client exception was catched while getting queue attributes", exception);
164             return Collections.emptyMap();
165         }
166     }

168     /**
169      * @return the queue URL
170      */
171     public String getQueueUrl(String queueName) {
172         if (queueName.startsWith("http://") || queueName.startsWith("https://")) {
173             return queueName;
174         }
175 
176         if (StringUtils.isNotEmpty(configuration.getQueueNamePrefix())) {
177             queueName = configuration.getQueueNamePrefix() + queueName;
178         }
179 
180         if (queueUrlByNames.isEmpty() && isCaching()) {
181             loadQueues();
182         }
183 
184         String queueUrl = isCaching() ? queueUrlByNames.get(queueName) : getQueueUrlDirect(queueName);
185 
186         if (StringUtils.isEmpty(queueUrl) && (configuration.isAutoCreateQueue())) {
187             queueUrl = createQueue(queueName);
188         }
189 
190         if (StringUtils.isEmpty(queueUrl)) {
191             throw QueueDoesNotExistException.builder().message("Queue " + queueName + " not found").build();
192         }
193 
194         return queueUrl;
195     }

225     /**
226      * @param queueName
227      * @param maxNumberOfMessages
228      * @param visibilityTimeout

230      * @return
231      */
232     public List<Message> receiveMessages(String queueName, int maxNumberOfMessages, int visibilityTimeout, int waitTimeSeconds) {
233         String queueUrl = getQueueUrl(queueName);
234 
235         ReceiveMessageRequest.Builder receiveMessageRequest = ReceiveMessageRequest.builder().queueUrl(queueUrl);
236         if (maxNumberOfMessages > 0) {
237             receiveMessageRequest.maxNumberOfMessages(maxNumberOfMessages);
238         }
239         if (visibilityTimeout > 0) {
240             receiveMessageRequest.visibilityTimeout(visibilityTimeout);
241         }
242         if (waitTimeSeconds > 0) {
243             receiveMessageRequest.waitTimeSeconds(waitTimeSeconds);
244         }
245         List<Message> messages = client.receiveMessage(receiveMessageRequest.build()).messages();
246         LOGGER.debug("Messages received (count={})", messages.size());
247         return messages;
248     }

256     public String sendMessage(String queueName, String messageBody, int delaySeconds, String groupId) {
257         String queueUrl = getQueueUrl(queueName);
258

261             request.delaySeconds(delaySeconds);

263 
264         if (StringUtils.isNotEmpty(groupId)) {

266         }
267 
268         String messageId = client.sendMessage(request.build()).messageId();
269         LOGGER.debug("Message sent (messageId={})", messageId);

296     // PRIVATE
297     private static String getQueueNameFromUrl(String queueUrl) {
298         return Arrays.stream(queueUrl.split("/")).reduce((first, second) -> second).orElse(null);
299     }

301     private String getQueueUrlDirect(String queueName) {
302         try {
303             return client.getQueueUrl(b -> b.queueName(queueName)).queueUrl();
304         } catch (QueueDoesNotExistException ignored) {
305             return null;
306         }
307     }

309     private void addQueue(String queueUrl) {
310         if (StringUtils.isEmpty(queueUrl)) {
311             throw new IllegalStateException("Queue URL cannot be null or empty");
312         }
313 
314         synchronized (queueUrlByNames) {
315             queueUrlByNames.put(getQueueNameFromUrl(queueUrl),  queueUrl);
316         }
317     }
318 
319     private void loadQueues() {
320         ListQueuesRequest.Builder listQueuesRequest = ListQueuesRequest.builder();
321         if (StringUtils.isNotEmpty(configuration.getQueueNamePrefix())) {
322             listQueuesRequest.queueNamePrefix(configuration.getQueueNamePrefix());
323         }
324 
325         Map<String, String> queueUrls = client.listQueues(listQueuesRequest.build())
326             .queueUrls()
327             .stream()
328             .collect(
329                 Collectors.toMap(
330                     DefaultSimpleQueueService::getQueueNameFromUrl,
331                     Function.identity()
332                 )
333             );
334 
335         synchronized (queueUrlByNames) {
336             queueUrlByNames.clear();
337             queueUrlByNames.putAll(queueUrls);
338         }
339     }
340 
341     private void removeQueue(String queueUrl) {
342         if (StringUtils.isEmpty(queueUrl)) {
343             throw new IllegalStateException("Queue URL cannot be null or empty");
344         }
345 
346         synchronized (queueUrlByNames) {
347             queueUrlByNames.remove(getQueueNameFromUrl(queueUrl));
348         }
349     }
