102     @Override
103     public String decodeRecordData(final Record record) {
104         String data = "";
105         try {
106             // Payload as UTF-8 chars.
107             data = decoder.decode(record.data().asByteBuffer()).toString();
108         } catch (CharacterCodingException e) {
109             LOGGER.error("Malformed data for record: " + record, e);
110         }
111 
112         return data;
113     }

120     @Override
121     public DescribeStreamResponse describeStream(String streamName) {
122         DescribeStreamRequest.Builder request = DescribeStreamRequest.builder().streamName(streamName);
123         DescribeStreamResponse describeStreamResult;
124 
125         String exclusiveStartShardId = null;
126         List<Shard> shards = new ArrayList<>();
127         while (true) {
128             request.exclusiveStartShardId(exclusiveStartShardId);
129             describeStreamResult = client.describeStream(request.build());
130             shards.addAll(describeStreamResult.streamDescription().shards());
131             if (!shards.isEmpty() && describeStreamResult.streamDescription().hasMoreShards() == Boolean.TRUE) {
132                 exclusiveStartShardId = shards.get(shards.size() - 1).shardId();
133             } else {
134                 break;
135             }
136         }
137 
138         return describeStreamResult.toBuilder().streamDescription(
139             describeStreamResult.streamDescription().toBuilder().shards(shards).build()
140         ).build();
141     }

143     @Override
144     public Shard getShard(String streamName, final String shardId) {
145         DescribeStreamResponse describeStreamResult = describeStream(streamName);
146         if (!describeStreamResult.streamDescription().shards().isEmpty()) {
147             Optional<Shard> shard = describeStreamResult
148                 .streamDescription()
149                 .shards()
150                 .stream()
151                 .filter(it -> it.shardId().equals(shardId))
152                 .findFirst();
153             if (shard.isPresent()) {
154                 return shard.get();
155             }
156         }
157 
158         return null;
159     }

166     @Override
167     public Publisher<Record> getShardRecords(String streamName, Shard shard, ShardIteratorType shardIteratorType, String startingSequenceNumber, int batchLimit) {
168         return Flux
169             .generate(() -> {
170                 GetShardIteratorRequest.Builder getShardIteratorRequest = GetShardIteratorRequest.builder()
171                     .streamName(streamName)
172                     .shardId(shard.shardId())
173                     .shardIteratorType(shardIteratorType.toString());
174                 if (shardIteratorType == ShardIteratorType.AFTER_SEQUENCE_NUMBER || shardIteratorType == ShardIteratorType.AT_SEQUENCE_NUMBER) {
175                     if (startingSequenceNumber == null || startingSequenceNumber.length() == 0) {
176                         throw new IllegalArgumentException("Starting sequence number must not be null!");
177                     }
178                     getShardIteratorRequest.startingSequenceNumber(Objects.requireNonNull(startingSequenceNumber));
179                 }
180 
181                 GetShardIteratorResponse getShardIteratorResult = client.getShardIterator(getShardIteratorRequest.build());
182                 return getShardIteratorResult.shardIterator();
183             }, (String shardIterator, SynchronousSink<List<Record>> recordEmitter) -> {
184                 GetRecordsRequest.Builder getRecordsRequest = GetRecordsRequest.builder().shardIterator(shardIterator);
185 
186                 if (batchLimit > 0) {
187                     getRecordsRequest.limit(batchLimit);
188                 }
189 
190                 GetRecordsResponse getRecordsResult = client.getRecords(getRecordsRequest.build());
191 
192                 List<Record> records = getRecordsResult.records();
193                 if (records.isEmpty()) {
194                     return shardIterator;
195                 }
196                 recordEmitter.next(records);
197                 return getRecordsResult.nextShardIterator();
198             })
199             .delayUntil(records -> {
200                 if (records.isEmpty()) {
201                     return Mono.just(records).delayElement(Duration.ofMillis(DEFAULT_GET_RECORDS_WAIT));
202                 }
203 
204                 return Mono.just(records);
205             })
206             .flatMap(Flux::fromIterable);
207     }

209     @Override
210     public PutRecordResponse putEvent(String streamName, Event event, String sequenceNumberForOrdering) {
211         if (!StringUtils.isEmpty(configuration.getConsumerFilterKey())) {
212             event.setConsumerFilterKey(configuration.getConsumerFilterKey());
213         }
214 
215         try {
216             return putRecord(streamName, event.getPartitionKey(), objectMapper.writeValueAsString(event), sequenceNumberForOrdering);
217         } catch (JsonProcessingException e) {
218             throw new IllegalArgumentException("Cannot write value as JSON: " + event, e);
219         }
220     }
221 
222     @Override
223     public PutRecordsResponse putEvents(String streamName, List<Event> events) {
224         if (events.size() > MAX_PUT_RECORDS_SIZE) {
225             throw new IllegalArgumentException("Max put events size is " + MAX_PUT_RECORDS_SIZE);
226         }
227         return putRecords(streamName, events.stream().map(event -> {
228             if (!StringUtils.isEmpty(configuration.getConsumerFilterKey())) {
229                 event.setConsumerFilterKey(configuration.getConsumerFilterKey());
230             }
231 
232             try {
233                 return PutRecordsRequestEntry.builder()
234                     .data(SdkBytes.fromByteArray(objectMapper.writeValueAsString(event).getBytes()))
235                     .partitionKey(event.getPartitionKey())
236                     .build();
237             } catch (JsonProcessingException e) {
238                 throw new IllegalArgumentException("Cannot write value as JSON: " + event, e);
239             }
240         }).collect(Collectors.toList()));
241     }

267     @Override
268     public SplitShardResponse splitShard(String streamName, Shard shard, String newStartingHashKey) {
269         if (StringUtils.isEmpty(newStartingHashKey)) {
270             // Determine the hash key value that is half-way between the lowest and highest values in the shard
271             BigInteger startingHashKey = new BigInteger(shard.hashKeyRange().startingHashKey());
272             BigInteger endingHashKey = new BigInteger(shard.hashKeyRange().endingHashKey());
273             newStartingHashKey = startingHashKey.add(endingHashKey).divide(BigInteger.valueOf(2)).toString();
274         }
275 
276         return client.splitShard(SplitShardRequest.builder()
277             .streamName(streamName)
278             .shardToSplit(shard.shardId())
279             .newStartingHashKey(newStartingHashKey).build());
280     }

282     @Override
283     public void waitForStatus(String streamName, StreamStatus status) {
284         while (!status.equals(describeStream(streamName).streamDescription().streamStatus())) {
285             try {
286                 Thread.sleep(100);
287             } catch (InterruptedException e) {
288                 throw new IllegalStateException("Waiting for stream to become active was interrupted!", e);
289             }
290         }
