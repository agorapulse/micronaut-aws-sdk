18 package com.agorapulse.micronaut.amazon.awssdk.core;
19 
20 import io.micronaut.context.env.Environment;
21 import software.amazon.awssdk.auth.credentials.AwsBasicCredentials;
22 import software.amazon.awssdk.auth.credentials.AwsCredentials;
23 import software.amazon.awssdk.auth.credentials.AwsCredentialsProvider;
24 import software.amazon.awssdk.auth.credentials.AwsSessionCredentials;
25 import software.amazon.awssdk.core.exception.SdkClientException;
26 import software.amazon.awssdk.utils.StringUtils;
27 
28 /**
29  * A {@link AwsCredentialsProvider} that reads from the {@link Environment}.
30  */
31 public class EnvironmentAwsCredentialsProvider implements AwsCredentialsProvider {
32 
33     /**
34      * Environment variable name for the AWS access key ID.
35      */
36     public static final String ACCESS_KEY_ENV_VAR = "aws.access-key-id";
37 
38     /**
39      * Alternate environment variable name for the AWS access key ID.
40      */
41     public static final String ALTERNATE_ACCESS_KEY_ENV_VAR = "aws.access-key";
42 
43     /**
44      * Environment variable name for the AWS secret key.
45      */
46     public static final String SECRET_KEY_ENV_VAR = "aws.secret-key";
47 
48     /**
49      * Alternate environment variable name for the AWS secret key.
50      */
51     public static final String ALTERNATE_SECRET_KEY_ENV_VAR = "aws.secret-access-key";
52 
53     /**
54      * Environment variable name for the AWS session token.
55      */
56     public static final String AWS_SESSION_TOKEN_ENV_VAR = "aws.session-token";
57 
58     public static AwsCredentialsProvider create(Environment environment) {
59         return new EnvironmentAwsCredentialsProvider(environment);
60     }
61 
62     private final Environment environment;
63 
64     /**
65      * Constructor.
66      * @param environment environment
67      */
68     private EnvironmentAwsCredentialsProvider(Environment environment) {
69         this.environment = environment;
70     }
71 
72     @Override
73     public AwsCredentials resolveCredentials() {
74         String accessKey = environment.getProperty(ACCESS_KEY_ENV_VAR, String.class, environment.getProperty(ALTERNATE_ACCESS_KEY_ENV_VAR, String.class, (String) null));
75 
76         String secretKey = environment.getProperty(SECRET_KEY_ENV_VAR, String.class, environment.getProperty(ALTERNATE_SECRET_KEY_ENV_VAR, String.class, (String) null));
77         accessKey = StringUtils.trim(accessKey);
78         secretKey = StringUtils.trim(secretKey);
79         String sessionToken = StringUtils.trim(environment.getProperty(AWS_SESSION_TOKEN_ENV_VAR, String.class, (String) null));
80 
81         if (StringUtils.isBlank(accessKey) || StringUtils.isBlank(secretKey)) {
82             throw SdkClientException.create(
83                 "Unable to load AWS credentials from environment "
84                     + "(" + ACCESS_KEY_ENV_VAR + " (or " + ALTERNATE_ACCESS_KEY_ENV_VAR + ") and "
85                     + SECRET_KEY_ENV_VAR + " (or " + ALTERNATE_SECRET_KEY_ENV_VAR + "))");
86         }
87 
88         return sessionToken == null
89             ? AwsBasicCredentials.create(accessKey, secretKey)
90             : AwsSessionCredentials.create(accessKey, secretKey, sessionToken);
91     }
92 
93     @Override
94     public String toString() {
95         return getClass().getSimpleName();
96     }
97 
98 }
