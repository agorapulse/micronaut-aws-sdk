018 package com.agorapulse.micronaut.amazon.awssdk.dynamodb;
019 
020 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.Builders;
021 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.DetachedQuery;
022 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.DetachedScan;
023 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.DetachedUpdate;
024 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.QueryBuilder;
025 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.ScanBuilder;
026 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder.UpdateBuilder;
027 import io.reactivex.Flowable;
028 import software.amazon.awssdk.enhanced.dynamodb.DynamoDbTable;
029 import software.amazon.awssdk.enhanced.dynamodb.Key;
030 
031 import javax.annotation.Nullable;
032 import java.util.function.Consumer;
033 import java.util.function.Function;
034 
035 /**
036  * Middle-level service for interaction with the DynamoDB.
037  *
038  * @param <T> the type of the items.
039  * @see DynamoDBServiceProvider for obtaining instances of this class.
040  */
041 public interface DynamoDbService<T> {
042 
043     /**
044      * @return the type of the items handled by this service
045      */
046     Class<T> getItemType();
047 
048     /**
049      * Returns the low-level {@link DynamoDbTable} instance to execute operations not covered by this service.
050      *
051      * Warning: no events are triggered while using the low level API.
052      *
053      * @return the low-level {@link DynamoDbTable} instance to execute operations not covered by this service
054      */
055     DynamoDbTable<T> getTable();
056 
057     /**
058      * Executes the prepared query and returns the items matching the query.
059      * @param query the query
060      * @return the items matching the given query
061      */
062     Flowable<T> query(DetachedQuery<T> query);
063 
064     /**
065      * Defines the query using the query builder and returns the items matching the query.
066      * @param query the query definition
067      * @return the items matching the given query
068      */
069     default Flowable<T> query(Consumer<QueryBuilder<T>> query) {
070         return query(Builders.query(query));
071     }
072 
073     /**
074      * Executes the prepared scan (non-index query) and returns the items matching the scan.
075      * @param scan the scan
076      * @return the items matching the scan
077      */
078     Flowable<T> scan(DetachedScan<T> scan);
079 
080     /**
081      * Defines the scan (non-index query) using the scan builder and returns the items matching the scan.
082      * @param scan the scan definition
083      * @return the items matching the scan
084      */
085     default Flowable<T> scan(Consumer<ScanBuilder<T>> scan) {
086         return scan(Builders.scan(scan));
087     }
088 
089     /**
090      * Finds all the items for given partition key.
091      *
092      * If the sort key is present it either returns {@link Flowable} with single item or an empty one.
093      *
094      * @param partitionKey the partition key
095      * @param sortKey the sort key
096      * @return flowable of all items with given partition key and sort key (if present)
097      */
098     Flowable<T> findAll(Object partitionKey, @Nullable Object sortKey);
099 
100     /**
101      * Finds all the items for given partition key.
102      *
103      * @param partitionKey the partition key
104      * @return flowable of all items with given partition key
105      */
106     default Flowable<T> findAll(Object partitionKey) {
107         return findAll(partitionKey, null);
108     }
109 
110     /**
111      * Updates the item using the given update definition and returns the result according it's return definition.
112      * @param update the update definition
113      * @return the result according to update definition's return settings.
114      */
115     <R> R update(DetachedUpdate<T, R> update);
116 
117     /**
118      * Updates the item using the given update definition and returns the result according it's return definition.
119      * @param update the update definition
120      * @return the result according to update definition's return settings.
121      */
122     default <R> R update(Function<UpdateBuilder<T, T>, UpdateBuilder<T, R>> update) {
123         return update(Builders.update(update));
124     }
125 
126     int updateAll(Flowable<T> items, UpdateBuilder<T, ?> update);
127 
128     default <R> int updateAll(Flowable<T> items, Function<UpdateBuilder<T, T>, UpdateBuilder<T, R>> update) {
129         return updateAll(items, Builders.update(update));
130     }
131 
132     T save(T entity);
133 
134     Flowable<T> saveAll(Flowable<T> itemsToSave);
135 
136     T delete(Object partitionKey, @Nullable Object sortKey);
137 
138     T delete(T item);
139 
140     T delete(Key key);
141 
142     int deleteAll(Flowable<T> items);
143 
144     T get(Object partitionKey, Object sortKey);
145 
146     Flowable<T> getAll(Object partitionKey, Flowable<?> sortKeys);
147 
148     T get(Key key);
149 
150     int count(DetachedQuery<T> query);
151 
152     default int countUsingQuery(Consumer<QueryBuilder<T>> query) {
153         return count(Builders.query(query));
154     }
155 
156     int count(DetachedScan<T> scan);
157 
158     default int countUsingScan(Consumer<ScanBuilder<T>> scan) {
159         return count(Builders.scan(scan));
160     }
161 
162     int count(Object partitionKey, @Nullable Object sortKey);
163 
164     void createTable();
165 }
