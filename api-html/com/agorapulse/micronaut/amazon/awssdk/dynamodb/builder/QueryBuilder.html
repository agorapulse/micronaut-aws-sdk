018 package com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder;
019 
020 import software.amazon.awssdk.enhanced.dynamodb.model.QueryEnhancedRequest;
021 
022 import java.util.Arrays;
023 import java.util.Collection;
024 import java.util.function.Consumer;
025 
026 /**
027  * Builder for DynamoDB queries.
028  * @param <T> type of the DynamoDB entity
029  */
030 public interface QueryBuilder<T> extends DetachedQuery<T> {
031 
032     /**
033      * Sort the results by the sort key
034      * @param sort the sort keyword
035      * @return self
036      * @deprecated use {@link #order(Builders.Sort)} instead
037      */
038     @Deprecated
039     default QueryBuilder<T> sort(Builders.Sort sort) {
040         return order(sort);
041     }
042 
043     /**
044      * Orders the results by the sort key
045      * @param sort the order keyword
046      * @return self
047      */
048     QueryBuilder<T> order(Builders.Sort sort);
049 
050     /**
051      * Demand consistent reads.
052      * @param read the read keyword
053      * @return self
054      */
055     QueryBuilder<T> consistent(Builders.Read read);
056 
057     /**
058      * Demand inconsistent reads.
059      * @param read the read keyword
060      * @return self
061      */
062     QueryBuilder<T> inconsistent(Builders.Read read);
063 
064     /**
065      * Select the index on which this query will be executed.
066      * @param name the name of the index to be used
067      * @return self
068      */
069     QueryBuilder<T> index(String name);
070 
071     /**
072      * Sets the partition key value for the query.
073      *
074      * This parameter is required for every query.
075      *
076      * @param key the partition key of the query
077      * @return self
078      * @deprecated use {@link #partitionKey(Object)} instead
079      */
080     @Deprecated
081     default QueryBuilder<T> hash(Object key) {
082         return partitionKey(key);
083     }
084 
085     /**
086      * Sets the partition key value for the query.
087      *
088      * This parameter is required for every query.
089      *
090      * @param key the partition key of the query
091      * @return self
092      */
093     QueryBuilder<T> partitionKey(Object key);
094 
095     /**
096      * Creates a sort key condition.
097      * @param conditions consumer to build the conditions
098      * @return self
099      */
100     QueryBuilder<T> sortKey(Consumer<KeyConditionCollector<T>> conditions);
101 
102     /**
103      * Creates a sort key condition.
104      * @param conditions consumer to build the conditions
105      * @return self
106      * @deprecated use {@link #sortKey(Consumer)} instead
107      */
108     @Deprecated
109     default QueryBuilder<T> range(Consumer<KeyConditionCollector<T>> conditions) {
110         return sortKey(conditions);
111     }
112 
113     /**
114      * One or more filter conditions.
115      *
116      * These conditions are resolved on the result set before returning the values and therefore they don't require an existing index
117      * but they consume more resources as all the result set must be traversed.
118      *
119      * @param conditions consumer to build the conditions
120      * @return self
121      */
122     QueryBuilder<T> filter(Consumer<FilterConditionCollector<T>> conditions);
123 
124     /**
125      * Sets the desired pagination of the queries.
126      *
127      * This only sets the optimal pagination of the queries and does not limit the number of items returned.
128      *
129      * Use <code>{@link io.reactivex.Flowable#take(long)}</code> to limit the number results returned from the query.
130      *
131      * @param page number of entities loaded by one query request (not a number of total entities returned)
132      * @return self
133      */
134     QueryBuilder<T> page(int page);
135 
136     /**
137      * Sets the maximum number of items to be returned from the queries.
138      *
139      * This is a shortcut for calling <code>{@link io.reactivex.Flowable#take(long)}</code> on the result Flowable.
140      *
141      * @param max the maximum number of items returned
142      * @return self
143      */
144     QueryBuilder<T> limit(int max);
145 
146 
147     /**
148      * Sets the query offset by defining the exclusive start value.
149      * @param lastEvaluatedKey exclusive start value
150      * @return self
151      */
152     QueryBuilder<T> lastEvaluatedKey(Object lastEvaluatedKey);
153 
154     /**
155      * Configures the native query expression.
156      *
157      * This method is an extension point which allows to configure properties which are not provides by this builder.
158      *
159      * @param configurer consumer to configure the native query expression
160      * @return self
161      */
162     QueryBuilder<T> configure(Consumer<QueryEnhancedRequest.Builder> configurer);
163 
164     /**
165      * Limits which properties of the returned entities will be populated.
166      * @param propertyPaths property paths to be populated in the returned entities
167      * @return self
168      */
169     QueryBuilder<T> only(Collection<String> propertyPaths);
170 
171     /**
172      * Limits which properties of the returned entities will be populated.
173      * @param propertyPaths property paths to be populated in the returned entities
174      * @return self
175      */
176     default QueryBuilder<T> only(String... propertyPaths) {
177         return only(Arrays.asList(propertyPaths));
178     }
179 
180 }
