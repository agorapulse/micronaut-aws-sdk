130     @Override
131     public Flowable<T> query(DynamoDbTable<T> mapper, AttributeConversionHelper attributeConversionHelper) {
132         QueryEnhancedRequest request = resolveRequest(mapper, attributeConversionHelper);
133         SdkIterable<Page<T>> iterable = this.__index == null ? mapper.query(request) : mapper.index(__index).query(request);
134         Flowable<T> results = fromIterable(iterable).flatMap(p -> fromIterable(p.items()));
135         if (__max < Integer.MAX_VALUE) {
136             return results.take(__max);
137         }
138         return results;
139     }

141     @Override
142     public QueryEnhancedRequest resolveRequest(MappedTableResource<T> mapper, AttributeConversionHelper attributeConversionHelper) {
143         applyRangeConditions(mapper, attributeConversionHelper, __expression::queryConditional);
144         String currentIndex = __index == null ? TableMetadata.primaryIndexName() : __index;
145         applyFilterConditions(mapper, attributeConversionHelper, cond -> __expression.filterExpression(cond.expression(mapper.tableSchema(), currentIndex)));
146         applyLastEvaluatedKey(__expression, mapper);
147         __configurer.accept(__expression);
148 
149         return __expression.build();
150     }

164     private void applyRangeConditions(
165         MappedTableResource<T> model,
166         AttributeConversionHelper attributeConversionHelper,
167         Consumer<QueryConditional> addFilterConsumer
168     ) {
169 
170         DefaultKeyConditionCollector<T> rangeCollector = new DefaultKeyConditionCollector<>(model, attributeConversionHelper, __index);
171 
172         if (!__queryConditionals.isEmpty()) {
173 
174             for (Consumer<KeyConditionCollector<T>> consumer : __queryConditionals) {
175                 consumer.accept(rangeCollector);
176             }
177 
178         }
179 
180         String partitionKey = model.tableSchema().tableMetadata().indexPartitionKey(__index);
181         QueryConditional hashCondition = QueryConditionalFactory.equalTo(partitionKey, attributeConversionHelper.convert(model, partitionKey, __hash));
182 
183         addFilterConsumer.accept(QueryConditionalFactory.and(hashCondition, rangeCollector.getCondition()));
184     }

186     private void applyFilterConditions(
187         MappedTableResource<T> model,
188         AttributeConversionHelper attributeConversionHelper,
189         Consumer<QueryConditional> addFilterConsumer
190     ) {
191         if (!__filterCollectorsConsumers.isEmpty()) {
192             DefaultFilterConditionCollector<T> filterCollector = new DefaultFilterConditionCollector<>(model, attributeConversionHelper);
193 
194             for (Consumer<FilterConditionCollector<T>> consumer : __filterCollectorsConsumers) {
195                 consumer.accept(filterCollector);
196             }
197 
198             addFilterConsumer.accept(filterCollector.getCondition());
199         }
200     }

202     private void applyLastEvaluatedKey(QueryEnhancedRequest.Builder exp, MappedTableResource<T> mapper) {
203         if (__lastEvaluatedKey == null) {
204             return;
205         }
206 
207         Map<String, AttributeValue> key;
208         TableSchema<T> schema = mapper.tableSchema();
209 
210         if (__lastEvaluatedKey instanceof Map) {
211             key = (Map<String, AttributeValue>) __lastEvaluatedKey;
212         } else {
213             key = schema.itemToMap((T) __lastEvaluatedKey, true);
214         }
215 
216         Set<String> indexKeys = new HashSet<>(schema.tableMetadata().primaryKeys());
217 
218         if (__index != null) {
219             indexKeys.addAll(schema.tableMetadata().indexKeys(__index));
220         }
221 
222         key = key.entrySet().stream().filter(e -> indexKeys.contains(e.getKey())).collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
223 
224         exp.exclusiveStartKey(key);
225     }
