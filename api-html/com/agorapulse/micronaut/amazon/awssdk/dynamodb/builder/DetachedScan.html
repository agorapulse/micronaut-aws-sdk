18 package com.agorapulse.micronaut.amazon.awssdk.dynamodb.builder;
19 
20 import com.agorapulse.micronaut.amazon.awssdk.dynamodb.AttributeConversionHelper;
21 import io.reactivex.Flowable;
22 import software.amazon.awssdk.enhanced.dynamodb.DynamoDbTable;
23 import software.amazon.awssdk.enhanced.dynamodb.MappedTableResource;
24 import software.amazon.awssdk.enhanced.dynamodb.model.ScanEnhancedRequest;
25 
26 /**
27  * An interface for scans which can be executed using supplied mapper.
28  * @param <T> type of the DynamoDB entity
29  */
30 public interface DetachedScan<T> {
31 
32     /**
33      * Executes a scan using provided mapper.
34      * @param mapper DynamoDB mapper
35      * @return flowable of entities found for the current scan
36      */
37     Flowable<T> scan(DynamoDbTable<T> mapper, AttributeConversionHelper attributeConversionHelper);
38 
39     /**
40      * Counts entities satisfying given scan using provided mapper.
41      * @param mapper DynamoDB mapper
42      * @return count of entities satisfying  for the current scan
43      */
44     int count(DynamoDbTable<T> mapper, AttributeConversionHelper attributeConversionHelper);
45 
46     /**
47      * Resolves the current scan into native scan expression using provided mapper.
48      * @param mapper DynamoDB mapper
49      * @return the current scan resolved into native scan expression
50      */
51     ScanEnhancedRequest resolveRequest(MappedTableResource<T> mapper, AttributeConversionHelper attributeConversionHelper);
52 
53 }
