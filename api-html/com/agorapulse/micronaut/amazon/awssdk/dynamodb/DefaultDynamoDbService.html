120     @Override
121     public int updateAll(Flowable<T> items, UpdateBuilder<T, ?> update) {
122         // there is no batch update API, we can do batch updates in transaction but in that case it would cause
123         // doubling the writes
124 
125         BeanIntrospection<T> introspection = EntityIntrospection.getBeanIntrospection(table);
126         TableMetadata tableMetadata = table.tableSchema().tableMetadata();
127 
128         AtomicInteger counter = new AtomicInteger();
129 
130         items.map(this::postLoad).subscribe(entity -> {
131             introspection.getProperty(tableMetadata.primaryPartitionKey()).ifPresent(p -> update.partitionKey(p.get(entity)));
132             tableMetadata.primarySortKey().flatMap(introspection::getProperty).ifPresent(p -> update.sortKey(p.get(entity)));
133 
134             update.update(table, client, attributeConversionHelper, publisher);
135 
136             counter.incrementAndGet();
137         });
138 
139         return counter.get();
140     }

150     @Override
151     public Flowable<T> saveAll(Flowable<T> itemsToSave) {
152         List<T> saved = new ArrayList<>();
153         List<T> unprocessed = itemsToSave.buffer(BATCH_SIZE).map(batchItems -> enhancedClient.batchWriteItem(b -> {
154             b.writeBatches(batchItems.stream().map(i -> {
155                 publisher.publishEvent(DynamoDbEvent.prePersist(i));
156                 saved.add(i);
157                 return WriteBatch.builder(table.tableSchema().itemType().rawClass()).mappedTableResource(table).addPutItem(i).build();
158             }).collect(Collectors.toList()));
159         })).flatMap(r -> Flowable.fromIterable(r.unprocessedPutItemsForTable(table))).toList().blockingGet();
160 
161         if (unprocessed.isEmpty()) {
162             saved.forEach(i -> publisher.publishEvent(DynamoDbEvent.postPersist(i)));
163             return Flowable.fromIterable(saved);
164         }
165 
166         throw new IllegalArgumentException("Following items couldn't be saved:" + unprocessed.stream().map(Object::toString).collect(Collectors.joining(", ")));
167     }

191     @Override
192     public int deleteAll(Flowable<T> items) {
193         TableSchema<T> tableSchema = table.tableSchema();
194         List<T> deleted = new ArrayList<>();
195         List<Key> unprocessed = items.buffer(BATCH_SIZE).map(batchItems -> enhancedClient.batchWriteItem(b -> {
196             b.writeBatches(batchItems.stream().map(i -> {
197                     publisher.publishEvent(DynamoDbEvent.preRemove(i));
198                     deleted.add(i);
199                     return WriteBatch.builder(tableSchema.itemType().rawClass()).mappedTableResource(table).addDeleteItem(i).build();
200                 }
201             ).collect(Collectors.toList()));
202         })).flatMap(r -> Flowable.fromIterable(r.unprocessedDeleteItemsForTable(table))).toList().blockingGet();
203 
204         if (unprocessed.isEmpty()) {
205             deleted.forEach(i -> publisher.publishEvent(DynamoDbEvent.postRemove(i)));
206             return deleted.size();
207         }
208 
209         throw new IllegalArgumentException("Following items couldn't be deleted:" + unprocessed.stream()
210             .map(k -> tableSchema.mapToItem(k.keyMap(tableSchema, TableMetadata.primaryIndexName()))).map(Object::toString)
211             .collect(Collectors.joining(", ")));
212     }

246     @Override
247     public void createTable() {
248         Map<String, ProjectionType> types = getProjectionTypes();
249         TableMetadata tableMetadata = table.tableSchema().tableMetadata();
250         tableMetadata.allKeys();
251         table.createTable(b -> {
252             List<EnhancedLocalSecondaryIndex> localSecondaryIndices = new ArrayList<>();
253             List<EnhancedGlobalSecondaryIndex> globalSecondaryIndices = new ArrayList<>();
254 
255             getIndices(tableMetadata).forEach(i -> {
256                 if (TableMetadata.primaryIndexName().equals(i)) {
257                     return;
258                 }
259                 ProjectionType type = types.getOrDefault(i, ProjectionType.KEYS_ONLY);
260                 if (tableMetadata.primaryPartitionKey().equals(tableMetadata.indexPartitionKey(i))) {
261                     localSecondaryIndices.add(EnhancedLocalSecondaryIndex.create(i, Projection.builder().projectionType(type).build()));
262                 } else {
263                     globalSecondaryIndices.add(EnhancedGlobalSecondaryIndex.builder().indexName(i).projection(Projection.builder().projectionType(type).build()).build());
264                 }
265             });
266 
267             if (!localSecondaryIndices.isEmpty()) {
268                 b.localSecondaryIndices(localSecondaryIndices);
269             }
270 
271             if (!globalSecondaryIndices.isEmpty()) {
272                 b.globalSecondaryIndices(globalSecondaryIndices);
273             }
274         });
275 
276     }

278     private DetachedQuery<T> simplePartitionAndSort(Object partitionKey, Object sortKey) {
279         return doWithKey(partitionKey, sortKey, key -> {
280             if (key.sortKeyValue().isPresent()) {
281                 return Builders.query(q -> q.partitionKey(key.partitionKeyValue()).sortKey(s -> s.eq(key.sortKeyValue().get())));
282             }
283             return Builders.query(q -> q.partitionKey(key.partitionKeyValue()));
284         });
285     }

287     private Flowable<T> getAll(AttributeValue hashKey, Flowable<AttributeValue> rangeKeys) {
288         TableSchema<T> tableSchema = table.tableSchema();
289         return rangeKeys.buffer(BATCH_SIZE).map(batchRangeKeys -> enhancedClient.batchGetItem(b -> {
290             b.readBatches(batchRangeKeys.stream().map(k ->
291                 ReadBatch.builder(tableSchema.itemType().rawClass()).mappedTableResource(table).addGetItem(Key.builder().partitionValue(hashKey).sortValue(k).build()).build()
292             ).collect(Collectors.toList()));
293         })).flatMap(r -> Flowable.fromIterable(r.resultsForTable(table)).map(this::postLoad));
294     }

296     private Map<String, ProjectionType> getProjectionTypes() {
297         Map<String, ProjectionType> types = new HashMap<>();
298 
299         BeanIntrospection<T> introspection = EntityIntrospection.getBeanIntrospection(table);
300         introspection.getBeanProperties().forEach(p -> {
301             AnnotationValue<com.agorapulse.micronaut.amazon.awssdk.dynamodb.annotation.Projection> projectionAnnotation = p.getAnnotation(com.agorapulse.micronaut.amazon.awssdk.dynamodb.annotation.Projection.class);
302 
303             if (projectionAnnotation == null) {
304                 return;
305             }
306 
307             AnnotationValue<DynamoDbSecondarySortKey> secondaryIndex = p.getAnnotation(DynamoDbSecondarySortKey.class);
308             Collection<String> indexNames = new ArrayList<>();
309             if (secondaryIndex != null) {
310                 indexNames.addAll(Arrays.asList(secondaryIndex.stringValues("indexNames")));
311             }
312 
313             AnnotationValue<DynamoDbSecondaryPartitionKey> secondaryGlobalIndex = p.getAnnotation(DynamoDbSecondaryPartitionKey.class);
314             if (secondaryGlobalIndex != null) {
315                 indexNames.addAll(Arrays.asList(secondaryGlobalIndex.stringValues("indexNames")));
316             }
317 
318             if (indexNames.isEmpty()) {
319                 return;
320             }
321 
322             ProjectionType type = projectionAnnotation.enumValue(ProjectionType.class).orElse(ProjectionType.KEYS_ONLY);
323             for (String name : indexNames) {
324                 types.put(name, type);
325             }
326         });
327 
328         return types;
329     }

331     private Set<String> getIndices(TableMetadata metadata) {
332         if (metadata instanceof StaticTableMetadata) {
333             try {
334                 Field indexByNameMapField = StaticTableMetadata.class.getDeclaredField("indexByNameMap");
335                 indexByNameMapField.setAccessible(true);
336                 Map<String, ?> indexByNameMap = (Map<String, ?>) indexByNameMapField.get(metadata);
337                 return indexByNameMap.keySet();
338             } catch (NoSuchFieldException | IllegalAccessException e) {
339                 LOGGER.error("Exception reading indices", e);
340             }
341         }
342 
343         return Collections.emptySet();
344     }

351     private <R> R doWithKey(Object partitionKey, Object sortKey, Function<Key, R> function) {
352         String hashKeyName = table.tableSchema().tableMetadata().primaryPartitionKey();
353         AttributeValue partitionKeyValue = attributeConversionHelper.convert(table, hashKeyName, partitionKey);
354 
355         if (sortKey == null) {
356             return function.apply(Key.builder().partitionValue(partitionKeyValue).build());
357         }
358 
359         String rangeKeyName = table.tableSchema().tableMetadata().primarySortKey().get();
360         AttributeValue sortKeyValue = attributeConversionHelper.convert(table, rangeKeyName, sortKey);
361         return function.apply(
362             Key.builder()
363                 .partitionValue(partitionKeyValue)
364                 .sortValue(sortKeyValue)
365                 .build()
366         );
367     }

369     private <R> Flowable<R> doWithKeys(Object partitionKey, Flowable<?> sortKeys, BiFunction<AttributeValue, Flowable<AttributeValue>, Flowable<R>> function) {
370         String hashKeyName = table.tableSchema().tableMetadata().primaryPartitionKey();
371         AttributeValue partitionKeyValue = attributeConversionHelper.convert(table, hashKeyName, partitionKey);
372 
373         Optional<String> sortKeyName = table.tableSchema().tableMetadata().primarySortKey();
374         return function.apply(partitionKeyValue, sortKeys.map(key -> attributeConversionHelper.convert(table, sortKeyName.get(), key)));
375     }
