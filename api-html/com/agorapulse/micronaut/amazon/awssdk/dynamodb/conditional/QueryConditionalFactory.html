142     public static QueryConditional attributeType(String property, EnhancedType<?> type) {
143         if (type == null) {
144             return attributeType(property, AttributeValueType.NULL);
145         }
146 
147         if (CharSequence.class.isAssignableFrom(type.rawClass())) {
148             return attributeType(property, AttributeValueType.S);
149         }
150 
151         if (Number.class.isAssignableFrom(type.rawClass())) {
152             return attributeType(property, AttributeValueType.N);
153         }
154 
155         if (Boolean.class.isAssignableFrom(type.rawClass())) {
156             return attributeType(property, AttributeValueType.BOOL);
157         }
158 
159         if (SdkBytes.class.isAssignableFrom(type.rawClass())) {
160             return attributeType(property, AttributeValueType.B);
161         }
162 
163         if (Map.class.isAssignableFrom(type.rawClass())) {
164             return attributeType(property, AttributeValueType.M);
165         }
166 
167         if (Iterable.class.isAssignableFrom(type.rawClass()) && type.rawClassParameters().size() > 0) {
168             Class<?> rawClassParameter = type.rawClassParameters().get(0).rawClass();
169             if (CharSequence.class.isAssignableFrom(rawClassParameter)) {
170                 return attributeType(property, stringValue("SS"));
171             }
172 
173             if (Number.class.isAssignableFrom(rawClassParameter)) {
174                 return attributeType(property, stringValue("NS"));
175             }
176 
177             if (SdkBytes.class.isAssignableFrom(rawClassParameter)) {
178                 return attributeType(property, stringValue("BS"));
179             }
180             return attributeType(property, AttributeValueType.L);
181         }
182 
183         throw new IllegalArgumentException("Cannot determine expected attribute type from " + type);
184     }
