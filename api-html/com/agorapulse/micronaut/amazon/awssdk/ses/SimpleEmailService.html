18 package com.agorapulse.micronaut.amazon.awssdk.ses;
19 
20 import java.util.function.Consumer;
21 
22 /**
23  * Service for sending emails via AWS Simple Email Service.
24  */
25 public interface SimpleEmailService {
26 
27     /**
28      * Builds a new email.
29      *
30      * @param composer email definition
31      * @return new email instance
32      */
33     static TransactionalEmail email(Consumer<TransactionalEmail> composer) {
34         TransactionalEmail email = new TransactionalEmail();
35         composer.accept(email);
36 
37         if (email.getRecipients().isEmpty()) {
38             throw new IllegalArgumentException("Email does not contain recepients: " + email);
39         }
40 
41         return email;
42     }
43 
44     /**
45      * Builds and sends an email.
46      * @param composer email definition
47      * @return delivery status
48      */
49     default EmailDeliveryStatus send(Consumer<TransactionalEmail> composer) {
50         return send(email(composer));
51     }
52 
53     /**
54      * Sends an email.
55      * @param email email to be sent
56      * @return delivery status
57      */
58     EmailDeliveryStatus send(TransactionalEmail email);
59 
60 }
