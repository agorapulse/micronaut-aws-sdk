018 package com.agorapulse.micronaut.amazon.awssdk.ses;
019 
020 import java.util.ArrayList;
021 import java.util.Arrays;
022 import java.util.Collections;
023 import java.util.HashMap;
024 import java.util.List;
025 import java.util.Map;
026 import java.util.function.Consumer;
027 
028 /**
029  * Email builder.
030  */
031 public class TransactionalEmail {
032 
033     private String from = "";
034     private String subject = "";
035     private String htmlBody = "<html><body></body></html>";
036     private String replyTo = "";
037     private String configurationSetName = "";
038     private Map<String, String> tags = new HashMap<>();
039 
040     private final List<String> recipients = new ArrayList<>();
041     private final List<TransactionalEmailAttachment> attachments = new ArrayList<>();
042 
043     // Constructed via SimpleEmailService
044     TransactionalEmail() { }
045 
046     public TransactionalEmail attachment(Consumer<TransactionalEmailAttachment> attachment) {
047         TransactionalEmailAttachment a = new TransactionalEmailAttachment();
048         attachment.accept(a);
049         attachments.add(a);
050         return this;
051     }
052 
053     public TransactionalEmail to(List<String> recipients) {
054         this.recipients.addAll(recipients);
055         return this;
056     }
057 
058     public TransactionalEmail to(String... recipients) {
059         this.to(Arrays.asList(recipients));
060         return this;
061     }
062 
063     public TransactionalEmail from(String str) {
064         this.from = str;
065         return this;
066     }
067 
068     public TransactionalEmail subject(String str) {
069         this.subject = str;
070         return this;
071     }
072 
073     public TransactionalEmail htmlBody(String str) {
074         this.htmlBody = str;
075         return this;
076     }
077 
078     public TransactionalEmail replyTo(String str) {
079         this.replyTo = str;
080         return this;
081     }
082 
083     public TransactionalEmail configurationSetName(String str) {
084         this.configurationSetName = str;
085         return this;
086     }
087 
088     public TransactionalEmail tags(Map<String, String> customTags) {
089         this.tags = customTags;
090         return this;
091     }
092 
093     public String getFrom() {
094         return from;
095     }
096 
097     public String getSubject() {
098         return subject;
099     }
100 
101     public String getHtmlBody() {
102         return htmlBody;
103     }
104 
105     public String getReplyTo() {
106         return replyTo;
107     }
108 
109     public List<String> getRecipients() {
110         return Collections.unmodifiableList(recipients);
111     }
112 
113     public List<TransactionalEmailAttachment> getAttachments() {
114         return Collections.unmodifiableList(attachments);
115     }
116 
117     public String getConfigurationSetName() {
118         return configurationSetName;
119     }
120 
121     public Map<String, String> getTags() {
122         return tags;
123     }
124 
125     // CHECKSTYLE:OFF
126     @Override
127     public String toString() {
128         return "TransactionalEmail{" +
129             "from='" + from + '\'' +
130             ", subject='" + subject + '\'' +
131             ", htmlBody='" + htmlBody + '\'' +
132             ", replyTo='" + replyTo + '\'' +
133             ", recipients=" + recipients +
134             ", attachments=" + attachments +
135             ", configurationSetName=" + configurationSetName +
136             ", tags=" + tags +
137             '}';
138     }
139     // CHECKSTYLE:ON
140 }
