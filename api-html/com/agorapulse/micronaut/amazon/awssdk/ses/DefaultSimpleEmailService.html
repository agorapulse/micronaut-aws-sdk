044 /**
045  * Default implementation of simple email service.
046  */
047 public class DefaultSimpleEmailService implements SimpleEmailService {
048 
049     private static final Logger LOGGER = LoggerFactory.getLogger(DefaultSimpleEmailService.class);
050 
051     private final SesClient client;
052 
053     DefaultSimpleEmailService(SesClient client) {
054         this.client = client;
055     }
056 
057     public EmailDeliveryStatus send(TransactionalEmail email) {
058         if (!email.getAttachments().isEmpty()) {
059             try {
060                 return sendEmailWithAttachment(email);
061             } catch (IOException | MessagingException exception) {
062                 LOGGER.error(
063                     String.format(
064                         "An exception was caught while sending email: destinationEmails=%s, from=%s, replyTo=%s, subject=%s",
065                         email.getRecipients(),
066                         email.getFrom(),
067                         email.getReplyTo(),
068                         email.getSubject()
069                     ),
070                     exception
071                 );
072                 return EmailDeliveryStatus.STATUS_NOT_DELIVERED;
073             }
074         }
075 
076         return sendWithoutAttachments(email);
077     }
078 
079     private static EmailDeliveryStatus handleSend(TransactionalEmail email, Runnable c) {
080         try {
081             c.run();
082             return EmailDeliveryStatus.STATUS_DELIVERED;
083         } catch (AwsServiceException exception) {
084             if (exception.getMessage().contains("blacklisted")) {
085                 LOGGER.warn(String.format("Address blacklisted destinationEmails=%s)", email.getRecipients()));
086                 return EmailDeliveryStatus.STATUS_BLACKLISTED;
087             }
088             LOGGER.error(
089                 String.format(
090                     "An amazon service exception was caught while sending email: destinationEmails=%s, from=%s, replyTo=%s, subject=%s",
091                     email.getRecipients(),
092                     email.getFrom(),
093                     email.getReplyTo(),
094                     email.getSubject()
095                 ),
096                 exception
097             );
098             return EmailDeliveryStatus.STATUS_NOT_DELIVERED;
099         }
100     }
101 
102     private EmailDeliveryStatus sendEmailWithAttachment(TransactionalEmail email) throws MessagingException, IOException {
103         Session session = Session.getInstance(new Properties());
104         MimeMessage mimeMessage = new MimeMessage(session);
105         mimeMessage.setSubject(email.getSubject());
106 
107         MimeMultipart mimeMultipart = new MimeMultipart();
108 
109         BodyPart p = new MimeBodyPart();
110         p.setContent(email.getHtmlBody(), "text/html");
111         mimeMultipart.addBodyPart(p);
112 
113         for (TransactionalEmailAttachment attachment : email.getAttachments()) {
114             if (!MimeType.isMimeTypeSupported(attachment.getMimeType())) {
115                 throw new UnsupportedAttachmentTypeException("Attachment type not supported for " + attachment.getFilename());
116             }
117 
118             MimeBodyPart mimeBodyPart = new MimeBodyPart();
119             mimeBodyPart.setFileName(attachment.getFilename());
120             mimeBodyPart.setDescription(attachment.getDescription(), StandardCharsets.UTF_8.name());
121             DataSource ds = new ByteArrayDataSource(new FileInputStream(new File(attachment.getFilepath())), attachment.getMimeType());
122             mimeBodyPart.setDataHandler(new DataHandler(ds));
123             mimeMultipart.addBodyPart(mimeBodyPart);
124         }
125 
126         mimeMessage.setContent(mimeMultipart);
127 
128         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
129         mimeMessage.writeTo(outputStream);
130 
131         SendRawEmailRequest rawEmailRequest = SendRawEmailRequest.builder()
132             .rawMessage(b -> b.data(SdkBytes.fromByteArray(outputStream.toByteArray())))
133             .destinations(email.getRecipients())
134             .source(email.getFrom())
135             .build();
136 
137         return handleSend(email, () -> client.sendRawEmail(rawEmailRequest));
138     }
139 
140     private EmailDeliveryStatus sendWithoutAttachments(TransactionalEmail email) {
141         SendEmailRequest.Builder builder = SendEmailRequest.builder()
142             .destination(b -> b.toAddresses(email.getRecipients()))
143             .message(b -> {
144                 b.subject(c -> c.data(email.getSubject()));
145                 b.body(body -> body.html(c -> c.data(email.getHtmlBody())));
146             })
147             .source(email.getFrom());
148 
149         if (email.getReplyTo() != null && email.getReplyTo().length() > 0) {
150             builder.replyToAddresses(email.getReplyTo());
151         }
152 
153         return handleSend(email, () -> client.sendEmail(builder.build()));
154     }
