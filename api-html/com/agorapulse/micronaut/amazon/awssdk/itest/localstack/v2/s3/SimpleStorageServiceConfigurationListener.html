18 package com.agorapulse.micronaut.amazon.awssdk.itest.localstack.v2.s3;
19 
20 import com.agorapulse.micronaut.amazon.awssdk.itest.localstack.LocalstackContainerHolder;
21 import com.agorapulse.micronaut.amazon.awssdk.s3.SimpleStorageServiceConfiguration;
22 import com.agorapulse.micronaut.amazon.awssdk.s3.SimpleStorageServiceFactory;
23 import io.micronaut.context.annotation.Requires;
24 import io.micronaut.context.event.BeanCreatedEvent;
25 import io.micronaut.context.event.BeanCreatedEventListener;
26 import org.testcontainers.containers.localstack.LocalStackContainer;
27 
28 import jakarta.inject.Singleton;
29 
30 @Singleton
31 @Requires(classes = SimpleStorageServiceFactory.class, beans = LocalstackContainerHolder.class)
32 public class SimpleStorageServiceConfigurationListener implements BeanCreatedEventListener<SimpleStorageServiceConfiguration> {
33 
34     private final LocalstackContainerHolder holder;
35 
36     public SimpleStorageServiceConfigurationListener(LocalstackContainerHolder holder) {
37         this.holder = holder.withServiceEnabled(LocalStackContainer.Service.S3);
38     }
39 
40     @Override
41     public SimpleStorageServiceConfiguration onCreated(BeanCreatedEvent<SimpleStorageServiceConfiguration> event) {
42         SimpleStorageServiceConfiguration conf = event.getBean();
43         if (conf.getEndpoint() != null) {
44             return conf;
45         }
46         conf.setEndpoint(holder.getEndpointOverride(LocalStackContainer.Service.S3).toString());
47         conf.setForcePathStyle(true);
48         return conf;
49     }
50 
51 }
