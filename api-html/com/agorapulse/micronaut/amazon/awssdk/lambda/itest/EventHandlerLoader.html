504         public ClassContext(Class<?> clazz, ClassContext curContext) {
505             this.typeParameters = clazz.getTypeParameters();
506             if (typeParameters.length == 0 || curContext.actualTypeArguments == null) {

508                 this.actualTypeArguments = null;
509             } else {

511                 for (int i = 0; i < types.length; i++) {
512                     types[i] = curContext.resolveTypeVariable(typeParameters[i]);

520         @SuppressWarnings({"rawtypes"})
521         public ClassContext(ParameterizedType type, ClassContext curContext) {
522             Type[] types = type.getActualTypeArguments();
523             for (int i = 0; i < types.length; i++) {
524                 Type t = types[i];
525                 if (t instanceof TypeVariable) {
526                     types[i] = curContext.resolveTypeVariable((TypeVariable) t);
527                 }
528             }
529 
530             Type t = type.getRawType();
531             if (t instanceof Class) {
532                 this.clazz = (Class) t;
533             } else if (t instanceof TypeVariable) {
534                 this.clazz = (Class) ((TypeVariable) t).getGenericDeclaration();
535             } else {
536                 throw new RuntimeException("Type " + t + " is of unexpected type " + t.getClass());
537             }
538             this.actualTypeArguments = types;
539         }

562     /**
563      * perform a breadth-first search for the first parameterized type for iface
564      *
565      * @return null of no type found. Otherwise the type found.
566      */
567     private static Type[] findInterfaceParameters(Class<?> clazz, Class<?> iface) {
568         LinkedList<ClassContext> clazzes = new LinkedList<>();
569         clazzes.addFirst(new ClassContext(clazz, (Type[]) null));
570         while (!clazzes.isEmpty()) {
571             final ClassContext curContext = clazzes.removeLast();
572             Type[] interfaces = curContext.clazz.getGenericInterfaces();
573 
574             for (Type type : interfaces) {
575                 if (type instanceof ParameterizedType) {
576                     ParameterizedType candidate = (ParameterizedType) type;
577                     Type rawType = candidate.getRawType();
578                     if (!(rawType instanceof Class)) {
579                         //should be impossible
580                         System.err.println("raw type is not a class: " + rawType);
581                         continue;
582                     }
583                     Class<?> rawClass = (Class<?>) rawType;
584                     if (iface.isAssignableFrom(rawClass)) {
585                         return new ClassContext(candidate, curContext).actualTypeArguments;
586                     } else {
587                         clazzes.addFirst(new ClassContext(candidate, curContext));
588                     }
589                 } else if (type instanceof Class) {
590                     clazzes.addFirst(new ClassContext((Class<?>) type, curContext));
591                 } else {
592                     //should never happen?
593                     System.err.println("Unexpected type class " + type.getClass().getName());
594                 }
595             }
596 
597             final Type superClass = curContext.clazz.getGenericSuperclass();
598             if (superClass instanceof ParameterizedType) {
599                 clazzes.addFirst(new ClassContext((ParameterizedType) superClass, curContext));
600             } else if (superClass != null) {
601                 clazzes.addFirst(new ClassContext((Class<?>) superClass, curContext));
602             }
603         }
604         return null;
605     }

793     /**
794      * Implement a comparator for Methods. We sort overloaded handler methods using this comparator, and then pick the
795      * lowest sorted method.
796      */
797     private static final Comparator<Method> methodPriority = new Comparator<Method>() {
798         public int compare(Method lhs, Method rhs) {
799 
800             //1. Non bridge methods are preferred over bridge methods.
801             if (!lhs.isBridge() && rhs.isBridge()) {
802                 return -1;
803             } else if (!rhs.isBridge() && lhs.isBridge()) {
804                 return 1;
805             }
806 
807             //2. We prefer longer signatures to shorter signatures. Except we count a method whose last argument is
808             //Context as having 1 more argument than it really does. This is a stupid thing to do, but we
809             //need to keep it for back compat reasons.
810             Class<?>[] lParams = lhs.getParameterTypes();
811             Class<?>[] rParams = rhs.getParameterTypes();
812 
813             int lParamCompareLength = lParams.length;
814             int rParamCompareLength = rParams.length;
815 
816             if (lastParameterIsContext(lParams)) {
817                 ++lParamCompareLength;
818             }
819 
820             if (lastParameterIsContext(rParams)) {
821                 ++rParamCompareLength;
822             }
823 
824             return -Integer.compare(lParamCompareLength, rParamCompareLength);
825         }
826     };

828     private static LambdaRequestHandler loadEventPojoHandler(HandlerInfo handlerInfo, ApplicationContext context) {
829         Method[] methods;
830         try {
831             methods = handlerInfo.clazz.getMethods();
832         } catch (NoClassDefFoundError e) {
833             return new LambdaRequestHandler.UserFaultHandler(new UserFault(
834                     "Error loading method " + handlerInfo.methodName + " on class " + handlerInfo.clazz.getName(),
835                     e.getClass().getName(),
836                     trace(e)
837             ));
838         }
839         if (methods.length == 0) {
840             final String msg = "Class "
841                     + handlerInfo.getClass().getName()
842                     + " has no public method named "
843                     + handlerInfo.methodName;
844             return new UserFaultHandler(makeUserFault(msg));
845         }
846 
847         /*
848          * We support the following signatures
849          * Anything (InputStream, OutputStream, Context)
850          * Anything (InputStream, OutputStream)
851          * Anything (OutputStream, Context)
852          * Anything (InputStream, Context)
853          * Anything (InputStream)
854          * Anything (OutputStream)
855          * Anything (Context)
856          * Anything (AlmostAnything, Context)
857          * Anything (AlmostAnything)
858          * Anything ()
859          *
860          * where AlmostAnything is any type except InputStream, OutputStream, Context
861          * Anything represents any type (primitive, void, or Object)
862          *
863          * prefer methods with longer signatures, add extra weight to those ending with a Context object
864          *
865          */
866 
867         int slide = 0;
868 
869         for (int i = 0; i < methods.length; i++) {
870             Method m = methods[i];
871             methods[i - slide] = m;
872             if (!m.getName().equals(handlerInfo.methodName)) {
873                 slide++;
874                 continue;
875             }
876         }
877 
878         final int end = methods.length - slide;
879         Arrays.sort(methods, 0, end, methodPriority);
880 
881         for (int i = 0; i < end; i++) {
882             Method m = methods[i];
883             Optional<LambdaRequestHandler> result = getHandlerFromOverload(handlerInfo.clazz, m, context);
884             if (result.isPresent()) {
885                 return result.get();
886             } else {
887                 continue;
888             }
889         }
890 
891         return new UserFaultHandler(makeUserFault("No public method named "
892                 + handlerInfo.methodName
893                 + " with appropriate method signature found on class "
894                 + handlerInfo.clazz.getName()));
895     }
