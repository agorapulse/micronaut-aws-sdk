287     /**
288      * Finds the concrete return type of a specified method, including handling for generic superclasses.
289      *
290      * @param clazz      The class or subclass to inspect.
291      * @param methodName The name of the method whose return type you're interested in.
292      * @return The concrete Type of the method's return type, considering superclass generic parameters if applicable.
293      */
294     private static Type findMethodConcreteReturnType(Class<?> clazz, String methodName) {
295         Type returnType = null;
296 
297         // First, attempt to find the method directly in the class or its superclasses.
298         Method method = findMethodInHierarchy(clazz, methodName);
299         if (method != null) {
300             returnType = method.getGenericReturnType();
301         }
302 
303         // If the returnType is not a ParameterizedType, attempt to match it against generic superclass parameters.
304         if (!(returnType instanceof ParameterizedType)) {
305             Type genericSuperclass = clazz.getGenericSuperclass();
306             if (genericSuperclass instanceof ParameterizedType) {
307                 ParameterizedType parameterizedType = (ParameterizedType) genericSuperclass;
308                 Type[] typeArguments = parameterizedType.getActualTypeArguments();
309                 for (Type typeArgument : typeArguments) {
310                     if (Objects.requireNonNull(returnType).getTypeName().equals(typeArgument.getTypeName())) {
311                         return typeArgument; // Match found in generic superclass parameters.
312                     }
313                 }
314             }
315         }
316 
317         if (returnType == Object.class) {
318             for (Type genericInterface : clazz.getGenericInterfaces()) {
319                 method = findMethodInHierarchy((Class<?>) ((ParameterizedType) genericInterface).getRawType(), methodName);
320                 if (method != null) {
321                     returnType = method.getGenericReturnType();
322                     break;
323                 }
324             }
325         }
326 
327         if (!(returnType instanceof ParameterizedType)) {
328             for (Type type : clazz.getGenericInterfaces()) {
329                 if (type instanceof ParameterizedType) {
330                     ParameterizedType parameterizedType = (ParameterizedType) type;
331                     Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
332                     Type[] typeArguments = ((Class<?>)parameterizedType.getRawType()).getTypeParameters();
333                     for (int i = 0; i < typeArguments.length; i++) {
334                         Type typeArgument = typeArguments[i];
335                         if (Objects.requireNonNull(returnType).getTypeName().equals(typeArgument.getTypeName())) {
336                             return actualTypeArguments[i]; // Match found in generic interface parameters.
337                         }
338                     }
339                 }
340             }
341         }
342 
343         return returnType; // Return the found returnType or null if not found.
344     }

346     private static Method findMethodInHierarchy(Class<?> clazz, String methodName) {
347         while (clazz != null) {
348             for (Method method : clazz.getDeclaredMethods()) {
349                 if (method.getName().equals(methodName)) {
350                     return method; // Method found in current class or up the hierarchy.
351                 }
352             }
353             clazz = clazz.getSuperclass(); // Move up in the class hierarchy.
354         }
355         return null; // Method not found in class hierarchy.
356     }

358     // backport from newer Java
359     private static byte[] readNBytes(InputStream is) throws IOException {
360         if (Integer.MAX_VALUE < 0) {
361             throw new IllegalArgumentException("len < 0");
362         }
363 
364         List<byte[]> bufs = null;
365         byte[] result = null;
366         int total = 0;
367         int remaining = Integer.MAX_VALUE;
368         int n;
369         do {
370             byte[] buf = new byte[Math.min(remaining, 8192)];
371             int nread = 0;
372 
373             // read to EOF which may read more or less than buffer size
374             while ((n = is.read(buf, nread,
375                 Math.min(buf.length - nread, remaining))) > 0) {
376                 nread += n;
377                 remaining -= n;
378             }
379 
380             if (nread > 0) {
381                 if (Integer.MAX_VALUE - 8 - total < nread) {
382                     throw new OutOfMemoryError("Required array size too large");
383                 }
384                 if (nread < buf.length) {
385                     buf = Arrays.copyOfRange(buf, 0, nread);
386                 }
387                 total += nread;
388                 if (result == null) {
389                     result = buf;
390                 } else {
391                     if (bufs == null) {
392                         bufs = new ArrayList<>();
393                         bufs.add(result);
394                     }
395                     bufs.add(buf);
396                 }
397             }
398             // if the last call to read returned -1 or the number of bytes
399             // requested have been read then break
400         } while (n >= 0 && remaining > 0);
401 
402         if (bufs == null) {
403             if (result == null) {
404                 return new byte[0];
405             }
406             return result.length == total ? result : Arrays.copyOf(result, total);
407         }
408 
409         result = new byte[total];
410         int offset = 0;
411         remaining = total;
412         for (byte[] b : bufs) {
413             int count = Math.min(b.length, remaining);
414             System.arraycopy(b, 0, result, offset, count);
415             offset += count;
416             remaining -= count;
417         }
418 
419         return result;
420     }
