18 package com.agorapulse.micronaut.grails;
19 
20 import io.micronaut.context.env.DefaultEnvironment;
21 import io.micronaut.core.convert.ArgumentConversionContext;
22 import org.springframework.core.env.Environment;
23 
24 import javax.annotation.Nullable;
25 import java.util.Optional;
26 import java.util.Set;
27 
28 class GrailsPropertyTranslatingEnvironment extends DefaultEnvironment {
29 
30     private final Environment environment;
31     private final PropertyTranslatingCustomizer customizer;
32 
33     GrailsPropertyTranslatingEnvironment(Environment environment, PropertyTranslatingCustomizer customizer) {
34         super(environment.getActiveProfiles());
35         this.environment = environment;
36         this.customizer = customizer;
37     }
38 
39     @Override
40     public io.micronaut.context.env.Environment start() {
41         return this;
42     }
43 
44     @Override
45     public io.micronaut.context.env.Environment stop() {
46         return this;
47     }
48 
49     @Override
50     public boolean containsProperty(@Nullable String name) {
51         if (environment.containsProperty(name)) {
52             return true;
53         }
54 
55         Set<String> alternativeNames = customizer.getAlternativeNames(name);
56         if (alternativeNames.isEmpty()) {
57             return false;
58         }
59 
60         return alternativeNames.stream().anyMatch(environment::containsProperty);
61     }
62 
63     @Override
64     public boolean containsProperties(@Nullable String name) {
65         return containsProperty(name);
66     }
67 
68     @Override
69     public <T> Optional<T> getProperty(@Nullable String name, ArgumentConversionContext<T> conversionContext) {
70         Class<T> type = conversionContext.getArgument().getType();
71         T value = environment.getProperty(name, type);
72         if (value != null) {
73             return Optional.of(value);
74         }
75 
76         Set<String> alternativeNames = customizer.getAlternativeNames(name);
77         if (alternativeNames.isEmpty()) {
78             return Optional.empty();
79         }
80 
81         for (String alternativeName : alternativeNames) {
82             T alternativeValue = environment.getProperty(alternativeName, type);
83             if (alternativeValue != null) {
84                 return Optional.of(alternativeValue);
85             }
86         }
87 
88         return Optional.empty();
89     }
