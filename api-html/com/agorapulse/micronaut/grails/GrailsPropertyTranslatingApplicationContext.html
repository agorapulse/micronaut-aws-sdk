18 package com.agorapulse.micronaut.grails;
19 
20 import io.micronaut.context.DefaultApplicationContext;
21 import io.micronaut.context.env.Environment;
22 
23 class GrailsPropertyTranslatingApplicationContext extends DefaultApplicationContext {
24 
25     private final Environment environment;
26 
27     GrailsPropertyTranslatingApplicationContext(org.springframework.core.env.Environment environment, PropertyTranslatingCustomizer customizer) {
28         super(environment.getActiveProfiles());
29         this.environment = new GrailsPropertyTranslatingEnvironment(environment, customizer);
30         this.environment.addPropertySource(new EnvVarLikeSystemPropertiesPropertySource());
31     }
32 
33     @Override
34     public io.micronaut.context.env.Environment getEnvironment() {
35         return environment;
36     }
37 
38 }
