040 /**
041  * Adds Micronaut beans to a Grails' Spring application context.  This processor will
042  * find all of the Micronaut beans of the specified types
043  * and add them as beans to the Spring application context.
044  * <p>
045  * The Grails properties can be translated to Micronaut properties if required.
046  *
047  * @since 1.0
048  */
049 public class GrailsMicronautBeanProcessor implements BeanFactoryPostProcessor, DisposableBean, EnvironmentAware {
050 
051     static final Logger LOGGER = LoggerFactory.getLogger(GrailsMicronautBeanProcessor.class);
052 
053     /**
054      * @deprecated please declare {@link MicronautBeanImporter} bean to avoid multiple Micronaut application context
055      * inside single application.
056      */
057     public static class Builder extends MicronautBeanImporter {
058 
059         Builder(PropertyTranslatingCustomizer customizer) {
060             customize(customizer);
061         }
062 
063     }
064 
065     /**
066      * Starts creation of bean processor using default {@link PropertyTranslatingCustomizer#grails()} customizer.
067      * @return bean processor builder using {@link PropertyTranslatingCustomizer#grails()}
068      * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
069      * inside single application.
070      */
071     public static GrailsMicronautBeanProcessor.Builder builder() {
072         return builder(grails());
073     }
074 
075     /**
076      * Starts creation of bean processor using given customizer.
077      * @param customizerBuilder customizer being used
078      * @return bean processor builder using given cusotomizer
079      * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
080      * inside single application.
081      */
082     public static GrailsMicronautBeanProcessor.Builder builder(PropertyTranslatingCustomizer.Builder customizerBuilder) {
083         return builder(customizerBuilder.build());
084     }
085 
086     /**
087      * Starts creation of bean processor using given customizer.
088      * @param customizer customizer being used
089      * @return bean processor builder using given cusotomizer
090      * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
091      * inside single application.
092      */
093     public static GrailsMicronautBeanProcessor.Builder builder(PropertyTranslatingCustomizer customizer) {
094         return new Builder(customizer);
095     }
096 
097     private static final String MICRONAUT_BEAN_TYPE_PROPERTY_NAME = "micronautBeanType";
098     private static final String MICRONAUT_CONTEXT_PROPERTY_NAME = "micronautContext";
099     private static final String MICRONAUT_QUALIFIER_PROPERTY_NAME = "micronautQualifier";
100     private static final String MICRONAUT_SINGLETON_PROPERTY_NAME = "micronautSingleton";
101 
102     private DefaultBeanContext micronautContext;
103     private final Map<String, Qualifier<?>> micronautBeanQualifiers;
104     private final List<PropertyTranslatingCustomizer> customizers;
105     private Environment environment;
106 
107     /**
108      * @param customizers properties translation customizer
109      * @param qualifiers the names and qualifiers of the Micronaut beans which should be added to the
110      *                   Spring application context.
111      */
112     GrailsMicronautBeanProcessor(Map<String, Qualifier<?>> qualifiers, List<PropertyTranslatingCustomizer> customizers) {
113         this.customizers = customizers;
114         this.micronautBeanQualifiers = qualifiers;
115     }
116 
117     @Override
118     public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
119         if (environment == null) {
120             throw new IllegalStateException("Spring environment not set!");
121         }
122 
123         micronautContext = new GrailsPropertyTranslatingApplicationContext(environment, of(collapse(customizers)));
124 
125         micronautContext.start();
126 
127         NoClassDefFoundError noClassDefFoundError = null;
128 
129         for (Map.Entry<String, Qualifier<?>> entry : micronautBeanQualifiers.entrySet()) {
130             String name = entry.getKey();
131             Qualifier<?> micronautBeanQualifier = entry.getValue();
132             try {
133                 Collection<BeanDefinition<?>> beanDefinitions = micronautContext.getBeanDefinitions((Qualifier<Object>) micronautBeanQualifier);
134 
135                 if (beanDefinitions.size() > 1) {
136                     throw new IllegalArgumentException("There is too many candidates for " + micronautBeanQualifier + "! Candidates: " + beanDefinitions);
137                 }
138 
139                 Optional<BeanDefinition<?>> firstBean = beanDefinitions.stream().findFirst();
140                 BeanDefinition<?> definition = firstBean.orElseThrow(()-> new IllegalArgumentException("There is no candidate for " + micronautBeanQualifier));
141 
142                 final BeanDefinitionBuilder beanDefinitionBuilder = BeanDefinitionBuilder
143                     .rootBeanDefinition(GrailsMicronautBeanFactory.class);
144                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_BEAN_TYPE_PROPERTY_NAME, definition.getBeanType());
145                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_QUALIFIER_PROPERTY_NAME, micronautBeanQualifier);
146                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_CONTEXT_PROPERTY_NAME, micronautContext);
147                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_SINGLETON_PROPERTY_NAME, definition.isSingleton());
148 
149                 ((DefaultListableBeanFactory) beanFactory).registerBeanDefinition(name, beanDefinitionBuilder.getBeanDefinition());
150             } catch (NoClassDefFoundError error) {
151                 LOGGER.error("Exception loading class for qualifier {}. Bean {} will not be available in the runtime", micronautBeanQualifier, name);
152                 LOGGER.error("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
153                 LOGGER.error("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
154                 LOGGER.error("Current class path: {}", System.getProperty("java.class.path"));
155                 noClassDefFoundError = error;
156             }
157         }
158 
159         if (noClassDefFoundError == null) {
160             if (LOGGER.isInfoEnabled()) {
161                 LOGGER.info("Successfully added following beans to the spring contest {} ", micronautBeanQualifiers);
162                 LOGGER.info("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
163                 LOGGER.info("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
164                 LOGGER.info("Current class path: {}", System.getProperty("java.class.path"));
165             }
166             return;
167         }
168 
169         throw noClassDefFoundError;
170     }
171 
172     private static String printClassLoader(ClassLoader classLoader) {
173         if (classLoader instanceof URLClassLoader) {
174             return "URLClassLoader for URLS:" + Arrays.toString(((URLClassLoader) classLoader).getURLs());
175         }
176         if (classLoader == null) {
177             return null;
178         }
179         return classLoader.toString();
180     }

182     @Override
183     public void destroy() {
184         if (micronautContext != null) {
185             micronautContext.close();
186         }
187     }

191         this.environment = environment;
