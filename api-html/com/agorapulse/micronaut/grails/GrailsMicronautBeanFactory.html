18 package com.agorapulse.micronaut.grails;
19 
20 import io.micronaut.context.DefaultApplicationContext;
21 import io.micronaut.context.Qualifier;
22 import io.micronaut.context.exceptions.BeanInstantiationException;
23 import org.springframework.beans.factory.FactoryBean;
24 
25 import java.util.Optional;
26 
27 /**
28  * A spring FactoryBean for adding Micronaut beans to a
29  * Spring application context.
30  *
31  * @since 1.0
32  */
33 class GrailsMicronautBeanFactory implements FactoryBean {
34 
35     private Class micronautBeanType;
36     private DefaultApplicationContext micronautContext;
37     private Qualifier<Object> micronautQualifier;
38     private boolean isMicronautSingleton;
39 
40     /**
41      * @param micronautBeanType The type of bean this factory will create
42      */
43     public void setMicronautBeanType(Class micronautBeanType) {
44         this.micronautBeanType = micronautBeanType;
45     }
46 
47     /**
48      * @param micronautContext The Micronaut application context
49      */
50     public void setMicronautContext(DefaultApplicationContext micronautContext) {
51         this.micronautContext = micronautContext;
52     }
53 
54     /**
55      *
56      * @param isMicronautSingleton indicates if the Micronaut bean is a singleton
57      */
58     public void setMicronautSingleton(boolean isMicronautSingleton) {
59         this.isMicronautSingleton = isMicronautSingleton;
60     }
61 
62     /**
63      * @param micronautQualifier micronaut qualifier of the bean
64      */
65     public void setMicronautQualifier(Qualifier<Object> micronautQualifier) {
66         this.micronautQualifier = micronautQualifier;
67     }
68 
69     @Override
70     public Object getObject() throws Exception {
71         Optional bean = micronautContext.findBean(micronautBeanType, micronautQualifier);
72         if (bean.isPresent()) {
73             return bean.get();
74         }
75 
76         throw new BeanInstantiationException("Could Not Create Bean [" + micronautBeanType + "]");
77     }
78 
79     @Override
80     public Class<?> getObjectType() {
81         return micronautBeanType;
82     }
83 
84     @Override
85     public boolean isSingleton() {
86         return isMicronautSingleton;
87     }
88 }
