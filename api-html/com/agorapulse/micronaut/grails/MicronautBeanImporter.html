018 package com.agorapulse.micronaut.grails;
019 
020 import io.micronaut.context.Qualifier;
021 import io.micronaut.core.naming.NameUtils;
022 import io.micronaut.inject.qualifiers.Qualifiers;
023 
024 import java.lang.annotation.Annotation;
025 import java.util.*;
026 
027 /**
028  * Defineds which Micronaut beans should be added into a Grails' Spring application context. The processor will then
029  * find all of the Micronaut beans of the specified types and add them as beans to the Spring application context.
030  * <p>
031  * <p>
032  * It can also manage beans to
033  *
034  * @since 1.2.3.1
035  */
036 public class MicronautBeanImporter {
037 
038     public static MicronautBeanImporter create() {
039         return new MicronautBeanImporter();
040     }
041 
042     private final Map<String, Qualifier<?>> micronautBeanQualifiers = new LinkedHashMap<>();
043     private final List<PropertyTranslatingCustomizer> customizers = new ArrayList<>();
044 
045     protected MicronautBeanImporter() {}
046 
047     public MicronautBeanImporter addByType(Class<?> type) {
048         return addByQualifiers(NameUtils.decapitalize(type.getSimpleName()), Qualifiers.byType(type));
049     }
050 
051     public MicronautBeanImporter addByType(String grailsBeanName, Class<?>... types) {
052         micronautBeanQualifiers.put(grailsBeanName, Qualifiers.byType(types));
053         return this;
054     }
055 
056     public MicronautBeanImporter addByStereotype(String grailsBeanName, Class<? extends Annotation> type) {
057         return addByQualifiers(grailsBeanName, Qualifiers.byStereotype(type));
058     }
059 
060     public MicronautBeanImporter addByName(String name) {
061         return addByName(name, name);
062     }
063 
064     public MicronautBeanImporter addByName(String grailsBeanName, String micronautName) {
065         micronautBeanQualifiers.put(grailsBeanName, Qualifiers.byName(micronautName));
066         return this;
067     }
068 
069     @SafeVarargs
070     public final <T> MicronautBeanImporter addByQualifiers(String grailsBeanName, Qualifier<T>... qualifiers) {
071         micronautBeanQualifiers.put(grailsBeanName, Qualifiers.byQualifiers(qualifiers));
072         return this;
073     }
074 
075     public MicronautBeanImporter customize(PropertyTranslatingCustomizer customizer) {
076         this.customizers.add(customizer);
077         return this;
078     }
079 
080     public MicronautBeanImporter customize(PropertyTranslatingCustomizer.Builder customizer) {
081         return customize(customizer.build());
082     }
083 
084     public Map<String, Qualifier<?>> getMicronautBeanQualifiers() {
085         return Collections.unmodifiableMap(micronautBeanQualifiers);
086     }
087 
088     public List<PropertyTranslatingCustomizer> getCustomizers() {
089         return Collections.unmodifiableList(customizers);
090     }
091 
092     /**
093      * @return new GrailsMicronautBeanProcessor for current builder
094      * @deprecated please declare {@link MicronautBeanImporter} bean to avoid multiple Micronaut application context
095      * inside single application.
096      */
097     public GrailsMicronautBeanProcessor build() {
098         try {
099             throw new IllegalStateException();
100         } catch (IllegalStateException th) {
101             GrailsMicronautBeanProcessor.LOGGER.error("Old style of importing Micronaut beans used. This will lead to having multiple Micronaut application context in the application");
102         }
103         return new GrailsMicronautBeanProcessor(getMicronautBeanQualifiers(), getCustomizers());
104     }
105 }
