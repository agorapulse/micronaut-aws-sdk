18 package com.agorapulse.micronaut.grails;
19 
20 import java.util.Collection;
21 import java.util.Collections;
22 import java.util.Set;
23 import java.util.stream.Collectors;
24 
25 /**
26  * Property translating customizer add an ability to re-use existing properties defined in Spring application context
27  * in Micronaut context using simple translating rules.
28  */
29 public interface PropertyTranslatingCustomizer {
30 
31     interface Builder {
32         /**
33          * Replace property name prefixes.
34          * @param original original prefix, can be empty
35          * @param replacement new prefix, can be empty
36          * @return self
37          */
38         Builder replacePrefix(String original, String replacement);
39 
40         /**
41          * Ignores exact match of the property name.
42          * @param property the property to be ignored
43          * @return self
44          */
45         Builder ignore(String property);
46 
47         /**
48          * Ignores by regex match of the property pattern.
49          * @param propertyPattern the pattern of properties being ignored
50          * @return self
51          */
52         Builder ignoreAll(String propertyPattern);
53 
54         /**
55          * @return the customizer instance
56          */
57         PropertyTranslatingCustomizer build();
58     }
59 
60     static PropertyTranslatingCustomizer none() {
61         return name -> Collections.emptySet();
62     }
63 
64     /**
65      * Create new customizer builder.
66      * @return self
67      */
68     static PropertyTranslatingCustomizer.Builder builder() {
69         return GrailsPropertyTranslatingCustomizer.create();
70     }
71 
72     /**
73      * @return builder for grails customizer which replaces Micronaut prefix with Grails and also allows to skip the Grails prefix
74      */
75     static PropertyTranslatingCustomizer.Builder grails() {
76         return GrailsPropertyTranslatingCustomizer.standard();
77     }
78 
79     static PropertyTranslatingCustomizer of(final Collection<PropertyTranslatingCustomizer> customizers) {
80         return name -> customizers.stream().flatMap(c -> c.getAlternativeNames(name).stream()).collect(Collectors.toSet());
81     }
82 
83     /**
84      * @param name original property name
85      * @return set of alternative names for the original property name
86      */
87     Set<String> getAlternativeNames(String name);
88 
89 }
