020 import io.micronaut.core.naming.NameUtils;
021 
022 import java.util.*;
023 import java.util.regex.Pattern;
024 import java.util.stream.Collectors;
025 
026 class GrailsPropertyTranslatingCustomizer implements PropertyTranslatingCustomizer, PropertyTranslatingCustomizer.Builder {
027 
028     private static class PrefixReplacement {
029         final String original;
030         final String replacement;
031 
032         PrefixReplacement(String original, String replacement) {
033             this.original = original;
034             this.replacement = replacement;
035         }
036 
037         @Override
038         public boolean equals(Object o) {
039             if (this == o) {
040                 return true;
041             }
042 
043             if (o == null || getClass() != o.getClass()) {
044                 return false;
045             }
046 
047             PrefixReplacement that = (PrefixReplacement) o;
048             return Objects.equals(original, that.original) && Objects.equals(replacement, that.replacement);
049         }
050 
051         @Override
052         public int hashCode() {
053             return Objects.hash(original, replacement);
054         }
055 
056         @Override
057         public String toString() {
058             return original + " -> " + replacement;
059         }
060     }
061 
062     private static final String GRAILS_PREFIX = "grails.";
063     private static final String MICRONAUT_PREFIX = "micronaut.";
064     private static final String EMPTY_STRING = "";
065 
066     static GrailsPropertyTranslatingCustomizer create() {
067         return new GrailsPropertyTranslatingCustomizer();
068     }
069 
070     static GrailsPropertyTranslatingCustomizer standard() {
071         return new GrailsPropertyTranslatingCustomizer()
072             .replacePrefix(MICRONAUT_PREFIX, GRAILS_PREFIX)
073             .replacePrefix(EMPTY_STRING, GRAILS_PREFIX);
074     }
075 
076     static List<PropertyTranslatingCustomizer> collapse(List<PropertyTranslatingCustomizer> customizers) {
077         List<PropertyTranslatingCustomizer> result = new ArrayList<>();
078         GrailsPropertyTranslatingCustomizer customizer = create();
079         result.add(customizer);
080 
081         for (PropertyTranslatingCustomizer c : customizers) {
082             if (c instanceof GrailsPropertyTranslatingCustomizer) {
083                 GrailsPropertyTranslatingCustomizer gc = (GrailsPropertyTranslatingCustomizer) c;
084                 customizer.prefixPrefixReplacements.addAll(gc.prefixPrefixReplacements);
085                 customizer.ignored.addAll(gc.ignored);
086             } else {
087                 result.add(c);
088             }
089         }
090 
091         return result;
092     }

094     private final Set<PrefixReplacement> prefixPrefixReplacements = new LinkedHashSet<>();
095     private final Set<Pattern> ignored = new LinkedHashSet<>();
096 
097     private GrailsPropertyTranslatingCustomizer() {
098         // prevents instantiation
099     }
100 
101     /**
102      * Replace property name prefixes.
103      *
104      * @param original    original prefix, can be empty
105      * @param replacement new prefix, can be empty
106      * @return self
107      */
108     @Override
109     public GrailsPropertyTranslatingCustomizer replacePrefix(String original, String replacement) {
110         prefixPrefixReplacements.add(new PrefixReplacement(original, replacement));
111         return this;
112     }
113 
114     /**
115      * Ignores exact match of the property name.
116      *
117      * @param property the property to be ignored
118      * @return self
119      */
120     @Override
121     public GrailsPropertyTranslatingCustomizer ignore(String property) {
122         ignored.add(Pattern.compile(Pattern.quote(property)));
123         return this;
124     }
125 
126     /**
127      * Ignores by regex match of the property pattern.
128      *
129      * @param propertyPattern the pattern of properties being ignored
130      * @return self
131      */
132     @Override
133     public GrailsPropertyTranslatingCustomizer ignoreAll(String propertyPattern) {
134         ignored.add(Pattern.compile(propertyPattern));
135         return this;
136     }
137 
138     @Override
139     public PropertyTranslatingCustomizer build() {
140         return this;
141     }
142 
143     @Override
144     public Set<String> getAlternativeNames(String name) {
145         if (name == null || name.length() == 0) {
146             return Collections.emptySet();
147         }
148 
149         if (ignored.stream().anyMatch(p -> p.matcher(name).matches())) {
150             return Collections.emptySet();
151         }
152 
153         Set<String> keys = new LinkedHashSet<>(prefixPrefixReplacements.size());
154         prefixPrefixReplacements.forEach(replacement -> {
155             if ((replacement.original.isEmpty() || name.startsWith(replacement.original)) && (replacement.replacement.isEmpty() || !name.startsWith(replacement.replacement))) {
156                 String keyNoPrefix = name.substring(replacement.original.length());
157                 String alternativeKey = replacement.replacement + keyNoPrefix;
158                 if (!alternativeKey.equals(name)) {
159                     keys.add(alternativeKey);
160                 }
161             }
162         });
163 
164         keys.addAll(keys.stream().map(GrailsPropertyTranslatingCustomizer::dehyphenate).collect(Collectors.toSet()));
165 
166         return keys;
167     }
168 
169     private static String dehyphenate(String key) {
170         return Arrays.stream(key.split("\\."))
171             .map(NameUtils::dehyphenate)
172             .map(name -> {
173                 if (name.length() <= 1) {
174                     return name.toLowerCase();
175                 }
176                 return name.substring(0, 1).toLowerCase() + name.substring(1);
177             })
178             .collect(Collectors.joining("."));
179     }
