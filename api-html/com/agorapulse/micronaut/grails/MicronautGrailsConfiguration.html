18 package com.agorapulse.micronaut.grails;
19 
20 import io.micronaut.context.Qualifier;
21 import org.springframework.context.annotation.Bean;
22 import org.springframework.context.annotation.Configuration;
23 
24 import java.util.List;
25 import java.util.Map;
26 import java.util.stream.Collectors;
27 
28 @Configuration
29 public class MicronautGrailsConfiguration {
30 
31     /**
32      * Guarantees at least one importer on the classpath.
33      * @return empty importer
34      */
35     @Bean
36     MicronautBeanImporter grailsTranslatorImporter() {
37         return MicronautBeanImporter.create().customize(PropertyTranslatingCustomizer.grails());
38     }
39 
40     @Bean
41     GrailsMicronautBeanProcessor defaultGrailsMicronautBeanProcessor(List<MicronautBeanImporter> importers) {
42         Map<String, Qualifier<?>> qualifierMap = importers
43             .stream()
44             .flatMap(i -> i.getMicronautBeanQualifiers().entrySet().stream())
45             .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
46 
47         List<PropertyTranslatingCustomizer> customizers = importers
48             .stream()
49             .flatMap(i -> i.getCustomizers().stream())
50             .collect(Collectors.toList());
51 
52         return new GrailsMicronautBeanProcessor(
53             qualifierMap,
54             customizers
55         );
56     }
57 
58 }
