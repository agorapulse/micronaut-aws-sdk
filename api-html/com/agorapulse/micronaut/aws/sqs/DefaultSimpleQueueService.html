030 /**
031  * Default simple queue service implementation.
032  */
033 @Slf4j
034 @CompileStatic
035 @SuppressWarnings('NoWildcardImports')
036 class DefaultSimpleQueueService implements SimpleQueueService {
037 
038     public static final String QUEUE_ARN = 'QueueArn'
039     private final AmazonSQS client
040     private final SimpleQueueServiceConfiguration configuration
041 
042     private final Map<String, String> queueUrlByNames = [:]
043 
044     DefaultSimpleQueueService(
045         AmazonSQS client,
046         SimpleQueueServiceConfiguration configuration
047     ) {
048         this.client = client
049         this.configuration = configuration
050     }

052     @Override
053     String getDefaultQueueName() {
054         assertDefaultQueueName()
055         return configuration.queue
056     }
057 
058     @Override
059     boolean isCaching() {
060         return configuration.cache
061     }

063     /**
064      *
065      * @param queueName
066      * @return queue URL
067      */
068     String createQueue(String queueName) {
069         QueueConfiguration configuration = configuration.copy()
070         configuration.queue = queueName
071 
072         return createQueue(configuration)
073     }
074 
075     /**
076      *
077      * @param queueName
078      * @return queue URL
079      */
080     String createQueue(QueueConfiguration configuration) {
081         CreateQueueRequest createQueueRequest = new CreateQueueRequest(configuration.queue)
082         if (configuration.delaySeconds) {
083             createQueueRequest.attributes['DelaySeconds'] = configuration.delaySeconds.toString()
084         }
085         if (configuration.messageRetentionPeriod) {
086             createQueueRequest.attributes['MessageRetentionPeriod'] = configuration.messageRetentionPeriod.toString()
087         }
088         if (configuration.maximumMessageSize) {
089             createQueueRequest.attributes['MaximumMessageSize'] = configuration.maximumMessageSize.toString()
090         }
091         if (configuration.visibilityTimeout) {
092             createQueueRequest.attributes['VisibilityTimeout'] = configuration.visibilityTimeout.toString()
093         }
094         if (configuration.fifo || configuration.queue.endsWith('.fifo')) {
095             createQueueRequest.attributes['FifoQueue'] = 'true'
096         }
097         String queueUrl = client.createQueue(createQueueRequest).queueUrl
098         log.debug "Queue created (queueUrl=$queueUrl)"
099         addQueue(queueUrl)
100         return queueUrl
101     }

103     /**
104      *
105      * @param queueUrl
106      * @param receiptHandle
107      */
108     void deleteMessage(String queueName,
109                        String receiptHandle) {
110         String queueUrl = getQueueUrl(queueName)
111         client.deleteMessage(new DeleteMessageRequest(queueUrl, receiptHandle))
112         log.debug "Message deleted (queueUrl=$queueUrl)"
113     }

116      *

118      */
119     void deleteQueue(String queueName) {

121         client.deleteQueue(queueUrl)
122         removeQueue(queueUrl)
123     }

125     /**
126      *
127      * @param queueName
128      * @return
129      */
130     Map getQueueAttributes(String queueName) {
131         String queueUrl = getQueueUrl(queueName)
132 
133         GetQueueAttributesRequest getQueueAttributesRequest = new GetQueueAttributesRequest(queueUrl).withAttributeNames(['All'])
134         Map attributes = [:]
135         try {
136             attributes = client.getQueueAttributes(getQueueAttributesRequest).attributes
137         } catch (AmazonServiceException exception) {
138             if (exception.errorCode == 'AWS.SimpleQueueService.NonExistentQueue') {
139                 removeQueue(queueUrl)
140             }
141             log.warn 'An amazon service exception was catched while getting queue attributes', exception
142         } catch (AmazonClientException exception) {
143             log.warn 'An amazon client exception was catched while getting queue attributes', exception
144         }
145         return attributes
146     }
147 
148     /**
149      *
150      * @param jobName
151      * @param autoCreate
152      * @return
153      */
154     @SuppressWarnings('ParameterReassignment')
155     String getQueueUrl(String queueName) {
156         if (queueName.startsWith('http://') || queueName.startsWith('https://')) {
157             return queueName
158         }
159         if (configuration.queueNamePrefix) {
160             queueName = configuration.queueNamePrefix + queueName
161         }
162         if (!queueUrlByNames && caching) {
163             loadQueues()
164         }
165 
166         String queueUrl = caching ? queueUrlByNames[queueName] : getQueueUrlDirect(queueName)
167         if (!queueUrl && (configuration.autoCreateQueue)) {
168             queueUrl = createQueue(queueName)
169         }
170         if (!queueUrl) {
171             throw new QueueDoesNotExistException("Queue ${queueName} not found")
172         }
173         return queueUrl
174     }
175 
176     @Override
177     String getQueueArn(String queueName) {
178         GetQueueAttributesResult attributes = client.getQueueAttributes(getQueueUrl(queueName), Collections.singletonList(QUEUE_ARN))
179         return attributes.attributes[QUEUE_ARN]
180     }

182     /**
183      *
184      * @param reload
185      * @return
186      */
187     List<String> listQueueNames(boolean reload) {
188         if (!queueUrlByNames || reload) {
189             loadQueues()
190         }
191         return queueUrlByNames?.keySet()?.sort()
192     }
193 
194     /**
195      *
196      * @param reload
197      * @return
198      */
199     List<String> listQueueUrls(boolean reload) {
200         if (!queueUrlByNames || reload) {
201             loadQueues()
202         }
203         return queueUrlByNames?.values()?.sort()
204     }

206     /**
207      *
208      * @param queueName
209      * @param maxNumberOfMessages
210      * @param visibilityTimeout
211      * @param waitTimeSeconds
212      * @return
213      */
214     List<Message> receiveMessages(String queueName, int maxNumberOfMessages, int visibilityTimeout, int waitTimeSeconds) {
215         String queueUrl = getQueueUrl(queueName)
216 
217         ReceiveMessageRequest receiveMessageRequest = new ReceiveMessageRequest(queueUrl)
218         if (maxNumberOfMessages) {
219             receiveMessageRequest.maxNumberOfMessages = maxNumberOfMessages
220         }
221         if (visibilityTimeout) {
222             receiveMessageRequest.visibilityTimeout = visibilityTimeout
223         }
224         if (waitTimeSeconds) {
225             receiveMessageRequest.waitTimeSeconds = waitTimeSeconds
226         }
227         List<Message> messages = client.receiveMessage(receiveMessageRequest).messages
228         log.debug "Messages received (count=${messages.size()})"
229         return messages
230     }

232     /**
233      *
234      * @param queueName
235      * @param messageBody
236      * @param delaySeconds
237      * @return
238      */
239     String sendMessage(String queueName, String messageBody, int delaySeconds, String groupId) {
240         String queueUrl = getQueueUrl(queueName)
241 
242         SendMessageRequest request = new SendMessageRequest(queueUrl, messageBody)
243         if (delaySeconds) {
244             request.delaySeconds = delaySeconds
245         }
246         if (groupId) {
247             request.messageGroupId = groupId
248         }
249         String messageId = client.sendMessage(request).messageId
250         log.debug "Message sent (messageId=$messageId)"
251         return messageId
252     }
253 
254     /**
255      *
256      * @param queueName
257      * @param messageBody
258      * @param delaySeconds
259      * @return
260      */
261     String sendMessage(String queueName, String messageBody, Consumer<SendMessageRequest> messageConfiguration) {
262         String queueUrl = getQueueUrl(queueName)
263 
264         SendMessageRequest request = new SendMessageRequest(queueUrl, messageBody)
265 
266         messageConfiguration.accept(request)
267 
268         String messageId = client.sendMessage(request).messageId
269         log.debug "Message sent (messageId=$messageId)"
270         return messageId
271     }

273     void assertDefaultQueueName() {
274         assert configuration.queue, 'Default queue must be defined'
275     }

277     // PRIVATE
278     private static String getQueueNameFromUrl(String queueUrl) {
279         return queueUrl.tokenize('/').last()
280     }
281 
282     @SuppressWarnings('ReturnNullFromCatchBlock')
283     private String getQueueUrlDirect(String queueName) {
284         try {
285             return client.getQueueUrl(queueName).queueUrl
286         } catch (QueueDoesNotExistException ignored) {
287             return null
288         }
289     }
290 
291     @Synchronized
292     private void addQueue(String queueUrl) {
293         assert queueUrl, 'Invalid queueUrl'
294         queueUrlByNames[getQueueNameFromUrl(queueUrl)] = queueUrl
295     }
296 
297     @Synchronized
298     private void loadQueues() {
299         ListQueuesRequest listQueuesRequest = new ListQueuesRequest()
300         if (configuration.queueNamePrefix) {
301             listQueuesRequest.withQueueNamePrefix configuration.queueNamePrefix
302         }
303         List queueUrls = client.listQueues(listQueuesRequest).queueUrls
304         queueUrlByNames.clear()
305         queueUrlByNames.putAll queueUrls?.collectEntries { queueUrl ->
306             [getQueueNameFromUrl(queueUrl), queueUrl]
307         }
308     }
309 
310     @Synchronized
311     private void removeQueue(String queueUrl) {
312         assert queueUrl, 'Invalid queueUrl'
313         queueUrlByNames.remove(getQueueNameFromUrl(queueUrl))
314     }
