018 package com.agorapulse.micronaut.aws.kinesis.worker;
019 
020 import com.agorapulse.micronaut.aws.kinesis.annotation.KinesisListener;
021 import com.amazonaws.services.cloudwatch.AmazonCloudWatch;
022 import com.amazonaws.services.dynamodbv2.AmazonDynamoDB;
023 import com.amazonaws.services.kinesis.AmazonKinesis;
024 import com.amazonaws.services.kinesis.clientlibrary.lib.worker.KinesisClientLibConfiguration;
025 import com.amazonaws.services.kinesis.clientlibrary.lib.worker.Worker;
026 import com.amazonaws.services.kinesis.model.Record;
027 import io.micronaut.context.event.ApplicationEventPublisher;
028 import io.micronaut.core.util.StringUtils;
029 import org.slf4j.Logger;
030 import org.slf4j.LoggerFactory;
031 
032 import java.util.List;
033 import java.util.Optional;
034 import java.util.concurrent.CopyOnWriteArrayList;
035 import java.util.concurrent.ExecutorService;
036 import java.util.concurrent.Executors;
037 import java.util.function.BiConsumer;
038 
039 class DefaultKinesisWorker implements KinesisWorker {
040 
041     private static final Logger LOGGER = LoggerFactory.getLogger(KinesisListener.class);
042 
043     DefaultKinesisWorker(
044         KinesisClientLibConfiguration configuration,
045         ApplicationEventPublisher applicationEventPublisher,
046         Optional<AmazonDynamoDB> amazonDynamoDB,
047         Optional<AmazonKinesis> amazonKinesis,
048         Optional<AmazonCloudWatch> amazonCloudWatch
049     ) {
050         this.configuration = configuration;
051         this.applicationEventPublisher = applicationEventPublisher;
052         this.amazonDynamoDB = amazonDynamoDB;
053         this.amazonKinesis = amazonKinesis;
054         this.amazonCloudWatch = amazonCloudWatch;
055     }
056 
057     @Override
058     public void start() {
059         Worker.Builder builder = new Worker.Builder().config(configuration);
060 
061         if (StringUtils.isEmpty(configuration.getDynamoDBEndpoint())) {
062             amazonDynamoDB.ifPresent(builder::dynamoDBClient);
063         }
064 
065         if (StringUtils.isEmpty(configuration.getKinesisEndpoint())) {
066             amazonKinesis.ifPresent(builder::kinesisClient);
067         }
068 
069         amazonCloudWatch.ifPresent(builder::cloudWatchClient);
070 
071         builder.recordProcessorFactory(DefaultRecordProcessorFactory.create((string, record) ->
072             consumers.forEach(c -> {
073                 try {
074                     c.accept(string, record);
075                 } catch (Exception e) {
076                     LOGGER.error("Exception processing Kinesis record " + record + " decoded as " + string, e);
077                 }
078             }))
079         );
080 
081         builder.workerStateChangeListener(s -> applicationEventPublisher.publishEvent(new WorkerStateEvent(s, configuration.getStreamName())));
082 
083         try {
084             LOGGER.info("Starting Kinesis worker for {}", configuration.getStreamName());
085             worker = builder.build();
086             executorService.execute(worker);
087         } catch (Exception t) {
088             LOGGER.error("Caught throwable while processing Kinesis data.", t);
089         }
090     }
091 
092     @Override
093     public void shutdown() {
094         if (worker != null) {
095             try {
096                 worker.shutdown();
097             } catch (Exception e) {
098                 LOGGER.warn("Exception shutting down worker", e);
099             }
100         }
101         try {
102             executorService.shutdown();
103         } catch (Exception e) {
104             LOGGER.warn("Exception shutting down worker's executor service", e);
105         }
106     }
107 
108     @Override
109     public void addConsumer(BiConsumer<String, Record> next) {
110         consumers.add(next);
111     }
112 
113     private final KinesisClientLibConfiguration configuration;
114     private final ExecutorService executorService = Executors.newCachedThreadPool();
115     private final ApplicationEventPublisher applicationEventPublisher;
116     private final Optional<AmazonDynamoDB> amazonDynamoDB;
117     private final Optional<AmazonKinesis> amazonKinesis;
118     private final Optional<AmazonCloudWatch> amazonCloudWatch;
119     private final List<BiConsumer<String, Record>> consumers = new CopyOnWriteArrayList<>();
120     private Worker worker;
121 }
