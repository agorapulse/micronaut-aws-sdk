225     String createPlatformEndpoint(String platformApplicationArn, String deviceToken, String customUserData) {
226         try {
227             log.debug("Creating platform endpoint with token $deviceToken")
228             CreatePlatformEndpointRequest request = new CreatePlatformEndpointRequest()
229                 .withPlatformApplicationArn(platformApplicationArn)
230                 .withToken(deviceToken)
231             if (customUserData) {
232                 request.customUserData = customUserData
233             }
234             return client.createPlatformEndpoint(request).endpointArn
235         } catch (InvalidParameterException ipe) {
236             String message = ipe.errorMessage
237             log.debug("Exception message: $message")
238             Pattern p = Pattern.compile('.*Endpoint (arn:aws:sns[^ ]+) already exists with the same Token.*')
239             Matcher m = p.matcher(message)
240             if (m.matches()) {
241                 // The platform endpoint already exists for this token, but with additional custom data that
242                 // createEndpoint doesn't want to overwrite. Just use the existing platform endpoint.
243                 return m.group(1)
244             }
245             // Rethrow the exception, the input is actually bad.
246             throw ipe
247         }
248     }

250     String validateDeviceToken(String platformApplicationArn, String endpointArn, String deviceToken, String customUserData = '') {
251         log.debug 'Retrieving platform endpoint data...'
252         // Look up the platform endpoint and make sure the data in it is current, even if it was just created.
253         try {
254             GetEndpointAttributesResult result = client.getEndpointAttributes(new GetEndpointAttributesRequest().withEndpointArn(endpointArn))
255             if (result.attributes.get('Token') == deviceToken && result.attributes.get('Enabled').equalsIgnoreCase(Boolean.TRUE.toString())) {
256                 setEndpointAttributes(endpointArn, [
257                     CustomUserData: customUserData
258                 ])
259                 return endpointArn
260             }
261         } catch (NotFoundException ignored) {
262             // We had a stored ARN, but the platform endpoint associated with it disappeared. Recreate it.
263             return createPlatformEndpoint(platformApplicationArn, deviceToken, customUserData)
264         }
265 
266         log.debug 'Platform endpoint update required...'
267 
268         // The platform endpoint is out of sync with the current data, update the token and enable it.
269         log.debug("Updating platform endpoint $endpointArn")
270         try {
271             setEndpointAttributes(endpointArn, [
272                 CustomUserData: customUserData,
273                 Enabled: Boolean.TRUE.toString(),
274             ])
275             return endpointArn
276         } catch (InvalidParameterException ignored) {
277             deleteEndpoint(endpointArn)
278             return createPlatformEndpoint(platformApplicationArn, deviceToken, customUserData)
279         }
280     }

335     @SuppressWarnings('UnnecessarySubstring')
336     private String ensureTopicArn(String nameOrArn) {
337         if (!nameOrArn) {
338             return ''
339         }
340         if (nameOrArn.startsWith('arn:aws:sns')) {
341             return nameOrArn
342         }
343 
344         if (namesToArn.containsKey(nameOrArn)) {
345             return namesToArn[nameOrArn]
346         }
347 
348         listTopics().takeUntil({ Topic topic -> topic.topicArn.endsWith(":$nameOrArn") } as Predicate<Topic>).subscribe { Topic topic ->
349             String topicName = topic.topicArn.substring(topic.topicArn.lastIndexOf(':') + 1)
350             namesToArn[topicName] = topic.topicArn
351         }
352 
353         String topicArn = namesToArn[nameOrArn]
354 
355         if (topicArn) {
356             return topicArn
357         }
358 
359         return createTopic(nameOrArn)
360     }
