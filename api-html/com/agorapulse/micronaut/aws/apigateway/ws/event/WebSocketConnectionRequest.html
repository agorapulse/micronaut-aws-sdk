24 /**
25  * Event which can be used as a parameter of Lambda functions which handles <code>$connect</code> or <code>$disconnect</code> route.
26  */
27 public class WebSocketConnectionRequest extends WebSocketRequest {
28 
29     private Map<String, String> headers;
30     private Map<String, List<String>> multiValueHeaders;
31 
32     public Map<String, String> getHeaders() {
33         return headers;
34     }
35 
36     public void setHeaders(Map<String, String> headers) {
37         this.headers = headers;
38     }
39 
40     public WebSocketConnectionRequest withHeaders(Map<String, String> headers) {
41         this.headers = headers;
42         return this;
43     }
44 
45     public Map<String, List<String>> getMultiValueHeaders() {
46         return multiValueHeaders;
47     }
48 
49     public void setMultiValueHeaders(Map<String, List<String>> multiValueHeaders) {
50         this.multiValueHeaders = multiValueHeaders;
51     }
52 
53     public WebSocketConnectionRequest withMultiValueHeaders(Map<String, List<String>> multiValueHeaders) {
54         this.multiValueHeaders = multiValueHeaders;
55         return this;
56     }
57 
58     // CHECKSTYLE:OFF
59     @Override
60     public boolean equals(Object o) {
61 
62         if (this == o) return true;
63         if (o == null || getClass() != o.getClass()) return false;
64         if (!super.equals(o)) return false;
65         WebSocketConnectionRequest that = (WebSocketConnectionRequest) o;
66         return Objects.equals(headers, that.headers) &&
67             Objects.equals(multiValueHeaders, that.multiValueHeaders);
68     }
69 
70     @Override
71     public int hashCode() {
72         return Objects.hash(super.hashCode(), headers, multiValueHeaders);
73     }
74 
75     @Override
76     public String toString() {
77         return "WebSocketConnectionRequest{" +
78             "headers=" + headers +
79             ", multiValueHeaders=" + multiValueHeaders +
80             "requestContext=" + getRequestContext() +
81             ", body='" + getBody() + '\'' +
82             ", isBase64Encoded=" + getIsBase64Encoded() +
83             '}';
84     }
