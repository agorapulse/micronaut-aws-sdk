18 package com.agorapulse.micronaut.aws;
19 
20 import com.amazonaws.client.builder.AwsClientBuilder;
21 import com.amazonaws.regions.AwsRegionProvider;
22 
23 import java.util.Optional;
24 
25 public interface RegionAndEndpointConfiguration {
26 
27     String getRegion();
28 
29     String getEndpoint();
30 
31     default <C, B extends AwsClientBuilder<B, C>> B configure(B builder, AwsRegionProvider awsRegionProvider) {
32         String region = Optional.ofNullable(getRegion()).orElseGet(awsRegionProvider::getRegion);
33         if (getEndpoint() != null) {
34             builder.setEndpointConfiguration(new AwsClientBuilder.EndpointConfiguration(getEndpoint(), region));
35         } else {
36             builder.setRegion(region);
37         }
38         return builder;
39     }
40 
41 }
