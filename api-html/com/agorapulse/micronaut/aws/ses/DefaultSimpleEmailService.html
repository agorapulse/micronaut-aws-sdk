042 /**
043  * Default implementation of simple email service.
044  */
045 @Slf4j
046 @Singleton
047 @CompileStatic
048 @Requires(classes = AmazonSimpleEmailService)
049 @SuppressWarnings([
050     'NoWildcardImports',
051     'DuplicateStringLiteral',
052 ])
053 class DefaultSimpleEmailService implements SimpleEmailService {
054 
055     static TransactionalEmail email(@DelegatesTo(value = TransactionalEmail, strategy = Closure.DELEGATE_FIRST) Closure composer) {
056         Closure cl = composer.clone() as Closure
057         TransactionalEmail transactionalEmail = new TransactionalEmail()
058         cl.delegate = transactionalEmail
059         cl.resolveStrategy = Closure.DELEGATE_FIRST
060         cl()
061 
062         if (!transactionalEmail.recipients) {
063             throw new IllegalArgumentException("Email does not contain recepients: $transactionalEmail")
064         }
065 
066         transactionalEmail
067     }
068 
069     private final AmazonSimpleEmailService client
070     private final SimpleEmailServiceConfiguration configuration
071 
072     DefaultSimpleEmailService(AmazonSimpleEmailService client, SimpleEmailServiceConfiguration configuration) {
073         this.client = client
074         this.configuration = configuration
075     }
076 
077     EmailDeliveryStatus send(TransactionalEmail email) throws Exception {
078         if (email.attachments) {
079             return sendEmailWithAttachment(email)
080         }
081 
082         return sendWithoutAttachments(email)
083     }
084 
085     private static EmailDeliveryStatus handleSend(TransactionalEmail email, Closure c) {
086         try {
087             c()
088             return EmailDeliveryStatus.STATUS_DELIVERED
089         } catch (AmazonClientException exception) {
090             if (exception.message.find('Address blacklisted')) {
091                 log.debug "Address blacklisted destinationEmails=${email.recipients}"
092                 return EmailDeliveryStatus.STATUS_BLACKLISTED
093             }
094             log.warn "An amazon service exception was catched while sending email: destinationEmails=$email.recipients," +
095                 " from=$email.from), replyTo=$email.replyTo, subject=$email.subject", exception
096             return EmailDeliveryStatus.STATUS_NOT_DELIVERED
097         }
098     }

100     @SuppressWarnings(['LineLength', 'ElseBlockBraces', 'JavaIoPackageAccess', 'AbcMetric'])
101     private EmailDeliveryStatus sendEmailWithAttachment(TransactionalEmail email) throws UnsupportedAttachmentTypeException {
102         Session session = Session.getInstance(new Properties())
103         MimeMessage mimeMessage = new MimeMessage(session)
104 
105         if (email.from) {
106             mimeMessage.from = new InternetAddress(email.from)
107         } else if (configuration.sourceEmail.present) {
108             mimeMessage.from = new InternetAddress(configuration.sourceEmail.get())
109         }
110 
111         if (email.replyTo) {
112             mimeMessage.replyTo = [new InternetAddress(email.replyTo)] as InternetAddress[]
113         }
114 
115         email.recipients.each { recipient ->
116             mimeMessage.addRecipients(TO, new InternetAddress(recipient))
117         }
118 
119         mimeMessage.subject = configuration.subjectPrefix.map { prefix ->
120             "$prefix $email.subject".toString()
121         } orElse(email.subject)
122 
123         MimeMultipart mimeMultipart = new MimeMultipart()
124 
125         BodyPart p = new MimeBodyPart()
126         p.setContent(email.htmlBody, 'text/html')
127         mimeMultipart.addBodyPart(p)
128 
129         for (TransactionalEmailAttachment attachment : email.attachments) {
130             if (!MimeType.isMimeTypeSupported(attachment.mimeType)) {
131                 throw new UnsupportedAttachmentTypeException()
132             }
133 
134             MimeBodyPart mimeBodyPart = new MimeBodyPart()
135             mimeBodyPart.fileName = attachment.filename
136             mimeBodyPart.setDescription(attachment.description, 'UTF-8')
137             DataSource ds = new ByteArrayDataSource(new FileInputStream(new File(attachment.filepath)), attachment.mimeType)
138             mimeBodyPart.dataHandler = new DataHandler(ds)
139             mimeMultipart.addBodyPart(mimeBodyPart)
140         }
141 
142         mimeMessage.content = mimeMultipart
143 
144         ByteArrayOutputStream outputStream = new ByteArrayOutputStream()
145         mimeMessage.writeTo(outputStream)
146         RawMessage rawMessage = new RawMessage(ByteBuffer.wrap(outputStream.toByteArray()))
147 
148         SendRawEmailRequest rawEmailRequest = new SendRawEmailRequest(rawMessage)
149 
150         rawEmailRequest.destinations = email.recipients
151         rawEmailRequest.source = email.from ?: configuration.sourceEmail.orElse(null)
152 
153         return handleSend(email) {
154             client.sendRawEmail(rawEmailRequest)
155             return EmailDeliveryStatus.STATUS_DELIVERED
156         }
157     }
158 
159     /**
160      *
161      * @param destinationEmail
162      * @param subject
163      * @param htmlBody
164      * @param sourceEmail
165      * @param replyToEmail
166      * @return 1 if successful, 0 if not sent, -1 if blacklisted
167      */
168     @SuppressWarnings(['LineLength', 'ElseBlockBraces'])
169     private EmailDeliveryStatus sendWithoutAttachments(TransactionalEmail email) {
170         Destination destination = new Destination(email.recipients)
171 
172         String subject = configuration.subjectPrefix.map { prefix ->
173             "$prefix $email.subject".toString()
174         } orElse(email.subject)
175 
176         Content messageSubject = new Content(subject)
177 
178         Body messageBody = new Body().withHtml(new Content(email.htmlBody))
179         Message message = new Message(messageSubject, messageBody)
180 
181         return handleSend(email) {
182             String from = email.from ?: configuration.sourceEmail.orElse(null)
183             SendEmailRequest sendEmailRequest = new SendEmailRequest(from, destination, message)
184             if (email.replyTo) {
185                 sendEmailRequest.replyToAddresses = singletonList(email.replyTo)
186             }
187             client.sendEmail(sendEmailRequest)
188         }
189     }
