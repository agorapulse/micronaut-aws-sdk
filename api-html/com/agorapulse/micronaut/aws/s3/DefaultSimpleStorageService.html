078     @SuppressWarnings('DuplicateStringLiteral')
079     static ObjectMetadata buildMetadataFromType(String type,
080                                                 String fileExtension,
081                                                 CannedAccessControlList cannedAcl = null) {
082         Map contentInfo
083         if (HTTP_CONTENTS[type]) {
084             contentInfo = HTTP_CONTENTS[type] as Map
085         } else if (type in ['image', 'photo']) {
086             // Return image/jpeg for images to fix Safari issue (download image instead of inline display)
087             contentInfo = [contentType: "image/${fileExtension == 'jpg' ? 'jpeg' : fileExtension}"]
088         } else if (fileExtension == 'swf') {
089             contentInfo = CONTENT_FLASH
090         } else {
091             contentInfo = CONTENT_BINARY
092         }
093 
094         ObjectMetadata metadata = new ObjectMetadata()
095         metadata.contentType = contentInfo.contentType as String
096         if (contentInfo.contentDisposition) {
097             metadata.contentDisposition = contentInfo.contentDisposition as String
098         }
099         if (cannedAcl) {
100             metadata.setHeader('x-amz-acl', cannedAcl.toString())
101         }
102         return metadata
103     }

105     @SuppressWarnings([
106         'DuplicateStringLiteral',
107         'UnnecessarySubstring',
108     ])
109     static String getBucketFromUri(String aURI) {
110         URI uri = new URI(aURI)
111         String path = uri.path.startsWith('/') ? uri.path.substring(1, uri.path.length()) : uri.path
112         if (uri.host) {
113             // direct bucket URI not using any CNAME
114             if (uri.host.endsWith('amazonaws.com')) {
115                 if (uri.host.startsWith('s3')) {
116                     return path.substring(0, path.indexOf('/'))
117                 }
118                 return uri.host.substring(0, uri.host.indexOf('.s3.'))
119             }
120             return uri.host
121         }
122         // consider the bucket name is using CNAME of the same name as the bucket
123         return path.substring(0, path.indexOf('/'))
124     }

126     @SuppressWarnings([
127         'DuplicateStringLiteral',
128         'UnnecessarySubstring',
129     ])
130     static String getKeyFromUri(String aURI) {
131         URI uri = new URI(aURI)
132         String path = uri.path.startsWith('/') ? uri.path.substring(1, uri.path.length()) : uri.path
133         if (uri.host) {
134             // direct bucket URI not using any CNAME
135             if (uri.host.endsWith('amazonaws.com') && uri.host.startsWith('s3')) {
136                 return path.substring(path.indexOf('/') + 1, path.length())
137             }
138             return path
139         }
140         // consider the bucket name is using CNAME of the same name as the bucket
141         return path.substring(path.indexOf('/') + 1, path.length())
142     }

208     /**
209      *
210      * @param String
211      * @param bucketName
212      * @param prefix
213      * @return
214      */
215     @SuppressWarnings('DuplicateStringLiteral')
216     boolean deleteFiles(String bucketName, String prefix) {
217         assert prefix.tokenize('/').size() >= 2, 'Multiple delete are only allowed in sub/sub directories'
218 
219         Flux<Boolean> results = Flux.from(listObjects(bucketName, prefix)).flatMap {
220             Flux.fromIterable(it.objectSummaries)
221         } map {
222             deleteFile(bucketName, it.key)
223         }
224 
225         return results.onErrorResume {
226             log.warn "Exception deleting objects in $bucketName/$prefix", it
227             return Mono.just(false)
228         }.filter { !it }.count().block() == 0
229     }

358      * @return

360     @SuppressWarnings([
361         'CouldBeElvis',
362         'ParameterReassignment',
363     ])
364     String storeMultipartFile(String bucketName, String path, PartData partData, CannedAccessControlList cannedAcl, ObjectMetadata metadata) {
365         if (!metadata) {
366             metadata = new ObjectMetadata()
367         }
368         metadata.setHeader(Headers.S3_CANNED_ACL, cannedAcl.toString())
369         metadata.contentLength = partData.bytes.size()
370         byte[] resultByte = DigestUtils.md5(partData.inputStream)
371         metadata.contentMD5 = resultByte.encodeBase64().toString()
372         partData.contentType.ifPresent { metadata.contentType = it.name }
373         return storeInputStream(bucketName, path, partData.inputStream, metadata)
374     }

381     /**
382      * Move S3 object to different location (key).
383      *
384      * Moving objects is useful in combination with
385      * <a href="https://docs.aws.amazon.com/AmazonS3/latest/dev/how-to-set-lifecycle-configuration-intro.html">
386      *     S3 Lifecycle Configurations
387      * </a> for prefixes.
388      *
389      * @param sourceBucketName      the name of the source bucket
390      * @param sourceKey             the key of the source object
391      * @param destinationBucketName the name of the destination bucket
392      * @param destinationKey        the key of the destination object
393      * @return the destination URL or <code>null</code> if the file wasn't moved
394      */
395     String moveObject(
396         String sourceBucketName,
397         String sourceKey,
398         String destinationBucketName,
399         String destinationKey
400     ) {
401         try {
402             CopyObjectRequest request = new CopyObjectRequest(sourceBucketName, sourceKey, destinationBucketName, destinationKey)
403 
404             S3Object object = client.getObject(sourceBucketName, sourceKey)
405 
406             if (object.taggingCount) {
407                 GetObjectTaggingRequest taggingRequest = new GetObjectTaggingRequest(sourceBucketName, sourceKey)
408                 GetObjectTaggingResult taggingResult = client.getObjectTagging(taggingRequest)
409                 request.withNewObjectTagging(new ObjectTagging(taggingResult.tagSet))
410             }
411 
412             request.withNewObjectMetadata(object.objectMetadata)
413 
414             AccessControlList acl = client.getObjectAcl(sourceBucketName, sourceKey)
415             AccessControlList newAcls = new AccessControlList()
416             newAcls.grantAllPermissions(acl.grantsAsList as Grant[])
417             request.withAccessControlList(newAcls)
418 
419             client.copyObject(request)
420             client.deleteObject(sourceBucketName, sourceKey)
421             return client.getUrl(destinationBucketName, destinationKey)
422         } catch (AmazonClientException e) {
423             log.error("Exception moving object $sourceBucketName/$sourceKey to $destinationBucketName/$destinationKey", e)
424         }
425         return null
426     }
