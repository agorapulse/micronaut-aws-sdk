18 package com.agorapulse.micronaut.aws.dynamodb;
19 
20 import com.agorapulse.micronaut.aws.util.AWSClientConfiguration;
21 import com.amazon.dax.client.dynamodbv2.AmazonDaxClient;
22 import com.amazon.dax.client.dynamodbv2.AmazonDaxClientBuilder;
23 import com.amazon.dax.client.dynamodbv2.ClientConfig;
24 import com.amazonaws.ClientConfiguration;
25 import com.amazonaws.auth.AWSCredentialsProvider;
26 import com.amazonaws.regions.AwsRegionProvider;
27 import com.amazonaws.services.dynamodbv2.AmazonDynamoDB;
28 import io.micronaut.context.annotation.Bean;
29 import io.micronaut.context.annotation.Factory;
30 import io.micronaut.context.annotation.Requires;
31 import io.micronaut.context.annotation.Value;
32 
33 import javax.inject.Singleton;
34 import java.util.Optional;
35 import java.util.concurrent.TimeUnit;
36 
37 /**
38  * Factory class which provides the {@link AmazonDynamoDB} instance backed by {@link AmazonDaxClient}.
39  */
40 @Factory
41 @Requires(classes = AmazonDaxClient.class)
42 public class DynamoDBDaxFactory {
43 
44     @Bean
45     @Singleton
46     @Requires(property = "aws.dax.endpoint")
47     AmazonDynamoDB amazonDynamoDB(
48         @Value("${aws.dax.endpoint}") String endpoint,
49         AWSClientConfiguration clientConfiguration,
50         AWSCredentialsProvider credentialsProvider,
51         AwsRegionProvider awsRegionProvider,
52         @Value("${aws.dax.region}") Optional<String> region
53     ) {
54         ClientConfig clientConfig = migrateClientConfig(clientConfiguration.getClientConfiguration())
55             .withEndpoints(endpoint)
56             .withRegion(region.orElseGet(awsRegionProvider::getRegion))
57             .withCredentialsProvider(credentialsProvider);
58         return AmazonDaxClientBuilder.standard()
59             .withCredentials(credentialsProvider)
60             .withRegion(region.orElseGet(awsRegionProvider::getRegion))
61             .withClientConfiguration(clientConfig)
62             .build();
63     }
64 
65     private ClientConfig migrateClientConfig(ClientConfiguration clientConfiguration) {
66         return new ClientConfig()
67             .withConnectTimeout(clientConfiguration.getConnectionTimeout(), TimeUnit.MILLISECONDS)
68             .withMaxPendingConnectsPerHost(clientConfiguration.getMaxConnections())
69             .withReadRetries(clientConfiguration.getMaxErrorRetry())
70             .withWriteRetries(clientConfiguration.getMaxErrorRetry());
71     }
72 }
