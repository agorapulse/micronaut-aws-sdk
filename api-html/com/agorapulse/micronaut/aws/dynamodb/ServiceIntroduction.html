042 /**
043  * Introduction for {@link Service} annotation.
044  */
045 @Singleton
046 @Requires(classes = IDynamoDBMapper.class)
047 public class ServiceIntroduction implements MethodInterceptor<Object, Object> {

049     private static final String HASH = "hash";
050     private static final String RANGE = "range";
051 
052     private static class HashAndRange {
053         Argument<?> hashKey;
054         Argument<?> rangeKey;
055 
056         boolean isValid() {
057             return hashKey != null;
058         }
059 
060     }

062     private final IDynamoDBMapper mapper;
063     private final AmazonDynamoDB amazonDynamoDB;
064     private final DynamoDBServiceProvider dynamoDBServiceProvider;

066     public ServiceIntroduction(IDynamoDBMapper mapper, AmazonDynamoDB amazonDynamoDB, DynamoDBServiceProvider dynamoDBServiceProvider) {
067         this.mapper = mapper;
068         this.amazonDynamoDB = amazonDynamoDB;
069         this.dynamoDBServiceProvider = dynamoDBServiceProvider;
070     }

072     @Override
073     public Object intercept(MethodInvocationContext<Object, Object> context) {
074         AnnotationValue<Service> serviceAnnotationValue = context.getAnnotation(Service.class);
075 
076         if (serviceAnnotationValue == null) {
077             throw new IllegalStateException("Invocation context is missing required annotation Service");
078         }
079 
080         Class type = serviceAnnotationValue.getValue(Class.class).orElseThrow(() -> new IllegalArgumentException("Annotation is missing the type value!"));
081         DynamoDBService service = dynamoDBServiceProvider.findOrCreate(type);
082 
083         try {
084             return doIntercept(context, type, service);
085         } catch (ResourceNotFoundException ignored) {
086             service.createTable();
087             return doIntercept(context, type, service);
088         }
089     }

091     private Object doIntercept(MethodInvocationContext<Object, Object> context, Class type, DynamoDBService service) {
092         String methodName = context.getMethodName();
093         if (methodName.startsWith("save")) {
094             return handleSave(service, context);
095         }
096 
097         if (methodName.startsWith("get") || methodName.startsWith("load")) {
098             return handleGet(service, context);
099         }
100 
101         if (context.getTargetMethod().isAnnotationPresent(Query.class)) {
102             DetachedQuery criteria = evaluateAnnotationType(context.getTargetMethod().getAnnotation(Query.class).value(), context);
103 
104             if (methodName.startsWith("count")) {
105                 return criteria.count(mapper);
106             }
107 
108             if (methodName.startsWith("delete")) {
109                 return service.deleteAllByConditions(criteria.resolveExpression(mapper), Collections.emptyMap());
110             }
111 
112             return flowableOrList(criteria.query(mapper), context.getReturnType().getType());
113         }
114 
115         if (context.getTargetMethod().isAnnotationPresent(Update.class)) {
116             DetachedUpdate criteria = evaluateAnnotationType(context.getTargetMethod().getAnnotation(Update.class).value(), context);
117 
118             return criteria.update(mapper, amazonDynamoDB);
119         }
120 
121         if (context.getTargetMethod().isAnnotationPresent(Scan.class)) {
122             DetachedScan criteria = evaluateAnnotationType(context.getTargetMethod().getAnnotation(Scan.class).value(), context);
123 
124             if (methodName.startsWith("count")) {
125                 return criteria.count(mapper);
126             }
127 
128             return flowableOrList(criteria.scan(mapper), context.getReturnType().getType());
129         }
130 
131         if (methodName.startsWith("count")) {
132             return simpleHashAndRangeQuery(type, context).count(mapper);
133         }
134 
135         if (methodName.startsWith("delete")) {
136             return handleDelete(type, service, context);
137         }
138 
139         if (methodName.startsWith("query") || methodName.startsWith("findAll") || methodName.startsWith("list")) {
140             return flowableOrList(simpleHashAndRangeQuery(type, context).query(mapper), context.getReturnType().getType());
141         }
142 
143         throw new UnsupportedOperationException("Cannot implement method " + context.getExecutableMethod());
144     }
145 
146     private Object flowableOrList(Flowable result, Class type) {
147         if (List.class.isAssignableFrom(type)) {
148             return result.toList().blockingGet();
149         }
150         return result;
151     }

153     private <T> T evaluateAnnotationType(Class<? extends Function<Map<String, Object>, T>> updateDefinitionType, MethodInvocationContext<Object, Object> context) {
154         Map<String, Object> parameterValueMap = new StrictMap<>(context.getParameterValueMap());
155 
156         if (Closure.class.isAssignableFrom(updateDefinitionType)) {
157             try {
158                 Closure<T> closure = (Closure<T>) updateDefinitionType.getConstructor(Object.class, Object.class).newInstance(parameterValueMap, parameterValueMap);
159                 closure.setDelegate(parameterValueMap);
160                 closure.setResolveStrategy(Closure.DELEGATE_FIRST);
161                 return closure.call(parameterValueMap);
162             } catch (InstantiationException | InvocationTargetException | NoSuchMethodException | IllegalAccessException e) {
163                 throw new IllegalArgumentException("Cannot instantiate closure! Type: " + updateDefinitionType, e);
164             }
165         }
166 
167         try {
168             return updateDefinitionType.newInstance().apply(parameterValueMap);
169         } catch (InstantiationException | IllegalAccessException e) {
170             throw new IllegalArgumentException("Cannot instantiate function! Type: " + updateDefinitionType, e);
171         }
172 
173     }

175     private Object handleSave(DynamoDBService service, MethodInvocationContext<Object, Object> context) {
176         Map<String, MutableArgumentValue<?>> params = context.getParameters();
177         Argument[] args = context.getArguments();
178 
179         if (args.length > 1) {
180             throw new UnsupportedOperationException("Method expects at most 1 parameters - item, iterable of items or array of items");
181         }
182 
183         Argument itemArgument = args[0];
184         Object item = params.get(itemArgument.getName()).getValue();
185 
186         if (itemArgument.getType().isArray()) {
187             return service.saveAll(Arrays.asList((Object[]) item));
188         }
189 
190         if (Iterable.class.isAssignableFrom(itemArgument.getType())) {
191             return service.saveAll(toList((Iterable) item));
192         }
193 
194         return service.save(item);
195     }

197     private Object handleDelete(Class type, DynamoDBService service, MethodInvocationContext<Object, Object> context) {
198         Map<String, MutableArgumentValue<?>> params = context.getParameters();
199         Argument[] args = context.getArguments();
200 
201         if (args.length == 1) {
202             Argument itemArgument = args[0];
203             Object item = params.get(itemArgument.getName()).getValue();
204 
205             if (itemArgument.getType().isArray() && type.isAssignableFrom(itemArgument.getType().getComponentType())) {
206                 service.deleteAll(Arrays.asList((Object[]) item));
207                 return null;
208             }
209 
210             if (Iterable.class.isAssignableFrom(itemArgument.getType()) && type.isAssignableFrom(itemArgument.getTypeParameters()[0].getType())) {
211                 service.deleteAll(toList((Iterable) item));
212                 return null;
213             }
214 
215             if (type.isAssignableFrom(itemArgument.getType())) {
216                 service.delete(item);
217                 return null;
218             }
219         }
220 
221         if (args.length > 2) {
222             throw new UnsupportedOperationException("Method expects at most 2 parameters - hash key and range key, an item, iterable of items or an array of items");
223         }
224 
225         HashAndRange hashAndRange = findHashAndRange(args);
226         Object hashKey = params.get(hashAndRange.hashKey.getName()).getValue();
227 
228         if (hashAndRange.rangeKey == null) {
229             service.deleteByHash(hashKey);
230             return null;
231         }
232 
233         Object rangeKey = params.get(hashAndRange.rangeKey.getName()).getValue();
234         service.delete(hashKey, rangeKey);
235         
236         return null;
237     }

239     private Object handleGet(DynamoDBService service, MethodInvocationContext<Object, Object> context) {
240         Map<String, MutableArgumentValue<?>> params = context.getParameters();
241         Argument[] args = context.getArguments();
242 
243         if (args.length > 2) {
244             throw new UnsupportedOperationException("Method expects at most 2 parameters - hash key and range key");
245         }
246 
247         HashAndRange hashAndRange = findHashAndRange(args);
248         Object hashKey = params.get(hashAndRange.hashKey.getName()).getValue();
249 
250         if (hashAndRange.rangeKey == null) {
251             return service.get(hashKey);
252         }
253 
254         Object rangeKey = params.get(hashAndRange.rangeKey.getName()).getValue();
255 
256         if (hashAndRange.rangeKey.getType().isArray()) {
257             return service.getAll(hashKey, Arrays.asList((Object[]) rangeKey));
258         }
259 
260         if (Iterable.class.isAssignableFrom(hashAndRange.rangeKey.getType())) {
261             return service.getAll(hashKey, toList((Iterable) rangeKey));
262         }
263 
264         return service.get(hashKey, rangeKey);
265     }

267     private DetachedQuery simpleHashAndRangeQuery(
268         Class type,
269         MethodInvocationContext<Object, Object> context
270     ) {
271         Map<String, MutableArgumentValue<?>> params = context.getParameters();
272         Argument[] args = context.getArguments();
273 
274         if (args.length > 2) {
275             throw new UnsupportedOperationException("Method expects at most 2 parameters - hash key and optional range key");
276         }
277         HashAndRange hashAndRange = findHashAndRange(args);
278         Object hashKey = params.get(hashAndRange.hashKey.getName()).getValue();
279 
280         if (hashAndRange.rangeKey == null) {
281             return Builders.query(type, q -> q.hash(hashKey));
282         }
283 
284         Object rangeKey = params.get(hashAndRange.rangeKey.getName()).getValue();
285         return Builders.query(type, q -> q.hash(hashKey).range(r -> r.eq(rangeKey)));
286     }

289     private HashAndRange findHashAndRange(Argument[] arguments) {
290         HashAndRange names = new HashAndRange();
291         for (Argument<?> argument : arguments) {
292             if (argument.isAnnotationPresent(RangeKey.class) || argument.getName().toLowerCase().contains(RANGE)) {
293                 names.rangeKey = argument;
294             } else if (argument.isAnnotationPresent(HashKey.class) || argument.getName().toLowerCase().contains(HASH)) {
295                 names.hashKey = argument;
296             }
297         }
298 
299         if (!names.isValid()) {
300             throw new UnsupportedOperationException("Method needs to have at least one argument annotated with @HashKey or with called 'hash'");
301         }
302 
303         return names;
304     }

306     private static <T> List<T> toList(Iterable<T> iterable) {
307         if (iterable instanceof List) {
308             return (List<T>) iterable;
309         }
310 
311         List<T> ret = new ArrayList<>();
312         iterable.forEach(ret::add);
313         return ret;
314     }
