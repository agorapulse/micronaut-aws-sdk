018 package com.agorapulse.micronaut.aws.dynamodb.builder;
019 
020 import com.amazonaws.services.dynamodbv2.datamodeling.DynamoDBScanExpression;
021 import com.amazonaws.services.dynamodbv2.model.ConditionalOperator;
022 import groovy.lang.Closure;
023 import groovy.lang.DelegatesTo;
024 import groovy.transform.stc.ClosureParams;
025 import groovy.transform.stc.FromString;
026 import space.jasan.support.groovy.closure.ConsumerWithDelegate;
027 
028 import java.util.Arrays;
029 import java.util.function.Consumer;
030 
031 /**
032  * Builder for DynamoDB scans.
033  * @param <T> type of the DynamoDB entity
034  */
035 public interface ScanBuilder<T> extends DetachedScan<T> {
036 
037     /**
038      * Demand consistent reads.
039      * @param read the read keyword
040      * @return self
041      */
042     ScanBuilder<T> consistent(Builders.Read read);
043 
044     /**
045      * Demand inconsistent reads.
046      * @param read the read keyword
047      * @return self
048      */
049     ScanBuilder<T> inconsistent(Builders.Read read);
050 
051     /**
052      * Select the index on which this scan will be executed.
053      * @param name the name of the index to be used
054      * @return self
055      */
056     ScanBuilder<T> index(String name);
057 
058     /**
059      * One or more filter conditions.
060      *
061      * @param conditions consumer to build the conditions
062      * @return self
063      */
064     ScanBuilder<T> filter(Consumer<RangeConditionCollector<T>> conditions);
065 
066     /**
067      * One or more filter conditions in conjunction.
068      *
069      * @param conditions consumer to build the conditions
070      * @return self
071      */
072     default ScanBuilder<T> and(Consumer<RangeConditionCollector<T>> conditions) {
073         filter(ConditionalOperator.AND);
074         return filter(conditions);
075     }
076 
077     /**
078      * One or more filter conditions in disjunction.
079      *
080      * @param conditions consumer to build the conditions
081      * @return self
082      */
083     default ScanBuilder<T> or(Consumer<RangeConditionCollector<T>> conditions) {
084         filter(ConditionalOperator.OR);
085         return filter(conditions);
086     }
087 
088     /**
089      * One or more filter conditions.
090      *
091      * @param conditions closure to build the conditions
092      * @return self
093      */
094     default ScanBuilder<T> filter(
095         @DelegatesTo(type = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>", strategy = Closure.DELEGATE_FIRST)
096         @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>")
097             Closure<RangeConditionCollector<T>> conditions
098     ) {
099         return filter(ConsumerWithDelegate.create(conditions));
100     }
101 
102     /**
103      * One or more filter conditions in disjunction.
104      *
105      * @param conditions closure to build the conditions
106      * @return self
107      */
108     default ScanBuilder<T> or(
109         @DelegatesTo(type = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>", strategy = Closure.DELEGATE_FIRST)
110         @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>")
111             Closure<RangeConditionCollector<T>> conditions
112     ) {
113         return or(ConsumerWithDelegate.create(conditions));
114     }
115 
116     /**
117      * One or more filter conditions in conjunction.
118      *
119      * @param conditions closure to build the conditions
120      * @return self
121      */
122     default ScanBuilder<T> and(
123         @DelegatesTo(type = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>", strategy = Closure.DELEGATE_FIRST)
124         @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.aws.dynamodb.builder.RangeConditionCollector<T>")
125             Closure<RangeConditionCollector<T>> conditions
126     ) {
127         return and(ConsumerWithDelegate.create(conditions));
128     }
129 
130     /**
131      * Sets the conditional operator for the filter.
132      *
133      * Default is <code>and</code>
134      *
135      * @param or the conditional operator, usually <code>or</code> to switch to disjunction of filter conditions
136      * @return self
137      */
138     ScanBuilder<T> filter(ConditionalOperator or);
139 
140     /**
141      * Sets the desired pagination of the scans.
142      *
143      * This only sets the optimal pagination of the scans and does not limit the number of items returned.
144      *
145      * Use <code>{@link io.reactivex.Flowable#take(long)}</code> to limit the number results returned from the scan.
146      *
147      * @param page number of entities loaded by one scan request (not a number of total entities returned)
148      * @return self
149      */
150     ScanBuilder<T> page(int page);
151 
152     /**
153      * Sets the maximum number of items to be returned from the queries.
154      *
155      * This is a shortcut for calling <code>{@link io.reactivex.Flowable#take(long)}</code> on the result Flowable.
156      *
157      * @param max the maximum number of items returned
158      * @return self
159      */
160     ScanBuilder<T> limit(int max);
161 
162     /**
163      * Sets the scan offset by defining the exclusive start hash and range key (hash and range key of the last entity returned).
164      * @param exclusiveStartKeyValue exclusive start key hash value
165      * @param exclusiveRangeStartKey exclusive start key range value
166      * @return self
167      */
168     ScanBuilder<T> offset(Object exclusiveStartKeyValue, Object exclusiveRangeStartKey);
169 
170     /**
171      * Sets the scan offset by defining the exclusive start hash key (hash key of the last entity returned).
172      * @param exclusiveStartKeyValue exclusive start key hash value
173      * @return self
174      */
175     default ScanBuilder<T> offset(Object exclusiveStartKeyValue) {
176         return offset(exclusiveStartKeyValue, null);
177     }
178 
179     /**
180      * Configures the native scan expression.
181      *
182      * This method is an extension point which allows to configure properties which are not provides by this builder.
183      *
184      * @param configurer consumer to configure the native scan expression
185      * @return self
186      */
187     ScanBuilder<T> configure(Consumer<DynamoDBScanExpression> configurer);
188 
189     /**
190      * Configures the native scan expression.
191      *
192      * This method is an extension point which allows to configure properties which are not provides by this builder.
193      *
194      * @param configurer closure to configure the native scan expression
195      * @return self
196      */
197     default ScanBuilder<T> configure(
198         @DelegatesTo(type = "com.amazonaws.services.dynamodbv2.datamodeling.DynamoDBScanExpression", strategy = Closure.DELEGATE_FIRST)
199         @ClosureParams(value = FromString.class, options = "com.amazonaws.services.dynamodbv2.datamodeling.DynamoDBScanExpression")
200             Closure<Object> configurer
201     ) {
202         return configure(ConsumerWithDelegate.create(configurer));
203     }
204 
205     /**
206      * Limits which properties of the returned entities will be populated.
207      * @param propertyPaths property paths to be populated in the returned entities
208      * @return self
209      */
210     ScanBuilder<T> only(Iterable<String> propertyPaths);
211 
212     /**
213      * Limits which properties of the returned entities will be populated.
214      * @param propertyPaths property paths to be populated in the returned entities
215      * @return self
216      */
217     default ScanBuilder<T> only(String... propertyPaths) {
218         return only(Arrays.asList(propertyPaths));
219     }
220 
221     /**
222      * Limits which properties of the returned entities will be populated.
223      * @param collector closure to collect the property paths
224      * @return self
225      */
226     default ScanBuilder<T> only(
227         @DelegatesTo(type = "T", strategy = Closure.DELEGATE_ONLY)
228         @ClosureParams(value = FromString.class, options = "T")
229             Closure<Object> collector
230     ) {
231         return only(PathCollector.collectPaths(collector).getPropertyPaths());
232     }
