018 package com.agorapulse.micronaut.aws.dynamodb.builder;
019 
020 import com.agorapulse.micronaut.aws.dynamodb.DynamoDBMetadata;
021 import com.amazonaws.services.dynamodbv2.AmazonDynamoDB;
022 import com.amazonaws.services.dynamodbv2.datamodeling.DynamoDBMapperTableModel;
023 import com.amazonaws.services.dynamodbv2.datamodeling.IDynamoDBMapper;
024 import com.amazonaws.services.dynamodbv2.model.*;
025 
026 import java.util.ArrayList;
027 import java.util.List;
028 import java.util.Map;
029 import java.util.function.Consumer;
030 import java.util.function.Function;
031 
032 class DefaultUpdateBuilder<T> implements UpdateBuilder<T> {
033 
034     private static class Update {
035         final String name;
036         final AttributeAction action;
037         final Object value;
038 
039         private Update(String name, AttributeAction action, Object value) {
040             this.name = name;
041             this.action = action;
042             this.value = value;
043         }
044     }
045 
046     private final Class<T> itemType;
047     private final List<Update> updates = new ArrayList<>();
048 
049     private Object hash;
050     private Object range;
051 
052     private ReturnValue returnValue = ReturnValue.NONE;
053     private Function<T, ?> returnValueMapper = Function.identity();
054 
055     private Consumer<UpdateItemRequest> configurer = u -> {};
056 
057     DefaultUpdateBuilder(Class<T> itemType) {
058         this.itemType = itemType;
059     }
060 
061     @Override
062     public UpdateBuilder<T> hash(Object key) {
063         this.hash = key;
064         return this;
065     }
066 
067     @Override
068     public UpdateBuilder<T> range(Object range) {
069         this.range = range;
070         return this;
071     }
072 
073     @Override
074     public UpdateBuilder<T> add(String attributeName, Object delta) {
075         updates.add(new Update(attributeName, AttributeAction.ADD, delta));
076         return this;
077     }
078 
079     @Override
080     public UpdateBuilder<T> put(String attributeName, Object value) {
081         updates.add(new Update(attributeName, AttributeAction.PUT, value));
082         return this;
083     }
084 
085     @Override
086     public UpdateBuilder<T> delete(String attributeName) {
087         updates.add(new Update(attributeName, AttributeAction.DELETE, null));
088         return this;
089     }
090 
091     @Override
092     public Object update(IDynamoDBMapper mapper, AmazonDynamoDB client) {
093         UpdateItemRequest request = resolveExpression(mapper);
094         UpdateItemResult result = client.updateItem(request);
095         Map<String, AttributeValue> attributes = result.getAttributes();
096 
097         if (attributes == null || attributes.isEmpty()) {
098             return null;
099         }
100 
101         return returnValueMapper.apply(mapper.getTableModel(itemType).unconvert(attributes));
102     }
103 
104     @Override
105     public UpdateItemRequest resolveExpression(IDynamoDBMapper mapper) {
106         DynamoDBMapperTableModel<T> tableModel = mapper.getTableModel(itemType);
107         DynamoDBMetadata<T> metadata = DynamoDBMetadata.create(itemType);
108 
109         UpdateItemRequest request = new UpdateItemRequest()
110             .withTableName(metadata.getMainTable().tableName())
111             .withKey(tableModel.convertKey(hash, range))
112             .withReturnValues(returnValue);
113 
114         for (Update u : updates) {
115             request.addAttributeUpdatesEntry(
116                 u.name,
117                 new AttributeValueUpdate().withAction(u.action).withValue(tableModel.field(u.name).convert(u.value))
118             );
119         }
120 
121         configurer.accept(request);
122 
123         return request;
124     }
125 
126     @Override
127     public UpdateBuilder<T> configure(Consumer<UpdateItemRequest> configurer) {
128         this.configurer = configurer;
129         return this;
130     }
131 
132     @Override
133     public UpdateBuilder<T> returns(ReturnValue returnValue, Function<T, ?> mapper) {
134         this.returnValue = returnValue;
135         this.returnValueMapper = mapper;
136 
137         return this;
138     }
139 }
