069     static void nullifyHashSets(Object object) {
070         object.properties.each { String prop, val ->
071             if (object.hasProperty(prop)
072                 && object[prop] instanceof HashSet
073                 && object[prop]?.size() == 0) {
074                 try {
075                     // log.debug("Nullifying collection ${prop} before sending to DynamoDB")
076                     Method getter = object.getClass().getMethod(NameUtils.getterNameFor(prop))
077                     if (getter != null && getter.getAnnotation(DynamoDBIgnore) == null) {
078                         object[prop] = null
079                     }
080                 } catch (Exception e) {
081                     log.error "failed to nullify collection ${prop} of ${object} before sending to DynamoDB", e
082                 }
083             }
084         }

165         DynamoDBTable table = metadata.itemClass.getAnnotation(DynamoDBTable)

172             CreateTableRequest createTableRequest = mapper.generateCreateTableRequest(metadata.itemClass)
173             // new CreateTableRequest().withTableName(table.tableName())
174 
175             // ProvisionedThroughput

177                 .withReadCapacityUnits(readCapacityUnits)
178                 .withWriteCapacityUnits(writeCapacityUnits)
179 
180             // ProvisionedThroughput for GSIs
181             if (createTableRequest.globalSecondaryIndexes) {
182                 createTableRequest.globalSecondaryIndexes.each { GlobalSecondaryIndex globalSecondaryIndex ->
183                     globalSecondaryIndex.provisionedThroughput = new ProvisionedThroughput()
184                         .withReadCapacityUnits(DEFAULT_READ_CAPACTIY)
185                         .withWriteCapacityUnits(DEFAULT_WRITE_CAPACITY)
186                     log.info("Creating DynamoDB GSI: ${globalSecondaryIndex}")
187                 }
188             }
189 
190             log.info("Creating DynamoDB table: ${createTableRequest}")
191 
192             return client.createTable(createTableRequest)

296     @SuppressWarnings('DuplicateNumberLiteral')
297     int deleteAllByConditions(DynamoDBQueryExpression query, Map settings) {
298         QueryResultPage itemsPage = mapper.queryPage(metadata.itemClass, query)
299 
300         int deletedItemsCount = -1
301         Map lastEvaluatedKey = itemsPage.lastEvaluatedKey
302         while (lastEvaluatedKey || deletedItemsCount == -1) {
303             if (deletedItemsCount == -1) {
304                 deletedItemsCount = 0
305             } else {
306                 query.exclusiveStartKey = lastEvaluatedKey
307             }
308             itemsPage = mapper.queryPage(metadata.itemClass, query)
309             if (itemsPage.results) {
310                 log.debug "Deleting ${itemsPage.results.size()} items, class: ${metadata.itemClass}"
311                 deletedItemsCount = deletedItemsCount + itemsPage.results.size()
312                 // Delete all items
313                 deleteAll(itemsPage.results, settings)
314             }
315             lastEvaluatedKey = itemsPage.lastEvaluatedKey
316         }
317         log.debug "Successfully deleted ${deletedItemsCount} items"
318         deletedItemsCount
319     }

332     /**
333      * Retrieve batched items corresponding to a list of item IDs, in the same order.
334      * Example: items = twitterItemDBService.getAll(1, [1, 2]).
335      *
336      * @param hashKey Hash Key of the items to retrieve
337      * @param rangeKey Range keys of the items to retrieve
338      * @param settings only used for setting throttle/readCapacityUnit when getting large sets
339      * @return a list of DynamoDBItem
340      */
341     @SuppressWarnings('AssignCollectionUnique')
342     List<TItemClass> getAll(Object hashKey, List rangeKeys, Map settings) {
343         Map result = [:]
344         List objects = rangeKeys.unique(false).collect { it -> metadata.itemClass.newInstance((hashKeyName): hashKey, (rangeKeyName): it) }
345         if (settings.throttle) {
346             int resultCursor = 0
347             long readCapacityUnit = settings.readCapacityUnit
348             if (!readCapacityUnit) {
349                 DescribeTableResult tableResult = client.describeTable(metadata.mainTable.tableName())
350                 readCapacityUnit = tableResult?.table?.provisionedThroughput?.readCapacityUnits ?: 10
351             }
352             objects.collate(20).each { List batchObjects ->
353                 result += mapper.batchLoad(batchObjects)
354                 resultCursor++
355                 if (readCapacityUnit && resultCursor >= (readCapacityUnit * 0.8)) {
356                     resultCursor = 0
357                     sleep(ONE_SECOND)
358                 }
359             }
360         } else {
361             result = mapper.batchLoad(objects)
362         }
363         if (result[metadata.mainTable.tableName()]) {
364             List<TItemClass> unorderedItems = result[metadata.mainTable.tableName()] as List<TItemClass>
365             List<TItemClass> items = []
366 
367             // Build an item list ordered in the same manner as the list of IDs we've been passed
368             rangeKeys.each { rangeKey ->
369                 TItemClass matchingItem = unorderedItems.find { item ->
370                     item[rangeKeyName] == rangeKey
371                 }
372                 if (matchingItem) {
373                     items.add(matchingItem)
374                 }
375                 // Remove the matching item from the unordered list to reduce the number of loops in the find above
376                 unorderedItems.remove(matchingItem)
377             }
378             items
379         } else {
380             []
381         }
382     }

422     QueryResultPage<TItemClass> query(Object hashKey, String rangeKeyName, Object rangeKeyValue, ComparisonOperator operator, Map settings) {

425                 rangeKeyName += INDEX_NAME_SUFFIX

427             queryByConditions(hashKey, [:], settings, rangeKeyName)
428         } else {

431         }

434     /**
435      * Optional settings:
436      * - batchGetDisabled (only when secondary indexes are used, useful for count when all item attributes are not required)
437      * - consistentRead (default to false)
438      * - exclusiveStartKey a map with the rangeKey (ex: [id: 2555]), with optional indexRangeKey when using LSI (ex.: [id: 2555, totalCount: 45])
439      * - limit
440      * - returnAll disable paging to return all items, WARNING: can be expensive in terms of throughput (default to false)
441      * - scanIndexForward (default to false)
442      * - throttle insert sleeps during execution to avoid reaching provisioned read throughput (default to false)
443      *
444      * @param hashKey
445      * @param rangeKeyConditions
446      * @param settings
447      * @return
448      */
449     QueryResultPage<TItemClass> queryByConditions(Object hashKey, Map<String, Condition> rangeKeyConditions, Map settings, String indexName) {
450         DynamoDBQueryExpression query = buildQuery(hashKey, settings, rangeKeyConditions, indexName)
451 
452         long readCapacityUnit = 0
453         int resultCursor = 0
454 
455         if (!settings.returnAll) {
456             // Query page
457             return fetchResults(mapper.queryPage(metadata.itemClass, query), rangeKeyConditions, indexName, settings, hashKey, readCapacityUnit)
458         }
459 
460         // Get table read Throughput
461         if (settings.throttle) {
462             DescribeTableResult tableResult = client.describeTable(metadata.mainTable.tableName())
463             readCapacityUnit = tableResult?.table?.provisionedThroughput?.readCapacityUnits ?: 0
464         }
465 
466         // Query all
467         Map lastEvaluatedKey = [items: 'none']
468 
469         QueryResultPage resultPage = new QueryResultPage()
470         resultPage.results = []
471 
472         while (lastEvaluatedKey) {
473             QueryResultPage currentPage = mapper.queryPage(metadata.itemClass, query)
474             resultPage.results.addAll(currentPage.results)
475             lastEvaluatedKey = currentPage.lastEvaluatedKey
476             query.exclusiveStartKey = currentPage.lastEvaluatedKey
477             if (settings.throttle) {
478                 resultCursor++
479                 if (readCapacityUnit && resultCursor >= (readCapacityUnit * 0.5)) {
480                     resultCursor = 0
481                     sleep(ONE_SECOND)
482                 }
483             }
484         }
485 
486         return fetchResults(resultPage, rangeKeyConditions, indexName, settings, hashKey, readCapacityUnit)
487     }

489     /**
490      * Query by dates with 'after' and/or 'before' range value
491      * 1) After a certain date : [after: new Date()]
492      * 2) Before a certain date : [before: new Date()]
493      * 3) Between provided dates : [after: new Date() + 1, before: new Date()]
494      *
495      * Optional settings:
496      * - batchGetDisabled (only when secondary indexes are used, useful for count when all item attributes are not required)
497      * - consistentRead (default to false)
498      * - exclusiveStartKey a map with the rangeKey (ex: [id: 2555]), with optional indexRangeKey when using LSI (ex.: [id: 2555, totalCount: 45])

636      *
637      * @param params
638      * @param maxAfterDate
639      * @return
640      */
641     static protected Map addMaxAfterDateCondition(Map params,
642                                                   Date maxAfterDate) {
643         if (!params.containsKey(AFTER_KEY)) {
644             params[AFTER_KEY] = maxAfterDate
645         } else if (((Date) params[AFTER_KEY]).before(maxAfterDate)) {
646             params[AFTER_KEY] = maxAfterDate
647         }
648         if (params.containsKey(BEFORE_KEY) && ((Date) params[BEFORE_KEY]).before((Date) params[AFTER_KEY])) {
649             // Make sure that 'before' date is after 'after' date, to generate valid BETWEEN DynamoDB query (even if query will return nothing, it won't break)
650             params[BEFORE_KEY] = params[AFTER_KEY]
651         }
652         params
653     }

662         'CouldBeSwitchStatement',

690     /**
691      *
692      * @param rangeKeyName
693      * @param rangeKeyDates
694      * @param maxAfterDate
695      * @return
696      */
697     static protected Map buildDateConditions(String rangeKeyName,
698                                              Map<String, Date> rangeKeyDates,
699                                              Date maxAfterDate = null) {
700         assert rangeKeyDates.keySet().any { it in [AFTER_KEY, BEFORE_KEY] }
701         ComparisonOperator operator
702         List attributeValueList = []
703         if (maxAfterDate) {
704             addMaxAfterDateCondition(rangeKeyDates, maxAfterDate)
705         }
706         if (rangeKeyDates.containsKey(AFTER_KEY)) {
707             operator = ComparisonOperator.GE
708             attributeValueList << new AttributeValue().withS(serializeDate(rangeKeyDates[AFTER_KEY]))
709         }
710         if (rangeKeyDates.containsKey(BEFORE_KEY)) {
711             operator = ComparisonOperator.LE
712             attributeValueList << new AttributeValue().withS(serializeDate(rangeKeyDates[BEFORE_KEY]))
713         }
714         if (rangeKeyDates.containsKey(AFTER_KEY) && rangeKeyDates.containsKey(BEFORE_KEY)) {
715             operator = ComparisonOperator.BETWEEN
716         }
717         [
718             (rangeKeyName): new Condition()
719                 .withComparisonOperator(operator)
720                 .withAttributeValueList(attributeValueList),
721         ]
722     }
