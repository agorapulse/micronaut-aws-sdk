018 package com.agorapulse.micronaut.aws.util;
019 
020 import com.amazonaws.SdkClientException;
021 import com.amazonaws.auth.AWSCredentials;
022 import com.amazonaws.auth.AWSCredentialsProvider;
023 import com.amazonaws.auth.BasicAWSCredentials;
024 import com.amazonaws.auth.BasicSessionCredentials;
025 import com.amazonaws.util.StringUtils;
026 import io.micronaut.context.env.Environment;
027 
028 /**
029  * A {@link AWSCredentialsProvider} that reads from the {@link Environment}.
030  */
031 public class EnvironmentAWSCredentialsProvider implements AWSCredentialsProvider {
032 
033     /**
034      * Environment variable name for the AWS access key ID.
035      */
036     public static final String ACCESS_KEY_ENV_VAR = "aws.access-key-id";
037 
038     /**
039      * Alternate environment variable name for the AWS access key ID.
040      */
041     public static final String ALTERNATE_ACCESS_KEY_ENV_VAR = "aws.access-key";
042 
043     /**
044      * Environment variable name for the AWS secret key.
045      */
046     public static final String SECRET_KEY_ENV_VAR = "aws.secret-key";
047 
048     /**
049      * Alternate environment variable name for the AWS secret key.
050      */
051     public static final String ALTERNATE_SECRET_KEY_ENV_VAR = "aws.secret-access-key";
052 
053     /**
054      * Environment variable name for the AWS session token.
055      */
056     public static final String AWS_SESSION_TOKEN_ENV_VAR = "aws.session-token";
057 
058     private final Environment environment;
059 
060     /**
061      * Constructor.
062      * @param environment environment
063      */
064     public EnvironmentAWSCredentialsProvider(Environment environment) {
065         this.environment = environment;
066     }
067 
068     @Override
069     public AWSCredentials getCredentials() {
070         String accessKey = environment.getProperty(ACCESS_KEY_ENV_VAR, String.class, environment.getProperty(ALTERNATE_ACCESS_KEY_ENV_VAR, String.class, (String) null));
071 
072         String secretKey = environment.getProperty(SECRET_KEY_ENV_VAR, String.class, environment.getProperty(ALTERNATE_SECRET_KEY_ENV_VAR, String.class, (String) null));
073         accessKey = StringUtils.trim(accessKey);
074         secretKey = StringUtils.trim(secretKey);
075         String sessionToken = StringUtils.trim(environment.getProperty(AWS_SESSION_TOKEN_ENV_VAR, String.class, (String) null));
076 
077         if (StringUtils.isNullOrEmpty(accessKey) || StringUtils.isNullOrEmpty(secretKey)) {
078             throw new SdkClientException(
079                 "Unable to load AWS credentials from environment "
080                     + "(" + ACCESS_KEY_ENV_VAR + " (or " + ALTERNATE_ACCESS_KEY_ENV_VAR + ") and "
081                     + SECRET_KEY_ENV_VAR + " (or " + ALTERNATE_SECRET_KEY_ENV_VAR + "))");
082         }
083 
084         return sessionToken == null
085             ?
086             new BasicAWSCredentials(accessKey, secretKey)
087             :
088             new BasicSessionCredentials(accessKey, secretKey, sessionToken);
089     }
090 
091     @Override
092     public void refresh() {
093 
094     }
095 
096     @Override
097     public String toString() {
098         return getClass().getSimpleName();
099     }
100 }
